' PUNTO DE ENTRADA - INICIALIZADOR MAESTRO
Public Sub InicializarSistema()
    Dim respuesta As VbMsgBoxResult
    Dim reinicioTotal As Boolean

    respuesta = MsgBox("¿Qué tipo de reinicio desea realizar?" & vbCrLf & vbCrLf & _
                      "SI = REINICIO TOTAL (Borra todos los datos y hojas)." & vbCrLf & _
                      "NO = REINICIO DE SISTEMA (Mantiene datos, repara formatos).", _
                      vbYesNoCancel + vbExclamation, "Inicialización del Sistema")

    If respuesta = vbCancel Then Exit Sub

    reinicioTotal = (respuesta = vbYes)

    Call InvalidarCache

    Application.ScreenUpdating = False

    Call CrearHojas(reinicioTotal)
    Call CrearHeaders                    ' ? AHORA TODOS LOS HEADERS EN FILA 3
    Call ConfigurarProtecciones
    Call CargarParametrosIniciales
    Call CrearDashboardInicio
    Call CrearHojaManual
    Call CrearBotonesAsignaciones
    Call CrearBotonesMaestros
    Call ActualizarCalendarioNoLaboral
    Call CrearBotonesVolver             ' ? AHORA INCLUYE REPORTES fila 2

    Application.ScreenUpdating = True

    MsgBox "OK - SISTEMA INICIALIZADO" & vbCrLf & _
           "Modo: " & IIf(reinicioTotal, "Total (Fábrica)", "Sistema (Conservación)") & vbCrLf & _
           "Calendario actualizado.", vbInformation
End Sub

' CREACIÓN DE HOJAS (SIN CAMBIOS)
Public Sub CrearHojas(Optional reiniciarFabrica As Boolean = False)
    Dim ws As Worksheet
    Dim nombresHojas As Variant
    Dim i As Long

    nombresHojas = Array("INICIO", "MANUAL", "ASIGNACIONES_DIARIAS", "OPERARIAS", "REFERENCIAS", _
                        "COLORES", "TALLAS", "TAREAS", "LOTES", "EFICIENCIAOPERARIAS", _
                        "DIASNOLABORALES", "AVANCEPRODUCCION", "REPORTELOTES", "INFORMEPRODUCCION", _
                        "RESUMENDIARIO", "PRODUCCIONFINALIZADA", "ANALISISTIEMPOS", "ARCHIVOASIGNACIONES", "PARAMETROS")

    If reiniciarFabrica Then
        Application.DisplayAlerts = False
        On Error Resume Next
        For i = LBound(nombresHojas) To UBound(nombresHojas)
            If ExisteHoja(CStr(nombresHojas(i))) Then
                If ThisWorkbook.Sheets.Count > 1 Then
                    ThisWorkbook.Sheets(nombresHojas(i)).Delete
                End If
            End If
        Next i
        On Error GoTo 0
        Application.DisplayAlerts = True
    End If

    For i = LBound(nombresHojas) To UBound(nombresHojas)
        If Not ExisteHoja(CStr(nombresHojas(i))) Then
            Set ws = ThisWorkbook.Sheets.Add(After:=ThisWorkbook.Sheets(ThisWorkbook.Sheets.Count))
            ws.Name = nombresHojas(i)
        Else
            Set ws = ThisWorkbook.Sheets(nombresHojas(i))
        End If

        ' Colores de pestañas
        Select Case nombresHojas(i)
            Case "INICIO": ws.Tab.color = RGB(0, 176, 240)
            Case "MANUAL": ws.Tab.color = RGB(255, 192, 0)
            Case "OPERARIAS", "REFERENCIAS", "COLORES", "TALLAS", "TAREAS", "DIASNOLABORALES": ws.Tab.color = RGB(0, 112, 192)
            Case "LOTES", "ASIGNACIONES_DIARIAS", "EFICIENCIAOPERARIAS": ws.Tab.color = RGB(192, 0, 0)
            Case "RESUMENDIARIO", "AVANCEPRODUCCION", "REPORTELOTES", "INFORMEPRODUCCION", "PRODUCCIONFINALIZADA", "ANALISISTIEMPOS": ws.Tab.color = RGB(0, 176, 80)
            Case "PARAMETROS", "ARCHIVOASIGNACIONES": ws.Tab.color = RGB(128, 128, 128)
        End Select
    Next i
End Sub

Public Function ExisteHoja(nombre As String) As Boolean
    On Error Resume Next
    ExisteHoja = Not ThisWorkbook.Sheets(nombre) Is Nothing
    On Error GoTo 0
End Function

' ? HEADERS UNIFICADOS - TODAS LAS HOJAS HEADER EN FILA 3

    ws.Range("A3:E3").Value = Array("IDReferencia", "NombreReferencia", "Descripcion", "Color", "TiempoTotalUnidadSeg")

    ' 2.1 COLORES - Fila 3 ?
    Set ws = ThisWorkbook.Sheets("COLORES")
    ws.Range("A3:B3").Value = Array("IDReferencia", "Color")

    ' 3. TALLAS - Fila 3 ?
    Set ws = ThisWorkbook.Sheets("TALLAS")
    ws.Range("A3:B3").Value = Array("IDReferencia", "Talla")

    ' 4. LOTES - Fila 3 ?
    Set ws = ThisWorkbook.Sheets("LOTES")
    ws.Range("A3:K3").Value = Array("IDLote", "IDReferencia", "Color", "Talla", "CantidadRecibida", _
                                    "FechaRecepcion", "FechaInicioProduccion", "UnidadesCompletadas", _
                                    "PorcentajeAvance", "Estado", "FechaFinEstimada")

    ' 5. TAREAS - Fila 3 ?
    Set ws = ThisWorkbook.Sheets("TAREAS")
    ws.Range("A3:F3").Value = Array("IDReferencia", "NumeroTarea", "NombreTarea", "TiempoSegundos", "DescripcionTarea", "Dificultad")

    ' 6. ASIGNACIONES_DIARIAS - Fila 3 ?
    Set ws = ThisWorkbook.Sheets("ASIGNACIONES_DIARIAS")
    ws.Range("A3:U3").Value = Array("Fecha", "IDOperaria", "NombreOperaria", "IDReferencia", "NombreReferencia", _
                                    "IDLote", "Color", "TallaAsignada", "NumeroTarea", "NombreTarea", _
                                    "CantidadUnidadesAsignadas", "TiempoUnitarioSegundos", "TiempoTotalAsignadoSegundos", _
                                    "TiempoTotalAsignadoMinutos", "EstadoAsignacion", "UnidadesCompletadasHoy", _
                                    "Observaciones", "TiempoLibreOperariaMinutos", "FechaEntregaReal", _
                                    "TiempoRealMinutos", "DesviacionMinutos")

    ' ? 7. REPORTES - AHORA HEADER EN FILA 3 (MOVIDO DESDE FILA 1)
    Set ws = ThisWorkbook.Sheets("PRODUCCIONFINALIZADA")
    ws.Range("A3:G3").Value = Array("FechaFinalizacion", "IDReferencia", "NombreReferencia", "Talla", _
                                    "TotalUnidadesProducidas", "DiasProduccion", "OperariasInvolucradas")

    Set ws = ThisWorkbook.Sheets("DIASNOLABORALES")
    ws.Range("A3:C3").Value = Array("Fecha", "DiaSemana", "Descripcion")

    Set ws = ThisWorkbook.Sheets("AVANCEPRODUCCION")
    ws.Range("A3:I3").Value = Array("IDReferencia", "NombreReferencia", "NombreTarea", "TiempoUnitSeg", _
                                    "TotalUnidadesReq", "UnidadesAsignadas", "UnidadesPendientes", _
                                    "TiempoPendienteMin", "EstadoAsignacion")

    Set ws = ThisWorkbook.Sheets("ANALISISTIEMPOS")
    ws.Range("A3:H3").Value = Array("IDReferencia", "NombreReferencia", "Tarea", "TiempoEstandarUnitseg", _
                                    "TiempoRealPromedioseg", "Desviacion", "Muestras", "Sugerencia")

    Set ws = ThisWorkbook.Sheets("REPORTELOTES")
    ws.Range("A3:K3").Value = Array("Lote", "Referencia", "Color", "Talla", "Cantidad", "TareasUnidad", _
                                    "Ops.Requeridas", "Ops.Completadas", "Avance", "Tiempo Est.Horas", "Estado")

    Set ws = ThisWorkbook.Sheets("RESUMENDIARIO")
    ws.Range("A3:F3").Value = Array("Operaria", "Capacidad min", "Asignado min", "Ocupacion %", "Disponible min", "Estado")

    Set ws = ThisWorkbook.Sheets("PARAMETROS")
    ws.Range("A3:D3").Value = Array("Parametro", "ValorActual", "Descripcion", "Tipo")

    ' Aplicar formato general a TODOS los headers (fila 3)
    Dim rngHeader As Range, ultCol As Long
    For Each ws In ThisWorkbook.Worksheets
        Set rngHeader = Nothing
        On Error Resume Next
        ultCol = ws.Cells(3, ws.Columns.Count).End(xlToLeft).Column
        If ultCol >= 1 Then
            Set rngHeader = ws.Range(ws.Cells(3, 1), ws.Cells(3, ultCol))
        End If
        On Error GoTo 0

        If Not rngHeader Is Nothing Then
            With rngHeader
                .Font.Bold = True
                .Interior.color = RGB(220, 220, 220)
                .Borders.LineStyle = xlContinuous
                .EntireColumn.AutoFit
            End With
        End If
    Next ws
End Sub

Private Sub CrearBotonDashboard(ws As Worksheet, texto As String, macro As String, fila As Integer, col As Integer, color As Long)
    Dim btn As Shape
    Set btn = ws.Shapes.AddShape(msoShapeRoundedRectangle, _
                                 ws.Cells(fila, col).Left, ws.Cells(fila, col).Top, 180, 35)
    With btn
        .Placement = xlMove
        .TextFrame.Characters.Text = texto
        .TextFrame.HorizontalAlignment = xlHAlignCenter
        .TextFrame.VerticalAlignment = xlVAlignCenter
        .Fill.ForeColor.RGB = color
        .Line.visible = msoFalse
        .OnAction = macro
    End With
End Sub

' BOTONES ESPECÍFICOS (SIN CAMBIOS)
Public Sub CrearBotonesAsignaciones()
    Dim ws As Worksheet, shp As Shape
    Set ws = ThisWorkbook.Sheets("ASIGNACIONES_DIARIAS")

    For Each shp In ws.Shapes
        If shp.Type = msoAutoShape And shp.AutoShapeType = msoShapeRoundedRectangle Then shp.Delete
    Next shp

    ws.Rows(1).RowHeight = 15
    ws.Rows(2).RowHeight = 45  ' ? Fila 2 para botones

    Call CrearBotonDashboard(ws, "NUEVA ASIGNACION", "CrearNuevaAsignacionUI", 2, 2, RGB(0, 122, 204))
    Call CrearBotonDashboard(ws, "EDITAR ASIGNACION", "RegistrarFilaEnFormulario", 2, 6, RGB(255, 140, 0))
End Sub

Public Sub CrearBotonesMaestros()
    Dim hojas As Variant, hoja As Variant
    Dim ws As Worksheet, macroNuevo As String, shp As Shape

    hojas = Array("OPERARIAS", "REFERENCIAS", "COLORES", "TALLAS", "LOTES", "TAREAS")

    For Each hoja In hojas
        Set ws = ThisWorkbook.Sheets(hoja)

        For Each shp In ws.Shapes
            If shp.Type = msoAutoShape And shp.AutoShapeType = msoShapeRoundedRectangle Then shp.Delete
        Next shp

        ws.Cells.ColumnWidth = 6
        ws.Cells.RowHeight = 20
        ws.Columns("B").ColumnWidth = 30
        ws.Rows(1).RowHeight = 15
        ws.Rows(2).RowHeight = 45  ' ? Fila 2 para botones

        Select Case hoja
            Case "OPERARIAS": macroNuevo = "AbrirGestorOperarias"
            Case "REFERENCIAS": macroNuevo = "AbrirGestorReferencias"
            Case "COLORES": macroNuevo = "AbrirGestorColores"
            Case "TALLAS": macroNuevo = "AbrirGestorTallas"
            Case "LOTES": macroNuevo = "AbrirGestorLotes"
            Case "TAREAS": macroNuevo = "AbrirGestorTareas"
        End Select

        Call CrearBotonDashboard(ws, "NUEVO", macroNuevo, 2, 2, RGB(0, 122, 204))
        Call CrearBotonDashboard(ws, "EDITAR", "EditarMaestroSeleccionado", 2, 5, RGB(255, 140, 0))
    Next hoja
End Sub

' ? BOTONES VOLVER CORREGIDO - INCLUYE REPORTES EN FILA 2
Public Sub CrearBotonesVolver()
    Dim ws As Worksheet
    Dim colBtn As Integer, filaBtn As Integer
    Dim i As Long, txt As String

    For Each ws In ThisWorkbook.Worksheets
        If ws.Name <> "INICIO" Then
            ' Eliminar botones VOLVER existentes
            For i = ws.Shapes.Count To 1 Step -1
                If ws.Shapes(i).Type = msoAutoShape And ws.Shapes(i).AutoShapeType = msoShapeRoundedRectangle Then
                    txt = "": On Error Resume Next: txt = ws.Shapes(i).TextFrame.Characters.Text: On Error GoTo 0
                    If Trim(txt) = "VOLVER A INICIO" Then ws.Shapes(i).Delete
                End If
            Next i

            ' ? ESTRUCTURA UNIFICADA - TODAS las hojas tienen botones en FILA 2
            filaBtn = 2
            colBtn = 10

            ' Ajustes específicos por hoja (columna)
            Select Case ws.Name
                Case "ASIGNACIONES_DIARIAS", "RESUMENDIARIO", "AVANCEPRODUCCION", "REPORTELOTES", _
                     "PRODUCCIONFINALIZADA", "ANALISISTIEMPOS", "EFICIENCIAOPERARIAS", "INFORMEPRODUCCION"
                    colBtn = 10  ' Derecha estándar para reportes
                Case "OPERARIAS", "REFERENCIAS", "COLORES", "TALLAS", "LOTES", "TAREAS"
                    colBtn = 12  ' Más a la derecha para maestros
                Case "MANUAL"
                    filaBtn = 1: colBtn = 5  ' Especial manual
                Case "DIASNOLABORALES", "PARAMETROS"
                    colBtn = 6   ' Izquierda para admin
            End Select

            Call CrearBotonDashboard(ws, "VOLVER A INICIO", "IrAInicio", filaBtn, colBtn, RGB(105, 105, 105))
        End If
    Next ws
End Sub

Public Sub PopulateRefSearchCombo()
    Dim wsInicio As Worksheet, cbo As OLEObject
    Dim refs As Collection, ref As Variant

    On Error Resume Next
    Set wsInicio = ThisWorkbook.Sheets("INICIO")
    Set cbo = wsInicio.OLEObjects("cboBuscarRef")
    If cbo Is Nothing Then Exit Sub
    On Error GoTo 0

    cbo.Object.Clear
    If Not EsCacheActualizado Then CargarCacheEnMemoria

    Set refs = ObtenerAllReferencias
    For Each ref In refs: cbo.Object.AddItem ref: Next ref
End Sub
