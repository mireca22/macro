Private Sub UserFormInitialize()
    Dim i As Integer
    Dim wsRefs As Worksheet
    Dim wsLotes As Worksheet

    lstColores.Visible = False
    btnAgregarColor.Visible = False
    btnQuitarColor.Visible = False

    btnQuitarTalla.Visible = False
    lblTareas.Visible = False

    btnQuitarTarea.Visible = False

        Me.Caption = "EDITAR " & TipoMaestroActual
    Else

    End If
    
    For i = 1 To 7

            ConfigurarCampo 2, "Nombre Completo", False
            ConfigurarCampo 3, "Tiempo Disp. Semanal min", False

            ConfigurarCampo 5, "Especialidad", False

            ConfigurarCampo 2, "Nombre Referencia", False
            ConfigurarCampo 3, "Descripción", False

            txt4.Visible = False

            cboRef.Visible = True
            cboRef.Top = txt1.Top
            cboRef.Left = txt1.Left
            cboRef.Width = txt1.Width

            If ModoEdicion Then
                cboRef.Value = ActiveSheet.Cells(FilaEdicion, 1).Value & " - " & ObtenerNombreReferencia(ActiveSheet.Cells(FilaEdicion, 1).Value)

            cboRef.Visible = True
            cboRef.Top = txt1.Top
            cboRef.Left = txt1.Left
            cboRef.Width = txt1.Width
            CargarReferenciasCombo
            
            If ModoEdicion Then

            cboTalla.Top = txt3.Top

Private Sub CargarDetallesReferencia(ByVal idRef As String)
    Dim colores As Collection
    Dim tallas As Collection
    Dim c As Variant
    Dim t As Variant
    Dim wsTareas As Worksheet
    Dim i As Long
    Dim ultFila As Long
    
    If Len(Trim$(idRef)) = 0 Then Exit Sub
    
    lstColores.Clear
    Set colores = ObtenerColoresDeReferencia(idRef)
    For Each c In colores
        lstColores.AddItem c
    Next c
    
    lstTallas.Clear
    Set tallas = ObtenerTallasDeReferencia(idRef)
    For Each t In tallas
        lstTallas.AddItem t
    Next t
    
    lstTareas.Clear
    lstTareas.ColumnCount = 3
    lstTareas.ColumnWidths = "30;150;50"
    
    Set wsTareas = ThisWorkbook.Sheets("TAREAS")
    ultFila = wsTareas.Cells(wsTareas.Rows.Count, 1).End(xlUp).Row
    
    For i = 4 To ultFila
        If CStr(wsTareas.Cells(i, 1).Value) = idRef Then
            lstTareas.AddItem wsTareas.Cells(i, 2).Value
            lstTareas.List(lstTareas.ListCount - 1, 1) = wsTareas.Cells(i, 3).Value
            lstTareas.List(lstTareas.ListCount - 1, 2) = wsTareas.Cells(i, 4).Value
        End If
    Next i
End Sub

Private Sub lstColores_DblClick(ByVal Cancel As MSForms.ReturnBoolean)
    Dim idRef As String
    Dim colorViejo As String
    Dim colorNuevo As String
    Dim ws As Worksheet
    Dim i As Long
    
    If lstColores.ListIndex = -1 Then Exit Sub
    
    idRef = txt1.Value
    colorViejo = lstColores.Value
    
    If TieneHijosColor(idRef, colorViejo) Then
        MsgBox "No se puede editar el color " & colorViejo & " porque ya está siendo usado en Lotes o Asignaciones.", vbCritical, "Acción Bloqueada"
        Exit Sub
    End If
    
    colorNuevo = InputBox("Editar color", "Editar Color", colorViejo)
    If Len(Trim$(colorNuevo)) = 0 Then Exit Sub
    If colorNuevo = colorViejo Then Exit Sub
    
    Set ws = ThisWorkbook.Sheets("COLORES")
    For i = ws.Cells(ws.Rows.Count, 1).End(xlUp).Row To 4 Step -1
        If ws.Cells(i, 1).Value = idRef And ws.Cells(i, 2).Value = colorViejo Then
            ws.Cells(i, 2).Value = colorNuevo
            Exit For
        End If
    Next i
    
    CargarCacheEnMemoria
    CargarDetallesReferencia idRef
End Sub

Private Sub lstTareas_DblClick(ByVal Cancel As MSForms.ReturnBoolean)
    Dim idRef As String
    Dim numTarea As String
    Dim wsTareas As Worksheet
    Dim i As Long
    Dim ultFila As Long
    Dim filaTarea As Long
    Dim frmTaskEditor As New frmMaestros
    
    If lstTareas.ListIndex = -1 Then Exit Sub
    
    idRef = txt1.Value
    numTarea = lstTareas.Value
    
    If MsgBox("¿Desea editar los detalles de la tarea N° " & numTarea & "?", vbYesNo + vbQuestion, "Editar Tarea") = vbNo Then Exit Sub
    
    Set wsTareas = ThisWorkbook.Sheets("TAREAS")
    ultFila = wsTareas.Cells(wsTareas.Rows.Count, 1).End(xlUp).Row
    filaTarea = 0
    
    For i = 4 To ultFila
        If wsTareas.Cells(i, 1).Value = idRef And CStr(wsTareas.Cells(i, 2).Value) = numTarea Then
            filaTarea = i
            Exit For
        End If
    Next i
    
    If filaTarea > 0 Then
        FilaEdicion = filaTarea
        ModoEdicion = True
        TipoMaestroActual = "TAREAS"
        Me.Hide
        frmTaskEditor.Show
        Me.Show
        CargarDetallesReferencia txt1.Value
    Else
        MsgBox "No se pudo encontrar la tarea para editar. Por favor, recargue el caché.", vbExclamation
    End If
End Sub

Private Sub CargarReferenciasCombo()
    Dim refs As Collection
    Dim ref As Variant
    
    Set refs = ObtenerAllReferencias()
    cboRef.Clear
    
    For Each ref In refs
        cboRef.AddItem ref
    Next ref
End Sub

Private Sub cboRef_Change()
    cboRefChange
End Sub

Private Sub cboRefChange()
    Dim tallas As Collection
    Dim colores As Collection
    Dim t As Variant
    Dim c As Variant
    
    If UCase$(TipoMaestroActual) = "TAREAS" Then
        If Len(Trim$(cboRef.Value)) > 0 And Not ModoEdicion Then
            txt2.Value = ObtenerSiguienteNumeroTarea(cboRef.Value)
        End If
    End If
    
    If UCase$(TipoMaestroActual) = "LOTES" Then
        If Len(Trim$(cboRef.Value)) > 0 Then
            Set tallas = ObtenerTallasDeReferencia(cboRef.Value)
            cboTalla.Clear
            For Each t In tallas
                cboTalla.AddItem t
            Next t
            
            Set colores = ObtenerColoresDeReferencia(cboRef.Value)
            cboColor.Clear
            For Each c In colores
                cboColor.AddItem c
            Next c
        End If
    End If
    
    ' Actualizar ID automáticamente para el contexto de Lotes
    ActualizarIDLoteAutomatico
End Sub

Private Sub cboColor_Change()
    ActualizarIDLoteAutomatico
End Sub

Private Sub cboTalla_Change()
    ActualizarIDLoteAutomatico
End Sub

'==============================================================================
' SUB: ActualizarIDLoteAutomatico
' Propósito: Refresca el campo ID_Lote basado en los inputs actuales (Ref, Col, Tal).
'==============================================================================
Private Sub ActualizarIDLoteAutomatico()
    If UCase$(TipoMaestroActual) = "LOTES" And Not ModoEdicion Then
        txt1.Value = GenerarIDLoteTextil(cboRef.Value, cboColor.Value, cboTalla.Value)
    End If
End Sub

Private Sub btnGuardar_Click()
    Dim datos(0 To 7) As Variant
    
    Dim rid As String: rid = cboRef.Value
    If InStr(rid, " - ") > 0 Then rid = Trim(Split(rid, " - ")(0))
    
    If UCase$(TipoMaestroActual) = "TALLAS" Or UCase$(TipoMaestroActual) = "TAREAS" Or UCase$(TipoMaestroActual) = "LOTES" Or UCase$(TipoMaestroActual) = "COLORES" Then
        If Len(Trim$(rid)) = 0 Then
            MsgBox "X Seleccione una referencia", vbCritical
            Exit Sub
        End If
        datos(0) = rid
    Else
        If Len(Trim$(txt1.Value)) = 0 Then
            MsgBox "X Campo ID obligatorio", vbCritical
            Exit Sub
        End If
        datos(0) = txt1.Value
    End If
    
    If UCase$(TipoMaestroActual) = "LOTES" Then
        datos(0) = txt1.Value
        datos(1) = rid
        datos(2) = cboColor.Value
        datos(3) = cboTalla.Value
        datos(4) = txt4.Value
        datos(5) = txt5.Value
        datos(6) = txt6.Value
        datos(7) = txt7.Value
    ElseIf UCase$(TipoMaestroActual) = "TAREAS" Then
        datos(1) = txt2.Value
        datos(2) = txt3.Value
        datos(3) = txt4.Value
        datos(4) = txt5.Value
        datos(5) = txt6.Value
        datos(6) = "Media"
    Else
        datos(1) = txt2.Value
        datos(2) = txt3.Value
        datos(3) = txt4.Value
        datos(4) = txt5.Value
        datos(5) = txt6.Value
    End If
    
    If UCase$(TipoMaestroActual) = "OPERARIAS" And Len(Trim$(txt2.Value)) = 0 Then
        MsgBox "X Nombre obligatorio", vbCritical
        Exit Sub
    End If
    
    If UCase$(TipoMaestroActual) = "REFERENCIAS" And Len(Trim$(txt2.Value)) = 0 Then
        MsgBox "X Nombre obligatorio", vbCritical
        Exit Sub
    End If
    
    If UCase$(TipoMaestroActual) = "COLORES" And Len(Trim$(txt2.Value)) = 0 Then
        MsgBox "X Color obligatorio", vbCritical
        Exit Sub
    End If
    
    If UCase$(TipoMaestroActual) = "TALLAS" And Len(Trim$(txt2.Value)) = 0 Then
        MsgBox "X Talla obligatoria", vbCritical
        Exit Sub
    End If
    
    If UCase$(TipoMaestroActual) = "LOTES" Then
        If Len(Trim$(txt1.Value)) = 0 Or Len(Trim$(cboColor.Value)) = 0 Or Len(Trim$(cboTalla.Value)) = 0 Or Len(Trim$(txt4.Value)) = 0 Then
            MsgBox "X Lote, Color, Talla y Cantidad obligatorias", vbCritical
            Exit Sub
        End If
        
        If Not IsDate(txt5.Value) Or Not IsDate(txt6.Value) Then
            MsgBox "X Fechas inválidas", vbCritical
            Exit Sub
        End If
    End If
    
    If UCase$(TipoMaestroActual) = "TAREAS" Then
        If Len(Trim$(txt3.Value)) = 0 Or Len(Trim$(txt4.Value)) = 0 Then
            MsgBox "X Nombre y Tiempo obligatorios", vbCritical
            Exit Sub
        End If
        
        If Not IsNumeric(txt4.Value) Then
            MsgBox "X El tiempo debe ser numérico", vbCritical
            Exit Sub
        End If
        
        If Not IsDate(txt5.Value) Then
            MsgBox "X La fecha de la tarea es inválida", vbCritical
            Exit Sub
        End If
    End If
    
    GuardarRegistroMaestro datos
    
    If UCase$(TipoMaestroActual) = "REFERENCIAS" And Not ModoEdicion Then
        MsgBox "Referencia creada. Ahora puede agregarle Colores, Tallas y Tareas.", vbInformation
        ModoEdicion = True
        UserFormInitialize
    ElseIf MsgBox("¿Desea agregar otro registro?", vbYesNo + vbQuestion) = vbYes Then
        UserFormInitialize
    Else
        Unload Me
    End If
End Sub

Private Sub btnCancelar_Click()
    Unload Me
End Sub
