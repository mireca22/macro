Sub InicializarSistema()
    '═══════════════════════════════════════════════════════════════════════════
    ' Inicializador Maestro del Sistema
    '═══════════════════════════════════════════════════════════════════════════
    
    Dim respuesta As VbMsgBoxResult
    Dim reinicioTotal As Boolean
    
    respuesta = MsgBox("¿Qué tipo de reinicio desea realizar?" & vbCrLf & vbCrLf & _
                       "• SÍ: REINICIO TOTAL (Borra todos los datos y hojas)." & vbCrLf & _
                       "• NO: REINICIO DE SISTEMA (Mantiene datos, repara formatos).", _
                       vbYesNoCancel + vbExclamation, "Inicialización del Sistema")
    
    If respuesta = vbCancel Then Exit Sub
    reinicioTotal = (respuesta = vbYes)
    
    'Invalidar el caché para forzar una recarga completa desde las hojas de Excel.
    Call InvalidarCache
    
    Application.ScreenUpdating = False
    
    Call CrearHojas(reinicioTotal)
    Call CrearHeaders
    Call ConfigurarProtecciones
    Call CargarParametrosIniciales
    Call CrearDashboardInicio
    Call CrearHojaManual
    Call CrearBotonesAsignaciones
    Call CrearBotonesMaestros
    Call ActualizarCalendarioNoLaboral
    Call CrearBotonesVolver
    
    Application.ScreenUpdating = True
    
    MsgBox "[OK] SISTEMA INICIALIZADO" & vbCrLf & _
           "Modo: " & IIf(reinicioTotal, "Total (Fábrica)", "Sistema (Conservación)") & vbCrLf & _
           "Calendario actualizado.", vbInformation
End Sub

Sub CrearHojas(Optional reiniciarFabrica As Boolean = False)
    Dim ws As Worksheet
    Dim nombresHojas As Variant
    Dim i As Long
    
    nombresHojas = Array("INICIO", "MANUAL", "ASIGNACIONES_DIARIAS", "OPERARIAS", "REFERENCIAS", "COLORES", "TALLAS", "TAREAS", "LOTES", "EFICIENCIA_OPERARIAS", _
                         "DIAS_NO_LABORALES", "AVANCE_PRODUCCION", "REPORTE_LOTES", "INFORME_PRODUCCION", "RESUMEN_DIARIO", _
                         "PRODUCCION_FINALIZADA", "ANALISIS_TIEMPOS", "ARCHIVO_ASIGNACIONES", "PARAMETROS")
    
    If reiniciarFabrica Then
        Application.DisplayAlerts = False
        On Error Resume Next
        For i = LBound(nombresHojas) To UBound(nombresHojas)
            If ExisteHoja(CStr(nombresHojas(i))) Then
                If ThisWorkbook.Sheets.Count = 1 Then ThisWorkbook.Sheets.Add
                ThisWorkbook.Sheets(nombresHojas(i)).Delete
            End If
        Next i
        On Error GoTo 0
        Application.DisplayAlerts = True
    End If
    
    For i = LBound(nombresHojas) To UBound(nombresHojas)
        If Not ExisteHoja(CStr(nombresHojas(i))) Then
            Set ws = ThisWorkbook.Sheets.Add(After:=ThisWorkbook.Sheets(ThisWorkbook.Sheets.Count))
            ws.Name = nombresHojas(i)
        Else
            Set ws = ThisWorkbook.Sheets(nombresHojas(i))
        End If
        
        'Colores de pestañas
        Select Case nombresHojas(i)
            Case "INICIO": ws.Tab.Color = RGB(0, 176, 240)       'Principal (Cyan)
            Case "MANUAL": ws.Tab.Color = RGB(255, 192, 0)       'Ayuda (Amarillo)
            Case "OPERARIAS", "REFERENCIAS", "COLORES", "TALLAS", "TAREAS", "DIAS_NO_LABORALES": ws.Tab.Color = RGB(0, 112, 192) 'Maestros (Azul)
            Case "LOTES", "ASIGNACIONES_DIARIAS", "EFICIENCIA_OPERARIAS": ws.Tab.Color = RGB(192, 0, 0) 'Procesos (Rojo)
            Case "RESUMEN_DIARIO", "AVANCE_PRODUCCION", "REPORTE_LOTES", "INFORME_PRODUCCION", "PRODUCCION_FINALIZADA", "ANALISIS_TIEMPOS": ws.Tab.Color = RGB(0, 176, 80) 'Reportes (Verde)
            Case "PARAMETROS", "ARCHIVO_ASIGNACIONES": ws.Tab.Color = RGB(128, 128, 128) 'Sistema (Gris)
        End Select
    Next i
End Sub

    ws.Range("A1:C1").Value = Array("Fecha", "Día_Semana", "Descripción")

    Set ws = ThisWorkbook.Sheets("ANALISIS_TIEMPOS")
    ws.Range("A1:H1").Value = Array("ID_Referencia", "Nombre_Referencia", "Tarea", "Tiempo_Estándar_Unit (seg)", "Tiempo_Real_Promedio (seg)", "Desviación (%)", "Muestras", "Sugerencia")

    '11. REPORTE_LOTES (NUEVO)
    Set ws = ThisWorkbook.Sheets("REPORTE_LOTES")
    ws.Range("A1:K1").Value = Array("Lote", "Referencia", "Color", "Talla", "Cantidad", "Tareas/Unidad", "Ops. Requeridas", "Ops. Completadas", "% Avance", "Tiempo Est. (Horas)", "Estado")

    '7. PARAMETROS
    Set ws = ThisWorkbook.Sheets("PARAMETROS")
    ws.Range("A1:D1").Value = Array("Parámetro", "Valor_Actual", "Descripción", "Tipo")
    
    'Formato general headers
    For Each ws In ThisWorkbook.Worksheets
        Dim rngHeader As Range
        Set rngHeader = Nothing
        Dim ultCol As Long
        
        Select Case ws.Name
            Case "ASIGNACIONES_DIARIAS", "OPERARIAS", "REFERENCIAS", "COLORES", "TALLAS", "LOTES", "TAREAS"
                'Headers en fila 3: Buscar última columna real
                ultCol = ws.Cells(3, ws.Columns.Count).End(xlToLeft).Column
                If ultCol >= 1 Then Set rngHeader = ws.Range(ws.Cells(3, 1), ws.Cells(3, ultCol))
            Case Else
                'Headers en fila 1: Buscar última columna real
                If ws.Cells(1, 1).Value <> "" Then
                    ultCol = ws.Cells(1, ws.Columns.Count).End(xlToLeft).Column
                    If ultCol >= 1 Then Set rngHeader = ws.Range(ws.Cells(1, 1), ws.Cells(1, ultCol))
                End If
        End Select
        
        If Not rngHeader Is Nothing Then
            With rngHeader
                .Font.Bold = True
                .Interior.Color = RGB(220, 220, 220)
                .Borders.LineStyle = xlContinuous
                .EntireColumn.AutoFit
            End With
        End If
    Next ws
End Sub

Function ExisteHoja(nombre As String) As Boolean
    On Error Resume Next
    ExisteHoja = Not ThisWorkbook.Sheets(nombre) Is Nothing
    On Error GoTo 0
End Function

Private Sub CrearBotonDashboard(ws As Worksheet, texto As String, macro As String, fila As Integer, col As Integer, color As Long)
    Dim btn As Shape
    Set btn = ws.Shapes.AddShape(msoShapeRoundedRectangle, ws.Cells(fila, col).Left, ws.Cells(fila, col).Top, 180, 35)
    With btn
        .Placement = xlMove
        .TextFrame.Characters.Text = texto
        .TextFrame.HorizontalAlignment = xlHAlignCenter
        .TextFrame.VerticalAlignment = xlVAlignCenter
        .Fill.ForeColor.RGB = color
        .Line.Visible = msoFalse
        .OnAction = macro
    End With
End Sub

Sub CrearBotonesVolver()
    Dim ws As Worksheet
    Dim colBtn As Integer
    Dim filaBtn As Integer
    Dim i As Long
    
    For Each ws In ThisWorkbook.Worksheets
        If ws.Name <> "INICIO" Then
            'Eliminar botón anterior si existe para evitar duplicados
            For i = ws.Shapes.Count To 1 Step -1
                If ws.Shapes(i).Type = msoAutoShape Then
                    If ws.Shapes(i).AutoShapeType = msoShapeRoundedRectangle Then
                        Dim txt As String
                        txt = ""
                        On Error Resume Next
                        txt = ws.Shapes(i).TextFrame.Characters.Text
                        On Error GoTo 0
                        If Trim(txt) = "VOLVER A INICIO" Then ws.Shapes(i).Delete
                    End If
                End If
            Next i
            
            filaBtn = 1
            colBtn = 10 'Posición por defecto (Columna J)
            
            Select Case ws.Name
                Case "ASIGNACIONES_DIARIAS"
                    filaBtn = 2
                    colBtn = 10 'J (A la derecha de Editar)
                Case "OPERARIAS", "REFERENCIAS", "COLORES", "TALLAS", "LOTES", "TAREAS"
                    filaBtn = 2
                    colBtn = 12 'I (A la derecha de los botones maestros)
                Case "MANUAL"
                    filaBtn = 1
                    colBtn = 5
                Case "DIAS_NO_LABORALES", "PARAMETROS"
                    colBtn = 6
            End Select
            
            CrearBotonDashboard ws, "VOLVER A INICIO", "IrAInicio", filaBtn, colBtn, RGB(105, 105, 105)
        End If
    Next ws
End Sub

    ThisWorkbook.Sheets("INICIO").Activate
End Sub

Public Sub PopulateRefSearchCombo()
    Dim wsInicio As Worksheet
    Dim cbo As OLEObject
    Dim refs As Collection
    Dim ref As Variant
    
    On Error Resume Next ' In case the sheet or object doesn't exist yet
    Set wsInicio = ThisWorkbook.Sheets("INICIO")
    Set cbo = wsInicio.OLEObjects("cboBuscarRef")
    If cbo Is Nothing Then Exit Sub
    On Error GoTo 0
    
    cbo.Object.Clear
    
    ' Ensure cache is loaded
    If Not EsCacheActualizado() Then CargarCacheEnMemoria
    
    Set refs = ObtenerAllReferencias()
    
    For Each ref In refs
        cbo.Object.AddItem ref
    Next ref
End Sub
