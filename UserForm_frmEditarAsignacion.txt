'═══════════════════════════════════════════════════════════════════════════════
' USERFORM: frmEditarAsignacion
' Formulario optimizado para edicion y creacion de asignaciones
' RENDIMIENTO MAXIMO: Cache, arrays, sin busquedas innecesarias
' CORRECCIÓN v3.2.2: btnGuardar_Click pasa CDate(txtFecha) a ActualizarAsignacion
'═══════════════════════════════════════════════════════════════════════════════

Option Explicit

Private datosActuales   As Variant
Private isNewAssignment As Boolean
Private fila            As Long
Private cacheOperarias  As Collection
Private cacheReferencias As Collection

Private Sub UserForm_Initialize()
    Application.ScreenUpdating = False

    If Not EsCacheActualizado() Then Call CargarCacheEnMemoria

    fila = ObtenerFilaActual()

    If fila < 4 Then
        isNewAssignment = True
    Else
        isNewAssignment = EsNuevaAsignacion()
    End If

    If isNewAssignment Then
        Me.Caption  = "CREAR NUEVA ASIGNACION"
        Me.BackColor = RGB(240, 250, 240)
    Else
        Me.Caption  = "EDITAR ASIGNACION"
        Me.BackColor = RGB(250, 250, 240)
    End If

    Call CargarListasOperarias
    Call CargarListasReferencias
    Call CargarListasEstados

    If Not isNewAssignment Then
        Call CargarDatosParaEditar
    Else
        txtFecha.Value = Format(Date, "dd/mm/yyyy")
        ReDim datosActuales(1 To 21)
    End If

    ' Preseleccion desde frmGestionLotes
    If PreselectRefID <> "" Then
        cboReferencia.Value = PreselectRefID
        PreselectRefID = ""
    End If
    If PreselectLoteID <> "" Then
        cboLote.Value = PreselectLoteID
        PreselectLoteID = ""
    End If

    Application.ScreenUpdating = True
End Sub

Private Sub CargarListasOperarias()
    Set cacheOperarias = ObtenerAllOperarias()
    Dim id As Variant
    cboOperaria.Clear
    If Not cacheOperarias Is Nothing Then
        For Each id In cacheOperarias
            cboOperaria.AddItem id
        Next id
    End If
End Sub

Private Sub CargarListasReferencias()
    Set cacheReferencias = ObtenerAllReferencias()
    Dim id As Variant
    cboReferencia.Clear
    If Not cacheReferencias Is Nothing Then
        For Each id In cacheReferencias
            cboReferencia.AddItem id
        Next id
    End If
End Sub

Private Sub CargarListasEstados()
    cboEstado.Clear
    cboEstado.AddItem "Pendiente"
    cboEstado.AddItem "En Proceso"
    cboEstado.AddItem "Completada"
    cboEstado.Value = "Pendiente"
End Sub

Private Sub CargarDatosParaEditar()
    datosActuales = ObtenerDatosAsignacion(fila)

    txtFecha.Value          = Format(datosActuales(1), "dd/mm/yyyy")
    cboOperaria.Value       = datosActuales(2)
    txtNombreOperaria.Value = datosActuales(3)
    cboReferencia.Value     = datosActuales(4)
    txtNombreReferencia.Value = datosActuales(5)

    Call CargarLotesDeReferencia(CStr(datosActuales(4)))
    cboLote.Value = datosActuales(6)

    Call CargarColoresDeReferencia(CStr(datosActuales(4)), CStr(datosActuales(6)))
    cboColor.Value = datosActuales(7)

    Call CargarTallasDeReferencia(CStr(datosActuales(4)), CStr(datosActuales(6)), CStr(datosActuales(7)))
    cboTalla.Value = datosActuales(8)

    Call CargarTareasDeReferencia(CStr(datosActuales(4)))
    cboNumeroTarea.Value = datosActuales(9)

    txtNombreTarea.Value    = datosActuales(10)
    txtCantidad.Value       = datosActuales(11)
    txtTiempoUnitario.Value = datosActuales(12)

    lblTiempoTotalSegundos.Caption = "Tiempo Total: " & Format(datosActuales(13), "#,##0") & " seg"
    lblTiempoTotalMinutos.Caption  = "(" & Format(datosActuales(14), "#,##0.00") & " min)"

    cboEstado.Value               = datosActuales(15)
    txtUnidadesCompletadas.Value  = datosActuales(16)
    txtObservaciones.Value        = datosActuales(17)

    If IsDate(datosActuales(19)) Then txtFechaReal.Value = Format(datosActuales(19), "dd/mm/yyyy")
    txtTiempoReal.Value = datosActuales(20)
End Sub

Private Sub cboOperaria_Change()
    If cboOperaria.Value <> "" Then
        txtNombreOperaria.Value = ObtenerNombreOperaria(cboOperaria.Value)
    Else
        txtNombreOperaria.Value = ""
    End If
End Sub

Private Sub cboReferencia_Change()
    If cboReferencia.Value <> "" Then
        txtNombreReferencia.Value = ObtenerNombreReferencia(cboReferencia.Value)
        txtNombreTarea.Value      = ""
        txtTiempoUnitario.Value   = ""
        Call CargarLotesDeReferencia(cboReferencia.Value)
        Call CargarTareasDeReferencia(cboReferencia.Value)
    Else
        txtNombreReferencia.Value = ""
        cboLote.Clear
        cboColor.Clear
        cboTalla.Clear
        cboNumeroTarea.Clear
    End If
End Sub

Private Sub cboLote_Change()
    If cboReferencia.Value <> "" And cboLote.Value <> "" Then
        Call CargarColoresDeReferencia(cboReferencia.Value, cboLote.Value)
    Else
        cboColor.Clear
    End If
End Sub

Private Sub cboColor_Change()
    If cboReferencia.Value <> "" And cboLote.Value <> "" And cboColor.Value <> "" Then
        Call CargarTallasDeReferencia(cboReferencia.Value, cboLote.Value, cboColor.Value)
    Else
        cboTalla.Clear
    End If
End Sub

Private Sub CargarLotesDeReferencia(idRef As String)
    Dim lotesCol As Collection
    Dim l As Variant
    Set lotesCol = ObtenerLotesDeReferencia(idRef)
    cboLote.Clear
    If Not lotesCol Is Nothing Then
        For Each l In lotesCol
            cboLote.AddItem l
        Next l
    End If
    If cboLote.ListCount > 0 Then cboLote.ListIndex = 0
End Sub

Private Sub CargarColoresDeReferencia(idRef As String, Optional idLote As String = "")
    Dim coloresCol As Collection
    Dim c As Variant
    If idLote <> "" Then
        Set coloresCol = ObtenerColoresDeReferenciaYLote(idRef, idLote)
    Else
        Set coloresCol = ObtenerColoresDeReferencia(idRef)
    End If
    cboColor.Clear
    For Each c In coloresCol
        cboColor.AddItem c
    Next c
    If cboColor.ListCount > 0 Then cboColor.ListIndex = 0
End Sub

Private Sub CargarTallasDeReferencia(idRef As String, idLote As String, Optional color As String = "")
    Dim tallasCol As Collection
    Dim talla As Variant
    If color <> "" And idLote <> "" Then
        Set tallasCol = ObtenerTallasDeReferenciaLoteColor(idRef, idLote, color)
    Else
        Set tallasCol = New Collection
    End If
    cboTalla.Clear
    For Each talla In tallasCol
        cboTalla.AddItem talla
    Next talla
    If cboTalla.ListCount > 0 Then cboTalla.ListIndex = 0
End Sub

Private Sub CargarTareasDeReferencia(idRef As String)
    Dim tareasCol As Collection
    Dim tarea As Variant
    Set tareasCol = ObtenerTareasDeReferencia(idRef)
    cboNumeroTarea.Clear
    For Each tarea In tareasCol
        cboNumeroTarea.AddItem tarea
    Next tarea
    If cboNumeroTarea.ListCount > 0 Then
        cboNumeroTarea.ListIndex = 0
        Call cboNumeroTarea_Change
    End If
End Sub

Private Sub cboNumeroTarea_Change()
    If cboNumeroTarea.Value <> "" And cboReferencia.Value <> "" Then
        txtNombreTarea.Value    = ObtenerNombreTarea(cboReferencia.Value, cboNumeroTarea.Value)
        txtTiempoUnitario.Value = ObtenerTiempoUnitario(cboReferencia.Value, cboNumeroTarea.Value)
        Call RecalcularTiempos
    End If
End Sub

Private Sub txtCantidad_Change()
    Call RecalcularTiempos
End Sub

Private Sub RecalcularTiempos()
    Dim cantidad       As Long
    Dim tiempoUnit     As Long
    Dim tiempoTotalSeg As Long
    Dim tiempoTotalMin As Double
    On Error Resume Next
    If IsNumericValue(txtCantidad.Value) And IsNumericValue(txtTiempoUnitario.Value) Then
        cantidad       = CLng(txtCantidad.Value)
        tiempoUnit     = CLng(txtTiempoUnitario.Value)
        tiempoTotalSeg = cantidad * tiempoUnit
        tiempoTotalMin = tiempoTotalSeg / 60
        lblTiempoTotalSegundos.Caption = "Tiempo Total: " & Format(tiempoTotalSeg, "#,##0") & " seg"
        lblTiempoTotalMinutos.Caption  = "(" & Format(tiempoTotalMin, "#,##0.00") & " min)"
    End If
    On Error GoTo 0
End Sub

Private Sub btnGuardar_Click()
    ' ── Validaciones ────────────────────────────────────────────────────────
    If cboOperaria.Value = "" Then
        MsgBox "[X] Seleccione una operaria", vbCritical: cboOperaria.SetFocus: Exit Sub
    End If
    If cboReferencia.Value = "" Then
        MsgBox "[X] Seleccione una referencia", vbCritical: cboReferencia.SetFocus: Exit Sub
    End If
    If cboLote.Value = "" Then
        MsgBox "[X] Seleccione un Lote", vbCritical: cboLote.SetFocus: Exit Sub
    End If
    If cboColor.Value = "" Then
        MsgBox "[X] Seleccione un color", vbCritical: cboColor.SetFocus: Exit Sub
    End If
    If cboTalla.Value = "" Then
        MsgBox "[X] Seleccione una talla", vbCritical: cboTalla.SetFocus: Exit Sub
    End If
    If cboNumeroTarea.Value = "" Then
        MsgBox "[X] Seleccione un numero de tarea", vbCritical: cboNumeroTarea.SetFocus: Exit Sub
    End If
    If txtCantidad.Value = "" Or Not IsNumericValue(txtCantidad.Value) Or CLng(txtCantidad.Value) <= 0 Then
        MsgBox "[X] La cantidad debe ser mayor a 0", vbCritical: txtCantidad.SetFocus: Exit Sub
    End If
    If Not IsDate(txtFecha.Value) Then
        MsgBox "[X] La fecha no es valida (dd/mm/yyyy)", vbCritical: txtFecha.SetFocus: Exit Sub
    End If

    ' ── Guardar ─────────────────────────────────────────────────────────────
    Dim resultado As Boolean
    resultado = False

    If isNewAssignment Then
        resultado = CrearAsignacion( _
            cboOperaria.Value, cboReferencia.Value, cboLote.Value, _
            cboColor.Value, cboTalla.Value, cboNumeroTarea.Value, _
            CLng(txtCantidad.Value), txtObservaciones.Value)

        If resultado Then
            If MsgBox("Desea crear otra asignacion?", vbYesNo + vbQuestion, "Continuar") = vbYes Then
                Call btnLimpiar_Click
            Else
                Unload Me
            End If
        End If
    Else
        ' CORRECCIÓN v3.2.2: se pasa CDate(txtFecha.Value) como segundo argumento
        resultado = ActualizarAsignacion( _
            fila, _
            CDate(txtFecha.Value), _
            cboOperaria.Value, cboReferencia.Value, cboLote.Value, _
            cboColor.Value, cboTalla.Value, cboNumeroTarea.Value, _
            CLng(txtCantidad.Value), cboEstado.Value, _
            CLng(txtUnidadesCompletadas.Value), txtObservaciones.Value, _
            txtFechaReal.Value, Val(txtTiempoReal.Value))

        If resultado Then Unload Me
    End If
End Sub

Private Sub btnCancelar_Click()
    Unload Me
End Sub

Private Sub btnLimpiar_Click()
    If isNewAssignment Then
        Call LimpiarControles(Me)
        txtFecha.Value  = Format(Date, "dd/mm/yyyy")
        cboEstado.Value = "Pendiente"
    End If
End Sub
