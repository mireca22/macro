'==============================================================================
' MODULO 3 - modGestorAsignaciones v3.3.6 (DOCUMENTACIÓN EXHAUSTIVA)
' Propósito: Núcleo de lógica de negocio para la gestión de asignaciones.
'            Maneja la creación, edición, validación de stock y capacidad,
'            así como el seguimiento de avance en la hoja LOTES.
'==============================================================================

'==============================================================================
' FUNCIÓN: CrearAsignacion
' Propósito: Procesa la creación de una nueva asignación de tarea.
' Parámetros: idOper, idRef, idLote, color, talla, numTarea, cantidad, observaciones.
' Retorno: Boolean (True si se registró correctamente).
'==============================================================================
Public Function CrearAsignacion(idOper As String, idRef As String, idLote As String, _
    color As String, talla As String, numTarea As String, cantidad As Long, _
    observaciones As String) As Boolean

    Dim ws As Worksheet, nOp$, nRef$, nTar$, tUnit&, tTotalSeg&, tTotalMin#, tDisp#, tAsig#, capOK As Boolean
    
    ' Optimización: desactivar actualización de pantalla
    Application.ScreenUpdating = False
    Set ws = ThisWorkbook.Sheets("ASIGNACIONES_DIARIAS")
    
    ' Obtener metadatos desde el Gestor de Datos Maestros
    nOp = ObtenerNombreOperaria(idOper)
    nRef = ObtenerNombreReferencia(idRef)
    nTar = ObtenerNombreTarea(idRef, numTarea)
    tUnit = ObtenerTiempoUnitario(idRef, numTarea)
    
    ' VALIDACIÓN 0: Tarea única por lote (Evitar duplicidad en nuevas asignaciones)
    If ValidarTareaAsignadaPreviamente(idRef, idLote, color, talla, numTarea) Then
        MsgBox "[X] Error: La tarea N° " & numTarea & " (" & nTar & ") ya ha sido asignada previamente para este lote (" & idLote & ") y talla (" & talla & ").", vbCritical, "Validación de Duplicados"
        CrearAsignacion = False: Exit Function
    End If

    ' Cálculos de carga de tiempo
    tTotalSeg = cantidad * tUnit: tTotalMin = tTotalSeg / 60

    ' VALIDACIÓN 1: Stock físico disponible en hoja LOTES
    If Not ValidarStockProduccion(idRef, idLote, color, talla, numTarea, cantidad, 0) Then
        CrearAsignacion = False: Exit Function
    End If
    
    ' VALIDACIÓN 2: Capacidad de tiempo de la operaria
    capOK = ValidarCapacidadOperaria(idOper, Date, tTotalMin, tDisp, tAsig)

    If Not capOK Then
        ' Flujo alternativo: División de carga si excede capacidad
        Dim minL#, resp As VbMsgBoxResult: minL = tDisp - tAsig: If minL < 0 Then minL = 0
        resp = MsgBox("LA OPERARIA EXCEDE SU CAPACIDAD." & vbCrLf & _
            "Disponible: " & minL & " min. Requerido: " & tTotalMin & " min." & vbCrLf & _
            "¿Desea dividir la carga automáticamente en días posteriores?", _
            vbYesNoCancel + vbExclamation, "Control de Capacidad")
            
        If resp = vbYes Then
            Call GenerarAsignacionesDivididas(idOper, nOp, idRef, nRef, idLote, color, talla, numTarea, nTar, cantidad, tUnit, observaciones)
            CrearAsignacion = True
        Else
            CrearAsignacion = False
        End If
        Exit Function
    End If

    ' PROCESO: Inserción de fila única
    Call InsertarFilaAsignacion(Date, idOper, nOp, idRef, nRef, idLote, color, talla, numTarea, nTar, cantidad, tUnit, observaciones)
    Call AplicarFormatoAsignaciones
    ' Nota: Se llama a GenerarReporteEnHoja para actualizar el resumen diario
    Call GenerarReporteEnHoja(Date, "DIARIO", False)
    
    Application.ScreenUpdating = True
    MsgBox "Asignación creada exitosamente.", vbInformation
    CrearAsignacion = True
End Function

'==============================================================================
' SUB: GenerarAsignacionesDivididas
' Propósito: Distribuye una cantidad de trabajo en múltiples días basándose
'            en la capacidad remanente diaria de la operaria.
'==============================================================================
Private Sub GenerarAsignacionesDivididas(idOper$, nOp$, idRef$, nRef$, idLote$, color$, talla$, numTarea$, nTar$, cantTotal&, tUnit&, obs$)
    Dim fIter As Date, pend&, capD#, asigD#, minL#, uHoy&, dUsados%
    fIter = Date: pend = cantTotal: dUsados = 0
    
    Application.ScreenUpdating = False
    Do While pend > 0
        ' Saltar días no laborales (festivos/fines de semana)
        Do While Not EsDiaLaboral(fIter): fIter = fIter + 1: Loop
        
        Call ValidarCapacidadOperaria(idOper, fIter, 0, capD, asigD)
        minL = capD - asigD
        
        If minL > 0 Then
            uHoy = Int(minL * 60 / tUnit)
            If uHoy > pend Then uHoy = pend
            
            If uHoy > 0 Then
                Call InsertarFilaAsignacion(fIter, idOper, nOp, idRef, nRef, idLote, color, talla, numTarea, nTar, uHoy, tUnit, obs & " (P." & dUsados + 1 & ")")
                pend = pend - uHoy
                dUsados = dUsados + 1
            End If
        End If
        If pend > 0 Then fIter = fIter + 1
        If dUsados > 365 Then Exit Do ' Límite preventivo de bucle infinito
    Loop
    
    Call AplicarFormatoAsignaciones: Call GenerarReporteEnHoja(Date, "DIARIO", False)
    Application.ScreenUpdating = True
    MsgBox "Carga distribuida en " & dUsados & " días laborales.", vbInformation
End Sub

'==============================================================================
' SUB: InsertarFilaAsignacion
' Propósito: Escritura optimizada en la hoja ASIGNACIONES_DIARIAS.
'==============================================================================
Private Sub InsertarFilaAsignacion(fecha As Date, idOper$, nOp$, idRef$, nRef$, idLote$, color$, talla$, numTarea$, nTar$, cant&, tUnit&, obs$)
    Dim ws As Worksheet, fila&, tSeg&, tMin#, tDisp#, tAsig#
    Set ws = ThisWorkbook.Sheets("ASIGNACIONES_DIARIAS")
    tSeg = cant * tUnit: tMin = tSeg / 60
    
    Call ValidarCapacidadOperaria(idOper, fecha, 0, tDisp, tAsig)
    fila = ws.Cells(ws.Rows.Count, 1).End(xlUp).Row + 1
    
    ' Escritura masiva vía Array (1 solo acceso COM)
    Application.EnableEvents = False
    ws.Range(ws.Cells(fila, 1), ws.Cells(fila, 21)).Value = Array(fecha, idOper, nOp, idRef, nRef, idLote, color, talla, numTarea, nTar, cant, tUnit, tSeg, tMin, "Pendiente", 0, obs, tDisp - tAsig - tMin, "", 0, 0)
    Application.EnableEvents = True

    ' Sincronizar estado del Lote (Avance Ponderado)
    Call ActualizarAvanceTalla(idRef, idLote, color, talla)
End Sub

'==============================================================================
' FUNCIÓN: ActualizarAsignacion
' Propósito: Edita una asignación existente y procesa la terminación si aplica.
'==============================================================================
Public Function ActualizarAsignacion(fila&, fecha As Date, idOper$, idRef$, idLote$, color$, talla$, numTarea$, cant&, estado$, uComp&, obs$, Optional fReal As Variant = "", Optional tRealMin# = 0) As Boolean
    Dim ws As Worksheet, nOp$, nRef$, nTar$, tUnit&, tSeg&, tMin#
    Set ws = ThisWorkbook.Sheets("ASIGNACIONES_DIARIAS")
    
    nOp = ObtenerNombreOperaria(idOper): nRef = ObtenerNombreReferencia(idRef): nTar = ObtenerNombreTarea(idRef, numTarea)
    tUnit = ObtenerTiempoUnitario(idRef, numTarea): tSeg = cant * tUnit: tMin = tSeg / 60
    
    If Not ValidarStockProduccion(idRef, idLote, color, talla, numTarea, cant, fila) Then
        ActualizarAsignacion = False: Exit Function
    End If
    
    Application.EnableEvents = False
    ' Actualización campo por campo
    ws.Cells(fila, 1).Value = fecha: ws.Cells(fila, 2).Value = idOper: ws.Cells(fila, 3).Value = nOp: ws.Cells(fila, 4).Value = idRef: ws.Cells(fila, 5).Value = nRef: ws.Cells(fila, 6).Value = idLote: ws.Cells(fila, 7).Value = color: ws.Cells(fila, 8).Value = talla: ws.Cells(fila, 9).Value = numTarea: ws.Cells(fila, 10).Value = nTar: ws.Cells(fila, 11).Value = cant: ws.Cells(fila, 12).Value = tUnit: ws.Cells(fila, 13).Value = tSeg: ws.Cells(fila, 14).Value = tMin: ws.Cells(fila, 15).Value = estado: ws.Cells(fila, 16).Value = uComp: ws.Cells(fila, 17).Value = obs
    
    ' Datos de desempeño si está terminada
    If estado = "Completada" And IsDate(fReal) Then
        ws.Cells(fila, 19).Value = CDate(fReal): ws.Cells(fila, 20).Value = tRealMin: ws.Cells(fila, 21).Value = tRealMin - tMin
    End If
    Application.EnableEvents = True
    
    ' Sincronizar estado del Lote (Avance Ponderado por Operaciones)
    Call ActualizarAvanceTalla(idRef, idLote, color, talla)
    Call VerificarReferenciaCompletada(idRef)
    
    Call GenerarReporteEnHoja(fecha, "DIARIO", False)
    MsgBox "Asignación actualizada correctamente.", vbInformation
    ActualizarAsignacion = True
End Function

'==============================================================================
' FUNCIÓN: ValidarCapacidadOperaria
' Propósito: Analiza la carga de una operaria en una fecha para ver si tiene espacio.
'==============================================================================
Public Function ValidarCapacidadOperaria(idOper$, fecha As Date, minN#, ByRef outD#, ByRef outA#) As Boolean
    Dim ws As Worksheet, i&, ultF&, data
    outD = ObtenerHorasDisponiblesOperaria(idOper): outA = 0
    Set ws = ThisWorkbook.Sheets("ASIGNACIONES_DIARIAS"): ultF = ws.Cells(ws.Rows.Count, 1).End(xlUp).Row
    
    If ultF >= 4 Then
        data = ws.Range(ws.Cells(4, 1), ws.Cells(ultF, 14)).Value
        For i = 1 To UBound(data, 1)
            If data(i, 2) = idOper And IsDate(data(i, 1)) Then
                If Int(data(i, 1)) = Int(fecha) Then outA = outA + Val(data(i, 14))
            End If
        Next i
    End If
    ValidarCapacidadOperaria = (outA + minN <= outD)
End Function

'==============================================================================
' FUNCIÓN: ValidarStockProduccion
' Propósito: Verifica que no se asigne más de lo que se recibió en LOTES.
'==============================================================================
Public Function ValidarStockProduccion(idRef$, idLote$, color$, talla$, numTarea$, cantN&, filaE&) As Boolean
    Dim wsL As Worksheet, wsA As Worksheet, i&, ultF&, data, tRec&, tAsig&, filaAbs&
    Set wsL = ThisWorkbook.Sheets("LOTES"): Set wsA = ThisWorkbook.Sheets("ASIGNACIONES_DIARIAS"): tRec = 0: ultF = wsL.Cells(wsL.Rows.Count, 1).End(xlUp).Row
    
    ' 1. Buscar total disponible en LOTES
    If ultF >= 4 Then
        data = wsL.Range(wsL.Cells(4, 1), wsL.Cells(ultF, 5)).Value
        For i = 1 To UBound(data, 1)
            If Trim(data(i, 1)) = idLote And Trim(data(i, 2)) = idRef And Trim(data(i, 3)) = color And Trim(data(i, 4)) = talla Then tRec = CLng(Val(data(i, 5))): Exit For
        Next i
    End If
    If tRec = 0 Then MsgBox "No hay stock para este lote/talla.", vbCritical: ValidarStockProduccion = False: Exit Function
    
    ' 2. Calcular total ya consumido/asignado
    tAsig = 0: ultF = wsA.Cells(wsA.Rows.Count, 1).End(xlUp).Row
    If ultF >= 4 Then
        data = wsA.Range(wsA.Cells(4, 1), wsA.Cells(ultF, 11)).Value
        For i = 1 To UBound(data, 1)
            filaAbs = i + 3: If filaAbs <> filaE Then If data(i, 4) = idRef And data(i, 6) = idLote And data(i, 7) = color And data(i, 8) = talla And data(i, 9) = numTarea Then tAsig = tAsig + CLng(Val(data(i, 11)))
        Next i
    End If
    
    If tAsig + cantN > tRec Then
        MsgBox "Exceso de cantidad. Disponible: " & (tRec - tAsig), vbCritical: ValidarStockProduccion = False
    Else
        ValidarStockProduccion = True
    End If
End Function

'==============================================================================
' FUNCIÓN: ValidarTareaAsignadaPreviamente
' Propósito: Comprueba si una tarea específica ya existe para un lote/talla.
'==============================================================================
Private Function ValidarTareaAsignadaPreviamente(idRef$, idLote$, color$, talla$, numTarea$) As Boolean
    Dim wsA As Worksheet, i&, ultF&, data
    Set wsA = ThisWorkbook.Sheets("ASIGNACIONES_DIARIAS")
    ultF = wsA.Cells(wsA.Rows.Count, 1).End(xlUp).Row
    
    If ultF < 4 Then ValidarTareaAsignadaPreviamente = False: Exit Function

    ' Leemos columnas 4 (Ref), 6 (Lote), 7 (Color), 8 (Talla) y 9 (Tarea)
    data = wsA.Range(wsA.Cells(4, 4), wsA.Cells(ultF, 9)).Value
    For i = 1 To UBound(data, 1)
        If CStr(data(i, 1)) = idRef And CStr(data(i, 3)) = idLote And _
           CStr(data(i, 4)) = color And CStr(data(i, 5)) = talla And _
           CStr(data(i, 6)) = numTarea Then
            ValidarTareaAsignadaPreviamente = True: Exit Function
        End If
    Next i
    ValidarTareaAsignadaPreviamente = False
End Function

'==============================================================================
' SUB: ActualizarAvanceTalla
' Propósito: Recalcula el avance real del lote basado en todas sus tareas asignadas.
'==============================================================================
Private Sub ActualizarAvanceTalla(idRef As String, idLote As String, color As String, talla As String)
    Dim wsL As Worksheet, wsA As Worksheet
    Dim i As Long, ultFL As Long, ultFA As Long, nTasks As Long
    Dim totalOpsDone As Long, cantRecibida As Long, filaLote As Long
    Dim dataA As Variant, unidCompEquiv As Double, avance As Double

    Set wsL = ThisWorkbook.Sheets("LOTES")
    Set wsA = ThisWorkbook.Sheets("ASIGNACIONES_DIARIAS")
    
    ' 1. Obtener número de tareas requeridas para esta referencia (desde caché)
    nTasks = ObtenerTareasDeReferencia(idRef).Count
    If nTasks = 0 Then Exit Sub

    ' 2. Localizar el registro del Lote en la hoja física
    ultFL = wsL.Cells(wsL.Rows.Count, 1).End(xlUp).Row
    filaLote = 0
    For i = 4 To ultFL
        If Trim(wsL.Cells(i, 1).Value) = Trim(idLote) And Trim(wsL.Cells(i, 2).Value) = Trim(idRef) And _
           Trim(wsL.Cells(i, 3).Value) = Trim(color) And Trim(wsL.Cells(i, 4).Value) = Trim(talla) Then
            filaLote = i
            cantRecibida = Val(wsL.Cells(i, 5).Value)
            Exit For
        End If
    Next i
    If filaLote = 0 Then Exit Sub

    ' 3. Calcular suma de unidades completadas en todas las tareas/asignaciones de este lote
    ultFA = wsA.Cells(wsA.Rows.Count, 1).End(xlUp).Row
    totalOpsDone = 0
    If ultFA >= 4 Then
        ' Leemos columnas 4 (Ref), 6 (Lote), 7 (Color), 8 (Talla) y 16 (UnidComp)
        dataA = wsA.Range(wsA.Cells(4, 4), wsA.Cells(ultFA, 16)).Value
        For i = 1 To UBound(dataA, 1)
            ' Validación robusta de fila de asignación
            If Trim(CStr(dataA(i, 1))) = Trim(idRef) And Trim(CStr(dataA(i, 3))) = Trim(idLote) And _
               Trim(CStr(dataA(i, 4))) = Trim(color) And Trim(CStr(dataA(i, 5))) = Trim(talla) Then
                totalOpsDone = totalOpsDone + Val(dataA(i, 13)) ' 13 es el índice relativo para la col 16
            End If
        Next i
    End If

    ' 4. Calcular Unidades Equivalentes y Avance Ponderado
    unidCompEquiv = Round(totalOpsDone / nTasks, 2)
    avance = 0: If cantRecibida > 0 Then avance = Round(unidCompEquiv / cantRecibida, 4)

    ' 5. Persistir en Hoja LOTES con reglas de negocio solicitadas
    wsL.Cells(filaLote, 8).Value = unidCompEquiv
    wsL.Cells(filaLote, 9).Value = avance
    
    If avance >= 1 Then
        wsL.Cells(filaLote, 10).Value = "Completada"
        wsL.Cells(filaLote, 8).Value = cantRecibida ' Asegurar igualdad absoluta al finalizar
    ElseIf avance > 0 Then
        wsL.Cells(filaLote, 10).Value = "En Proceso"
    Else
        wsL.Cells(filaLote, 10).Value = "Pendiente"
    End If
End Sub

'==============================================================================
' SUB: VerificarReferenciaCompletada
' Propósito: Consolida la referencia en PRODUCCION_FINALIZADA si todo está al 100%.
'==============================================================================
Private Sub VerificarReferenciaCompletada(idRef$)
    Dim wsL As Worksheet, wsP As Worksheet, i&, ultF&, data, tComp As Boolean, tUnits&, findR As Boolean
    Set wsL = ThisWorkbook.Sheets("LOTES"): ultF = wsL.Cells(wsL.Rows.Count, 1).End(xlUp).Row: If ultF < 4 Then Exit Sub
    data = wsL.Range(wsL.Cells(4, 1), wsL.Cells(ultF, 10)).Value: tComp = True: findR = False: tUnits = 0
    For i = 1 To UBound(data, 1)
        If data(i, 2) = idRef Then findR = True: tUnits = tUnits + Val(data(i, 5)): If data(i, 10) <> "Completada" Then tComp = False: Exit For
    Next i
    If findR And tComp Then
        Set wsP = ThisWorkbook.Sheets("PRODUCCION_FINALIZADA"): Dim nF&: nF = wsP.Cells(wsP.Rows.Count, 1).End(xlUp).Row + 1: If nF < 4 Then nF = 4
        wsP.Cells(nF, 1).Value = Date: wsP.Cells(nF, 2).Value = idRef: wsP.Cells(nF, 3).Value = ObtenerNombreReferencia(idRef): wsP.Cells(nF, 4).Value = "TODAS": wsP.Cells(nF, 5).Value = tUnits: wsP.Cells(nF, 6).Value = 1: wsP.Cells(nF, 7).Value = "Varios"
        MsgBox "¡REFERENCIA " & idRef & " COMPLETADA TOTALMENTE!", vbInformation
    End If
End Sub
