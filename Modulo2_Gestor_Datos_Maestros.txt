'==============================================================================
' MODULO 2: modGestorDatosMaestros
' Cache en memoria para busquedas ultra-rapidas (100x mas rapido que VLOOKUP)
' Sistema de Gestion de Produccion Textil v3.0
'==============================================================================

Option Explicit

'--- VARIABLES GLOBALES PARA CACHE EN MEMORIA ---
Private dictOperarias   As Object   ' Dictionary: {OP001: Array(Nombre, HorasDisp, Estado)}
Private dictReferencias As Object   ' Dictionary: {REF001: Array(Nombre, Desc, TiempoTotal)}
Private dictTareas      As Object   ' Dictionary: {REF001+1: Array(NombreTarea, TiempoSeg)}
Private dictColores     As Object   ' Dictionary: {REF001+Color: Array(Color)}
Private dictTallas      As Object   ' Dictionary: {REF001+Talla: Array(Talla)}
Private dictInventario  As Object   ' Dictionary: {RefLoteColor: Collection(Tallas)}
Private dictLotes       As Object   ' Dictionary: {REF001: Collection(LotesID)}
Private dictLoteColores As Object   ' Dictionary: {RefLote: Collection(Colores)}
Private cacheActualizado As Boolean ' Flag para validar si cache es valido

'==============================================================================
' SUB PRINCIPAL: CargarCacheEnMemoria
' EJECUTAR ESTA PRIMERA antes de usar el sistema
'==============================================================================
Public Sub CargarCacheEnMemoria()
    ' Cargar TODOS los datos maestros a memoria una sola vez
    ' Esto es 100x mas rapido que buscar en la hoja cada vez
    Application.ScreenUpdating = False

    Set dictOperarias   = CreateObject("Scripting.Dictionary")
    Set dictReferencias = CreateObject("Scripting.Dictionary")
    Set dictTareas      = CreateObject("Scripting.Dictionary")
    Set dictColores     = CreateObject("Scripting.Dictionary")
    Set dictTallas      = CreateObject("Scripting.Dictionary")
    Set dictInventario  = CreateObject("Scripting.Dictionary")
    Set dictLotes       = CreateObject("Scripting.Dictionary")
    Set dictLoteColores = CreateObject("Scripting.Dictionary")

    On Error Resume Next
    Call CargarOperariasEnCache
    Call CargarReferenciasEnCache
    Call CargarColoresEnCache
    Call CargarTallasEnCache
    Call CargarTareasEnCache
    Call CargarInventarioEnCache
    cacheActualizado = True
    Application.ScreenUpdating = True
    Call PopulateRefSearchCombo
    On Error GoTo 0
End Sub

'--- CARGADORES PRIVADOS ---
Private Sub CargarOperariasEnCache()
    Dim ws As Worksheet
    Dim fila As Long
    Dim id As String, nombre As String
    Set ws = ThisWorkbook.Sheets("OPERARIAS")
    For fila = 4 To ws.Cells(ws.Rows.Count, 1).End(xlUp).Row
        id     = Trim(CStr(ws.Cells(fila, 1).Value))
        nombre = Trim(CStr(ws.Cells(fila, 2).Value))
        If id <> "" And id <> "Empty" Then
            dictOperarias(id) = Array(nombre, ws.Cells(fila, 4).Value, ws.Cells(fila, 5).Value)
        End If
    Next fila
End Sub

Private Sub CargarReferenciasEnCache()
    Dim ws As Worksheet
    Dim fila As Long
    Dim id As String, nombre As String
    Set ws = ThisWorkbook.Sheets("REFERENCIAS")
    For fila = 4 To ws.Cells(ws.Rows.Count, 1).End(xlUp).Row
        id     = Trim(CStr(ws.Cells(fila, 1).Value))
        nombre = Trim(CStr(ws.Cells(fila, 2).Value))
        If id <> "" And id <> "Empty" Then
            dictReferencias(id) = Array(nombre, ws.Cells(fila, 3).Value, _
                                        ws.Cells(fila, 4).Value, ws.Cells(fila, 5).Value)
        End If
    Next fila
End Sub

Private Sub CargarColoresEnCache()
    Dim ws As Worksheet
    Dim fila As Long
    Dim key As String
    Set ws = ThisWorkbook.Sheets("COLORES")
    For fila = 4 To ws.Cells(ws.Rows.Count, 1).End(xlUp).Row
        key = Trim(CStr(ws.Cells(fila, 1).Value)) & Trim(CStr(ws.Cells(fila, 2).Value))
        If key <> "" Then
            dictColores(key) = Array(ws.Cells(fila, 2).Value)
        End If
    Next fila
End Sub

Private Sub CargarTallasEnCache()
    Dim ws As Worksheet
    Dim fila As Long
    Dim key As String
    Set ws = ThisWorkbook.Sheets("TALLAS")
    For fila = 4 To ws.Cells(ws.Rows.Count, 1).End(xlUp).Row
        key = Trim(CStr(ws.Cells(fila, 1).Value)) & Trim(CStr(ws.Cells(fila, 2).Value))
        If key <> "" Then
            dictTallas(key) = Array(ws.Cells(fila, 2).Value)
        End If
    Next fila
End Sub

Private Sub CargarTareasEnCache()
    Dim ws As Worksheet
    Dim fila As Long
    Dim key As String
    Set ws = ThisWorkbook.Sheets("TAREAS")
    For fila = 4 To ws.Cells(ws.Rows.Count, 1).End(xlUp).Row
        key = Trim(CStr(ws.Cells(fila, 1).Value)) & Trim(CStr(ws.Cells(fila, 2).Value))
        If key <> "" Then
            dictTareas(key) = Array(ws.Cells(fila, 3).Value, ws.Cells(fila, 4).Value, _
                                    ws.Cells(fila, 5).Value)
        End If
    Next fila
End Sub

Private Sub CargarInventarioEnCache()
    Dim ws As Worksheet
    Dim fila As Long
    Dim key As String
    Dim talla As String, idLote As String, idRef As String, color As String
    Set ws = ThisWorkbook.Sheets("LOTES")
    For fila = 4 To ws.Cells(ws.Rows.Count, 1).End(xlUp).Row
        idLote = Trim(CStr(ws.Cells(fila, 1).Value))
        idRef  = Trim(CStr(ws.Cells(fila, 2).Value))
        color  = Trim(CStr(ws.Cells(fila, 3).Value))
        talla  = Trim(CStr(ws.Cells(fila, 4).Value))
        If idRef <> "" And idLote <> "" Then
            ' 1. Lotes por Referencia
            If Not dictLotes.Exists(idRef) Then dictLotes.Add idRef, New Collection
            Dim loteExists As Boolean, l As Variant
            loteExists = False
            For Each l In dictLotes(idRef)
                If l = idLote Then loteExists = True: Exit For
            Next l
            If Not loteExists Then dictLotes(idRef).Add idLote
            ' 2. Inventario Tallas por RefLoteColor
            key = idRef & idLote & color
            If Not dictInventario.Exists(key) Then dictInventario.Add key, New Collection
            Dim exists As Boolean, item As Variant
            exists = False
            For Each item In dictInventario(key)
                If item = talla Then exists = True: Exit For
            Next item
            If Not exists Then dictInventario(key).Add talla
            ' 3. Colores por Lote
            Dim keyLC As String
            keyLC = idRef & idLote
            If Not dictLoteColores.Exists(keyLC) Then dictLoteColores.Add keyLC, New Collection
            Dim colExists As Boolean, cVal As Variant
            colExists = False
            For Each cVal In dictLoteColores(keyLC)
                If cVal = color Then colExists = True: Exit For
            Next cVal
            If Not colExists Then dictLoteColores(keyLC).Add color
        End If
    Next fila
End Sub

Public Function ObtenerNombreOperaria(id As String) As String
    If Not cacheActualizado Then Call CargarCacheEnMemoria
    On Error Resume Next
    If dictOperarias.Exists(id) Then
        ObtenerNombreOperaria = dictOperarias(id)(0)
    Else
        ObtenerNombreOperaria = ""
    End If
    On Error GoTo 0
End Function

Public Function ObtenerTiempoUnitario(idRef As String, numTarea As String) As Long

        ObtenerTiempoUnitario = 0

Public Function ObtenerColoresDeReferencia(idRef As String) As Collection
    If Not cacheActualizado Then Call CargarCacheEnMemoria
    Dim col As Collection
    Set col = New Collection
    Dim key As Variant
    Dim keyPrefix As String
    Dim keys As Variant
    keyPrefix = CStr(idRef)
    If Not dictColores Is Nothing Then
        If dictColores.Count > 0 Then
            keys = dictColores.Keys
            For Each key In keys
                If InStr(1, key, keyPrefix) = 1 Then
                    col.Add dictColores(key)(0)
                End If
            Next key
        End If
    End If
    Set ObtenerColoresDeReferencia = col
End Function

'--- FUNCIONES DE CONTROL DE CACHE ---

Public Function EsCacheActualizado() As Boolean
    EsCacheActualizado = cacheActualizado
End Function

Public Sub ActualizarCacheCompleto()
    cacheActualizado = False
    Call CargarCacheEnMemoria
End Sub

Public Sub InvalidarCache()
    ' Marca el cache como desactualizado para forzar una recarga
    cacheActualizado = False
End Sub

'--- PLACEHOLDER: PopulateRefSearchCombo (implementar segun dashboard) ---
Private Sub PopulateRefSearchCombo()
    ' Si existe un ComboBox de busqueda en INICIO, cargarlo aqui
    On Error Resume Next
    Dim ws As Worksheet
    Dim cbo As Object
    Set ws = ThisWorkbook.Sheets("INICIO")
    Set cbo = ws.OLEObjects("cboSearchRef").Object
    If Not cbo Is Nothing Then
        cbo.Clear
        Dim refs As Collection
        Dim ref As Variant
        Set refs = ObtenerAllReferencias()
        For Each ref In refs
            cbo.AddItem ref
        Next ref
    End If
    On Error GoTo 0
End Sub
