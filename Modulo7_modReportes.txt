'==============================================================================
' SUB: RecalcularSistemaLotes (CORREGIDO v3.3.9)
' Propósito: Sincroniza tiempos totales de referencias y actualiza fechas estimadas
'            en la columna Fecha_Fin_Estimada (columna 11) de la hoja LOTES.
'==============================================================================
Public Sub RecalcularSistemaLotes()
    Dim wsR As Worksheet, wsL As Worksheet, wsT As Worksheet
    Dim i As Long, ultF As Long, idRef As String
    Dim cantPend As Long, cantRecibida As Long, unidadesCompletadas As Long
    Dim fInicio As Date, fechaFinEstimada As Date
    Dim tUnitTotal As Long, minutosRequeridos As Double, capD As Long, diasNecesarios As Double
    Dim dContados As Long, fIter As Date
    
    Application.ScreenUpdating = False
    On Error GoTo ErrorHandler
    
    ' 1. Actualizar Tiempos Totales en Referencias (Suma de tareas)
    Set wsR = ThisWorkbook.Sheets("REFERENCIAS")
    Set wsL = ThisWorkbook.Sheets("LOTES")
    Set wsT = ThisWorkbook.Sheets("TAREAS")
    
    ultF = wsR.Cells(wsR.Rows.Count, 1).End(xlUp).Row
    For i = 4 To ultF
        idRef = Trim(CStr(wsR.Cells(i, 1).Value))
        If idRef <> "" Then Call ActualizarTiempoTotalReferencia(idRef)
    Next i
    
    ' 2. Actualizar Fechas Estimadas en Lotes segun unidades pendientes
    ultF = wsL.Cells(wsL.Rows.Count, 1).End(xlUp).Row
    
    For i = 4 To ultF
        idRef = Trim(CStr(wsL.Cells(i, 2).Value))
        cantRecibida = Val(wsL.Cells(i, 5).Value)
        unidadesCompletadas = Val(wsL.Cells(i, 8).Value)
        
        ' Calcular unidades pendientes (no puede ser negativo)
        cantPend = cantRecibida - unidadesCompletadas
        If cantPend < 0 Then cantPend = 0
        
        ' Obtener fecha de inicio (columna 7: Fecha_Inicio_Produccion)
        If IsDate(wsL.Cells(i, 7).Value) Then
            fInicio = wsL.Cells(i, 7).Value
        Else
            fInicio = Date
        End If
        
        ' Calcular tiempo total unitario de la referencia (en segundos)
        tUnitTotal = ObtenerTiempoTotalUnidad(idRef)
        
        If tUnitTotal > 0 And cantPend > 0 Then
            ' Minutos requeridos para las unidades pendientes
            minutosRequeridos = (cantPend * tUnitTotal) / 60
            
            ' Capacidad diaria total del sistema
            capD = GetCapacidadDiariaTotal()
            If capD = 0 Then capD = 528  ' Valor por defecto: 528 minutos (8.8 horas)
            
            ' Dias necesarios (redondeando hacia arriba)
            diasNecesarios = Application.WorksheetFunction.RoundUp(minutosRequeridos / capD, 0)
            
            ' Calcular fecha sumando solo dias laborales
            fechaFinEstimada = fInicio
            dContados = 0
            
            Do While dContados < diasNecesarios
                fechaFinEstimada = fechaFinEstimada + 1
                If EsDiaLaboral(fechaFinEstimada) Then
                    dContados = dContados + 1
                End If
                If dContados > 1000 Then Exit Do  ' Seguridad contra bucle infinito
            Loop
        Else
            ' Si no hay unidades pendientes o no hay tiempo definido, la fecha estimada es hoy
            fechaFinEstimada = Date
        End If
        
        ' Escribir la fecha estimada en la columna 11 (Fecha_Fin_Estimada)
        wsL.Cells(i, 11).Value = fechaFinEstimada
        
        ' Si el estado es "Completada", asegurar que la fecha fin real esté presente
        If UCase(Trim(wsL.Cells(i, 10).Value)) = "COMPLETADA" Then
            If Not IsDate(wsL.Cells(i, 12).Value) Then
                wsL.Cells(i, 12).Value = Date  ' Fecha_Fin_Real = hoy si no tiene
            End If
        End If
    Next i
    
    ' 3. Ordenar la hoja LOTES por Referencia y Fecha_Recepcion
    Call OrdenarLotesPorReferenciaYFecha
    
    Application.ScreenUpdating = True
    MsgBox "Recalculo de tiempos estandar y proyecciones de entrega finalizado." & vbCrLf & _
           "Se actualizaron " & (ultF - 3) & " lotes.", vbInformation
    Exit Sub
    
ErrorHandler:
    Application.ScreenUpdating = True
    MsgBox "Error en RecalcularSistemaLotes: " & Err.Description & vbCrLf & _
           "Fila: " & i, vbCritical
End Sub
