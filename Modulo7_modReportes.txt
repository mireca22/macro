'==============================================================================
' SUB: RecalcularSistemaLotes (CORREGIDO v3.3.9)
' Propósito: Sincroniza tiempos totales de referencias y actualiza fechas estimadas
'            en la columna Fecha_Fin_Estimada (columna 11) de la hoja LOTES.
'==============================================================================
Public Sub RecalcularSistemaLotes()
    Dim wsR As Worksheet, wsL As Worksheet, wsT As Worksheet
    Dim i As Long, ultF As Long, idRef As String
    Dim cantPend As Long, cantRecibida As Long, unidadesCompletadas As Long
    Dim fInicio As Date, fechaFinEstimada As Date
    Dim tUnitTotal As Long, minutosRequeridos As Double, capD As Long, diasNecesarios As Double
    Dim dContados As Long, fIter As Date
    
    Application.ScreenUpdating = False
    On Error GoTo ErrorHandler
    
    ' 1. Actualizar Tiempos Totales en Referencias (Suma de tareas)
    Set wsR = ThisWorkbook.Sheets("REFERENCIAS")
    Set wsL = ThisWorkbook.Sheets("LOTES")
    Set wsT = ThisWorkbook.Sheets("TAREAS")
    
    ultF = wsR.Cells(wsR.Rows.Count, 1).End(xlUp).Row
    For i = 4 To ultF
        idRef = Trim(CStr(wsR.Cells(i, 1).Value))
        If idRef <> "" Then Call ActualizarTiempoTotalReferencia(idRef)
    Next i
    
    ' 2. Actualizar Fechas Estimadas en Lotes segun unidades pendientes
    ultF = wsL.Cells(wsL.Rows.Count, 1).End(xlUp).Row
    
    For i = 4 To ultF
        idRef = Trim(CStr(wsL.Cells(i, 2).Value))
        cantRecibida = Val(wsL.Cells(i, 5).Value)
        unidadesCompletadas = Val(wsL.Cells(i, 8).Value)
        
        ' Calcular unidades pendientes (no puede ser negativo)
        cantPend = cantRecibida - unidadesCompletadas
        If cantPend < 0 Then cantPend = 0
        
        ' Obtener fecha de inicio (columna 7: Fecha_Inicio_Produccion)
        If IsDate(wsL.Cells(i, 7).Value) Then
            fInicio = wsL.Cells(i, 7).Value
        Else
            fInicio = Date
        End If
        
        ' Calcular tiempo total unitario de la referencia (en segundos)
        tUnitTotal = ObtenerTiempoTotalUnidad(idRef)
        
        If tUnitTotal > 0 And cantPend > 0 Then
            ' Minutos requeridos para las unidades pendientes
            minutosRequeridos = (cantPend * tUnitTotal) / 60
            
            ' Capacidad diaria total del sistema
            capD = GetCapacidadDiariaTotal()
            If capD = 0 Then capD = 528  ' Valor por defecto: 528 minutos (8.8 horas)
            
            ' Dias necesarios (redondeando hacia arriba)
            diasNecesarios = Application.WorksheetFunction.RoundUp(minutosRequeridos / capD, 0)
            
            ' Calcular fecha sumando solo dias laborales
            fechaFinEstimada = fInicio
            dContados = 0
            
            Do While dContados < diasNecesarios
                fechaFinEstimada = fechaFinEstimada + 1
                If EsDiaLaboral(fechaFinEstimada) Then
                    dContados = dContados + 1
                End If
                If dContados > 1000 Then Exit Do  ' Seguridad contra bucle infinito
            Loop
        Else
            ' Si no hay unidades pendientes o no hay tiempo definido, la fecha estimada es hoy
            fechaFinEstimada = Date
        End If
        
        ' Escribir la fecha estimada en la columna 11 (Fecha_Fin_Estimada)
        wsL.Cells(i, 11).Value = fechaFinEstimada
        
        ' Si el estado es "Completada", asegurar que la fecha fin real esté presente
        If UCase(Trim(wsL.Cells(i, 10).Value)) = "COMPLETADA" Then
            If Not IsDate(wsL.Cells(i, 12).Value) Then
                wsL.Cells(i, 12).Value = Date  ' Fecha_Fin_Real = hoy si no tiene
            End If
        End If
    Next i
    
    ' 3. Ordenar la hoja LOTES por Referencia y Fecha_Recepcion
    Call OrdenarLotesPorReferenciaYFecha
    
    Application.ScreenUpdating = True
    MsgBox "Recalculo de tiempos estandar y proyecciones de entrega finalizado." & vbCrLf & _
           "Se actualizaron " & (ultF - 3) & " lotes.", vbInformation
    Exit Sub
    
ErrorHandler:
    Application.ScreenUpdating = True
    MsgBox "Error en RecalcularSistemaLotes: " & Err.Description & vbCrLf & _
           "Fila: " & i, vbCritical
End Sub



'==============================================================================
' SUB: GenerarInformeGerencial
' Propósito: Informe consolidado por Referencia para la gerencia (v4.0).
'==============================================================================
Public Sub GenerarInformeGerencial()
    Dim wsG As Worksheet, wsL As Worksheet, wsA As Worksheet, wsT As Worksheet
    Dim dictRef As Object, dictTaskCount As Object, i As Long, ultF As Long, fila As Long
    Dim data As Variant, dataT As Variant, idRef As String, cant As Long, unidCompEquiv As Double
    Dim k As Variant, res As Variant, av As Double, efic As Double, fEnt As Date, est As String, totalTimePerUnit As Double, pendCalc As Double
    Dim estadoLote As String, totalOpsProyectadas As Double
    
    Set wsG = ThisWorkbook.Sheets("INFORME_CONSOLIDADO")
    Set wsL = ThisWorkbook.Sheets("LOTES")
    Set wsA = ThisWorkbook.Sheets("ASIGNACIONES_DIARIAS")
    Set wsT = ThisWorkbook.Sheets("TAREAS")
    Set dictRef = CreateObject("Scripting.Dictionary")
    Set dictTaskCount = CreateObject("Scripting.Dictionary")
    
    ' 1. Obtener carga total de tiempo por unidad (Suma de Segundos de todas las tareas)
    ultF = wsT.Cells(wsT.Rows.count, 1).End(xlUp).Row
    If ultF >= 4 Then
        dataT = wsT.Range("A4:D" & ultF).Value
        For i = 1 To UBound(dataT, 1)
            idRef = Trim(CStr(dataT(i, 1)))
            If idRef <> "" Then dictTaskCount(idRef) = dictTaskCount(idRef) + val(dataT(i, 4))
        Next i
    End If
    
    Application.ScreenUpdating = False
    
    ultF = wsL.Cells(wsL.Rows.count, 1).End(xlUp).Row
    If ultF >= 4 Then
        data = wsL.Range(wsL.Cells(4, 1), wsL.Cells(ultF, 10)).Value
        For i = 1 To UBound(data, 1)
            idRef = Trim(CStr(data(i, 2)))
            cant = val(data(i, 5))
            estadoLote = Trim(CStr(data(i, 10)))
            
            If idRef <> "" Then
                If Not dictRef.Exists(idRef) Then
                    dictRef(idRef) = Array(ObtenerNombreReferencia(idRef), cant, 0, 0, 0, IIf(estadoLote = "Completada", cant, 0))
                Else
                    res = dictRef(idRef): res(1) = res(1) + cant: dictRef(idRef) = res
                    If estadoLote = "Completada" Then res(5) = res(5) + cant: dictRef(idRef) = res
                End If
            End If
        Next i
    End If
    
    ultF = wsA.Cells(wsA.Rows.count, 1).End(xlUp).Row
    If ultF >= 4 Then
        data = wsA.Range(wsA.Cells(4, 1), wsA.Cells(ultF, 21)).Value
        For i = 1 To UBound(data, 1)
            idRef = Trim(CStr(data(i, 4)))
            If dictRef.Exists(idRef) Then
                res = dictRef(idRef)
                res(2) = res(2) + (CDbl(data(i, 16)) * CDbl(data(i, 12)))
                res(3) = res(3) + val(data(i, 14)): res(4) = res(4) + val(data(i, 20))
                dictRef(idRef) = res
            End If
        Next i
    End If
    
    wsG.Rows("4:" & wsG.Rows.count).ClearContents
    wsG.Rows("4:" & wsG.Rows.count).Interior.ColorIndex = xlNone
    
    fila = 4
    For Each k In dictRef.Keys
        res = dictRef(k)
        totalTimePerUnit = 0: If dictTaskCount.Exists(CStr(k)) Then totalTimePerUnit = dictTaskCount(CStr(k))

        unidCompEquiv = res(5)

        totalOpsProyectadas = CDbl(res(1)) * totalTimePerUnit
        av = 0
        If totalOpsProyectadas > 0 Then av = Round(res(2) / totalOpsProyectadas, 4)

        ' Cálculo seguro de Eficiencia (evita error de división por cero / overflow)
        If res(4) > 0 Then
            efic = res(3) / res(4)
        Else
            efic = 0
        End If

        ' Cálculo seguro de unidades pendientes para proyección de fecha
        pendCalc = 0
        If totalTimePerUnit > 0 Then
            pendCalc = res(2) / totalTimePerUnit
        End If

        fEnt = CalcularFechaFinEstimada(CStr(k), CLng(res(1) - pendCalc), Date)
        
        If av >= 1 Then
            est = "TERMINADO"
        ElseIf av > 0 Then
            est = "EN CURSO"
        Else
            est = "PENDIENTE"
        End If
        
        wsG.Cells(fila, 1).Value = res(0)
        wsG.Cells(fila, 2).Value = res(1)
        wsG.Cells(fila, 3).Value = unidCompEquiv
        wsG.Cells(fila, 4).Value = av
        wsG.Cells(fila, 5).Value = res(3)
        wsG.Cells(fila, 6).Value = res(4)
        wsG.Cells(fila, 7).Value = efic
        wsG.Cells(fila, 8).Value = IIf(av >= 1, "ENTREGADO", fEnt)
        wsG.Cells(fila, 9).Value = est
        
        If est = "TERMINADO" Then
            wsG.Cells(fila, 9).Interior.color = RGB(198, 239, 206)
        ElseIf est = "EN CURSO" Then
            wsG.Cells(fila, 9).Interior.color = RGB(255, 235, 156)
        End If
        fila = fila + 1
    Next k
    
    If fila > 4 Then
        Dim rngD As Range, rngG As Range
        Set rngD = wsG.Range("D4:D" & fila - 1)
        Set rngG = wsG.Range("G4:G" & fila - 1)
        rngD.NumberFormat = "0.00%": rngG.NumberFormat = "0.00%"
        rngD.FormatConditions.Delete: rngG.FormatConditions.Delete
        rngD.FormatConditions.AddDatabar: rngG.FormatConditions.AddDatabar
        wsG.Range("H4:H" & fila - 1).NumberFormat = "dd/mm/yyyy"
    End If
    Call AjustarAnchoColumnasInteligente(wsG): wsG.Activate: Application.ScreenUpdating = True
End Sub




'==============================================================================
' SUB: GenerarReporteAvanceProduccion
' Propósito: Muestra el avance por referencia en la hoja AVANCE_PRODUCCION.
'==============================================================================
Public Sub GenerarReporteAvanceProduccion()
    Dim wsAv    As Worksheet   ' Hoja de salida: AVANCE_PRODUCCION
    Dim wsL     As Worksheet   ' Hoja LOTES
    Dim wsT     As Worksheet   ' Hoja TAREAS
    Dim wsAs    As Worksheet   ' Hoja ASIGNACIONES_DIARIAS
    
    ' Diccionario: Tiempo total por referencia (minutos por unidad completa)
    ' Key: idRef, Value: Double (tiempo total unitario en minutos)
    Dim dictTiempoTotalRef As Object
    
    ' Diccionario: Tiempo por tarea específica (para cálculo de minutos ejecutados)
    ' Key: idRef & "|" & numTarea, Value: Double (tiempo unitario en minutos)
    Dim dictTiempoTarea As Object
    
    ' Diccionario acumulador por referencia:
    ' Key: idRef
    ' Value: Array(
    '   0 = NombreRef              String
    '   1 = TotalCantidadLotes     Double   (suma de CantidadLote)
    '   2 = TotalMinutosRequeridos Double   (suma de CantidadLote × TiempoTotalRef)
    '   3 = TotalMinutosEjecutados Double   (suma de UnidadesCompletadas × TiempoUnitarioTarea)
    '   4 = UnidadesTerminadas     Double   (lotes donde todas las tareas >= CantidadLote)
    ' )
    Dim dictRef As Object
    
    ' Diccionario: Por lote, almacenar cantidad
    ' Key: idRef & "|" & idLote
    ' Value: Array(
    '   0 = CantidadLote           Double
    '   1 = MinutosRequeridosLote  Double (CantidadLote × TiempoTotalRef)
    ' )
    Dim dictLoteInfo As Object
    
    ' Diccionario para acumular minutos ejecutados por tarea dentro de cada lote
    ' Key: idRef & "|" & idLote & "|" & numTarea
    ' Value: Double (minutos ejecutados = UnidadesCompletadas × TiempoUnitarioTarea)
    Dim dictMinutosEjecutadosPorTarea As Object
    
    ' Diccionario para almacenar las tareas que tiene cada lote
    ' Key: idRef & "|" & idLote
    ' Value: Collection de números de tarea
    Dim dictTareasDelLote As Object
    
    Dim i               As Long
    Dim ultF            As Long
    Dim fila            As Long
    Dim idRef           As String
    Dim idLote          As String
    Dim numTarea        As String
    Dim cantLote        As Double
    Dim tiempoUnitarioSeg As Long
    Dim tiempoUnitarioMin As Double
    Dim tiempoTotalRefMin  As Double
    Dim minutosEjecutados  As Double
    Dim keyLote        As String
    Dim keyTarea       As String
    Dim keyTareaLote   As String
    Dim dataL          As Variant
    Dim dataT          As Variant
    Dim dataAs         As Variant
    Dim k              As Variant
    Dim arrRef         As Variant
    Dim tareaNum       As Variant
    Dim colTareas      As Collection
    Dim loteCompleto   As Boolean
    Dim tareaReq       As Variant
    Dim minutosNecesariosTarea As Double
    Dim minutosRealTarea As Double
    Dim estado         As String
    Dim avance         As Double
    Dim minutosReq     As Double
    Dim minutosEjec    As Double
    Dim unidTerminadas As Double
    
    Set wsAv = ThisWorkbook.Sheets("AVANCE_PRODUCCION")
    Set wsL = ThisWorkbook.Sheets("LOTES")
    Set wsT = ThisWorkbook.Sheets("TAREAS")
    Set wsAs = ThisWorkbook.Sheets("ASIGNACIONES_DIARIAS")
    
    Set dictTiempoTotalRef = CreateObject("Scripting.Dictionary")
    Set dictTiempoTarea = CreateObject("Scripting.Dictionary")
    Set dictRef = CreateObject("Scripting.Dictionary")
    Set dictLoteInfo = CreateObject("Scripting.Dictionary")
    Set dictMinutosEjecutadosPorTarea = CreateObject("Scripting.Dictionary")
    Set dictTareasDelLote = CreateObject("Scripting.Dictionary")
    
    Application.ScreenUpdating = False
    
    '--------------------------------------------------------------------------
    ' PASO 1: Cargar tiempos unitarios por tarea (en minutos) desde TAREAS
    '--------------------------------------------------------------------------
    ultF = wsT.Cells(wsT.Rows.count, 1).End(xlUp).Row
    If ultF >= 4 Then
        dataT = wsT.Range(wsT.Cells(4, 1), wsT.Cells(ultF, 4)).Value
        For i = 1 To UBound(dataT, 1)
            idRef = Trim(CStr(dataT(i, 1)))
            numTarea = Trim(CStr(dataT(i, 2)))
            tiempoUnitarioSeg = CLng(val(dataT(i, 4)))
            tiempoUnitarioMin = tiempoUnitarioSeg / 60
            
            If idRef <> "" And numTarea <> "" Then
                keyTarea = idRef & "|" & numTarea
                dictTiempoTarea(keyTarea) = tiempoUnitarioMin
            End If
        Next i
    End If
    
    '--------------------------------------------------------------------------
    ' PASO 2: Calcular tiempo total por referencia (suma de tiempos de tareas)
    '--------------------------------------------------------------------------
    For Each k In dictTiempoTarea.Keys
        idRef = Split(CStr(k), "|")(0)
        If Not dictTiempoTotalRef.Exists(idRef) Then
            dictTiempoTotalRef(idRef) = CDbl(dictTiempoTarea(k))
        Else
            dictTiempoTotalRef(idRef) = dictTiempoTotalRef(idRef) + CDbl(dictTiempoTarea(k))
        End If
    Next k
    
    '--------------------------------------------------------------------------
    ' PASO 3: Procesar LOTES para construir estructura base
    '--------------------------------------------------------------------------
    ultF = wsL.Cells(wsL.Rows.count, 1).End(xlUp).Row
    If ultF >= 4 Then
        dataL = wsL.Range(wsL.Cells(4, 1), wsL.Cells(ultF, 5)).Value
        For i = 1 To UBound(dataL, 1)
            idLote = Trim(CStr(dataL(i, 1)))
            idRef = Trim(CStr(dataL(i, 2)))
            cantLote = CDbl(val(dataL(i, 5)))
            
            If idRef <> "" And idLote <> "" And cantLote > 0 Then
                ' Obtener tiempo total unitario de la referencia (en minutos)
                tiempoTotalRefMin = 0
                If dictTiempoTotalRef.Exists(idRef) Then
                    tiempoTotalRefMin = dictTiempoTotalRef(idRef)
                End If
                
                ' Guardar información del lote
                keyLote = idRef & "|" & idLote
                If Not dictLoteInfo.Exists(keyLote) Then
                    dictLoteInfo(keyLote) = Array(cantLote, cantLote * tiempoTotalRefMin)
                End If
                
                ' Inicializar colección de tareas para este lote (usando SET)
                If Not dictTareasDelLote.Exists(keyLote) Then
                    Set dictTareasDelLote(keyLote) = New Collection
                End If
                
                ' Acumular en dictRef (estructura principal)
                If Not dictRef.Exists(idRef) Then
                    dictRef(idRef) = Array( _
                        ObtenerNombreReferencia(idRef), _
                        cantLote, _
                        cantLote * tiempoTotalRefMin, _
                        0, _
                        0 _
                    )
                Else
                    arrRef = dictRef(idRef)
                    arrRef(1) = arrRef(1) + cantLote
                    arrRef(2) = arrRef(2) + (cantLote * tiempoTotalRefMin)
                    dictRef(idRef) = arrRef
                End If
            End If
        Next i
    End If
    
    '--------------------------------------------------------------------------
    ' PASO 4: Procesar ASIGNACIONES_DIARIAS
    '   a) Acumular minutos ejecutados por referencia
    '   b) Acumular minutos ejecutados por lote+tarea
    '   c) Registrar qué tareas existen por lote
    '--------------------------------------------------------------------------
    ultF = wsAs.Cells(wsAs.Rows.count, 1).End(xlUp).Row
    If ultF >= 4 Then
        dataAs = wsAs.Range(wsAs.Cells(4, 1), wsAs.Cells(ultF, 16)).Value
        For i = 1 To UBound(dataAs, 1)
            idRef = Trim(CStr(dataAs(i, 4)))
            idLote = Trim(CStr(dataAs(i, 6)))
            numTarea = Trim(CStr(dataAs(i, 9)))
            Dim uComp As Double
            uComp = CDbl(val(dataAs(i, 16)))  ' Unidades Completadas
            
            If idRef <> "" And idLote <> "" And numTarea <> "" And uComp > 0 Then
                ' Obtener tiempo unitario de esta tarea (en minutos)
                keyTarea = idRef & "|" & numTarea
                tiempoUnitarioMin = 0
                If dictTiempoTarea.Exists(keyTarea) Then
                    tiempoUnitarioMin = dictTiempoTarea(keyTarea)
                End If
                
                ' Calcular minutos ejecutados en esta asignación
                minutosEjecutados = uComp * tiempoUnitarioMin
                
                ' a) Acumular minutos ejecutados en dictRef
                If dictRef.Exists(idRef) Then
                    arrRef = dictRef(idRef)
                    arrRef(3) = arrRef(3) + minutosEjecutados
                    dictRef(idRef) = arrRef
                End If
                
                ' b) Acumular minutos ejecutados por lote+tarea
                keyTareaLote = idRef & "|" & idLote & "|" & numTarea
                If Not dictMinutosEjecutadosPorTarea.Exists(keyTareaLote) Then
                    dictMinutosEjecutadosPorTarea(keyTareaLote) = minutosEjecutados
                Else
                    dictMinutosEjecutadosPorTarea(keyTareaLote) = dictMinutosEjecutadosPorTarea(keyTareaLote) + minutosEjecutados
                End If
                
                ' c) Registrar esta tarea en la colección del lote (usando SET)
                keyLote = idRef & "|" & idLote
                If dictTareasDelLote.Exists(keyLote) Then
                    Set colTareas = dictTareasDelLote(keyLote)
                    
                    ' Verificar si ya existe para evitar duplicados
                    Dim tareaExiste As Boolean
                    tareaExiste = False
                    For Each tareaNum In colTareas
                        If CStr(tareaNum) = numTarea Then
                            tareaExiste = True
                            Exit For
                        End If
                    Next tareaNum
                    
                    If Not tareaExiste Then
                        colTareas.Add numTarea
                    End If
                    
                    ' Reasignar la colección modificada
                    Set dictTareasDelLote(keyLote) = colTareas
                End If
            End If
        Next i
    End If
    
    '--------------------------------------------------------------------------
    ' PASO 5: Calcular Unidades Terminadas por referencia
    '   Un lote está TERMINADO cuando para CADA tarea que debería tener
    '   (según las tareas definidas en la referencia), los minutos ejecutados
    '   equivalen a haber completado al menos la CantidadLote.
    '
    '   Es decir: minutosEjecutados_en_tarea >= (CantidadLote × tiempoUnitario_tarea)
    '--------------------------------------------------------------------------
    For Each k In dictLoteInfo.Keys
        idRef = Split(CStr(k), "|")(0)
        idLote = Split(CStr(k), "|")(1)
        cantLote = dictLoteInfo(k)(0)
        
        ' Obtener el tiempo total unitario de la referencia
        tiempoTotalRefMin = 0
        If dictTiempoTotalRef.Exists(idRef) Then
            tiempoTotalRefMin = dictTiempoTotalRef(idRef)
        End If
        
        ' Determinar cuántas tareas DEBERÍA tener este lote
        ' (todas las tareas definidas para la referencia)
        Dim tareasRequeridas As New Collection
        Dim taskKey As Variant
        For Each taskKey In dictTiempoTarea.Keys
            If Left(CStr(taskKey), Len(idRef)) = idRef And Mid(CStr(taskKey), Len(idRef) + 1, 1) = "|" Then
                Dim tareaNumReq As String
                tareaNumReq = Split(CStr(taskKey), "|")(1)
                tareasRequeridas.Add tareaNumReq
            End If
        Next taskKey
        
        If tareasRequeridas.count = 0 Then GoTo SiguienteLote2
        
        ' Verificar que TODAS las tareas requeridas están completas al 100%
        loteCompleto = True
        For Each tareaReq In tareasRequeridas
            keyTareaLote = idRef & "|" & idLote & "|" & tareaReq
            
            ' Obtener tiempo unitario de esta tarea
            keyTarea = idRef & "|" & tareaReq
            tiempoUnitarioMin = 0
            If dictTiempoTarea.Exists(keyTarea) Then
                tiempoUnitarioMin = dictTiempoTarea(keyTarea)
            End If
            
            ' Minutos necesarios para completar esta tarea al 100%
            minutosNecesariosTarea = cantLote * tiempoUnitarioMin
            
            ' Minutos realmente ejecutados
            minutosRealTarea = 0
            If dictMinutosEjecutadosPorTarea.Exists(keyTareaLote) Then
                minutosRealTarea = dictMinutosEjecutadosPorTarea(keyTareaLote)
            End If
            
            ' Si no alcanza el 100%, el lote no está terminado
            If minutosRealTarea < minutosNecesariosTarea - 0.001 Then
                loteCompleto = False
                Exit For
            End If
        Next tareaReq
        
        ' Si el lote está completo, sumar su cantidad a UnidadesTerminadas
        If loteCompleto And tareasRequeridas.count > 0 Then
            If dictRef.Exists(idRef) Then
                arrRef = dictRef(idRef)
                arrRef(4) = arrRef(4) + cantLote
                dictRef(idRef) = arrRef
            End If
        End If
            
SiguienteLote2:
        Next k
        
        '--------------------------------------------------------------------------
        ' PASO 6: Escribir resultados en hoja AVANCE_PRODUCCION
        '--------------------------------------------------------------------------
        wsAv.Rows("4:" & wsAv.Rows.count).ClearContents
        wsAv.Rows("4:" & wsAv.Rows.count).Interior.ColorIndex = xlNone
        wsAv.Rows("4:" & wsAv.Rows.count).Font.ColorIndex = xlAutomatic
        
        ' Headers
        wsAv.Range("A3:H3").Value = Array( _
            "Referencia", _
            "Nombre", _
            "Total Unid. Lotes", _
            "Minutos Requeridos", _
            "Minutos Ejecutados", _
            "Avance %", _
            "Unidades Terminadas", _
            "Estado" _
        )
        wsAv.Range("A3:H3").Font.Bold = True
        wsAv.Range("A3:H3").Interior.color = RGB(220, 220, 220)
        
        fila = 4
        For Each k In dictRef.Keys
            arrRef = dictRef(k)
            
            minutosReq = CDbl(arrRef(2))
            minutosEjec = CDbl(arrRef(3))
            unidTerminadas = CDbl(arrRef(4))
            
            ' Avance = minutos ejecutados / minutos requeridos
            avance = 0
            If minutosReq > 0 Then
                avance = Round(minutosEjec / minutosReq, 4)
            End If
            
            ' Determinar estado
            If avance >= 1 Then
                estado = "TERMINADO"
            ElseIf avance > 0 Then
                estado = "EN CURSO"
            Else
                estado = "PENDIENTE"
            End If
            
            wsAv.Cells(fila, 1).Value = CStr(k)
            wsAv.Cells(fila, 2).Value = arrRef(0)
            wsAv.Cells(fila, 3).Value = arrRef(1)
            wsAv.Cells(fila, 4).Value = Round(minutosReq, 2)
            wsAv.Cells(fila, 5).Value = Round(minutosEjec, 2)
            wsAv.Cells(fila, 6).Value = avance
            wsAv.Cells(fila, 7).Value = unidTerminadas
            wsAv.Cells(fila, 8).Value = estado
            
            ' Formato condicional de fila
            Select Case estado
                Case "TERMINADO"
                    wsAv.Range("A" & fila & ":H" & fila).Interior.color = RGB(198, 239, 206)
                    wsAv.Range("A" & fila & ":H" & fila).Font.color = RGB(0, 97, 0)
                Case "EN CURSO"
                    wsAv.Range("A" & fila & ":H" & fila).Interior.color = RGB(255, 235, 156)
                    wsAv.Range("A" & fila & ":H" & fila).Font.color = RGB(156, 87, 0)
                Case "PENDIENTE"
                    wsAv.Range("A" & fila & ":H" & fila).Interior.color = RGB(255, 199, 206)
                    wsAv.Range("A" & fila & ":H" & fila).Font.color = RGB(156, 0, 6)
            End Select
            
            fila = fila + 1
        Next k
        
        If fila > 4 Then
            Dim rngAv As Range: Set rngAv = wsAv.Range("F4:F" & fila - 1)
            rngAv.NumberFormat = "0.00%"
            rngAv.FormatConditions.Delete
            rngAv.FormatConditions.AddDatabar
            wsAv.Range("D4:E" & fila - 1).NumberFormat = "#,##0.00"
        End If
        
        Call AjustarAnchoColumnasInteligente(wsAv)
        wsAv.Activate
        Application.ScreenUpdating = True
        MsgBox "Reporte de avance de producción generado (basado en minutos ponderados).", vbInformation

End Sub

'==============================================================================
' SUB: GenerarReporteLotes
' Propósito: Calcula el avance detallado por lote basado en operaciones (tareas).
'==============================================================================
Public Sub GenerarReporteLotes()
    Dim wsR As Worksheet, wsL As Worksheet, wsA As Worksheet, wsT As Worksheet
    Dim dictRefParams As Object, dictOpsComp As Object
    Dim i As Long, ultF As Long, fila As Long
    Dim idRef As String, idLote As String, color As String, talla As String
    Dim nTareas As Long, tUnitTotal As Long, cantLote As Long
    Dim opsReq As Double, opsComp As Double, avance As Double, tEstHoras As Double
    Dim estado As String, dataL As Variant, dataT As Variant, dataA As Variant
    Dim key As String
    
    Set wsR = ThisWorkbook.Sheets("REPORTE_LOTES")
    Set wsL = ThisWorkbook.Sheets("LOTES")
    Set wsA = ThisWorkbook.Sheets("ASIGNACIONES_DIARIAS")
    Set wsT = ThisWorkbook.Sheets("TAREAS")
    
    ' 1. Mapear Referencias -> Cantidad de Tareas y Tiempo Total
    Set dictRefParams = CreateObject("Scripting.Dictionary")
    ultF = wsT.Cells(wsT.Rows.count, 1).End(xlUp).Row
    If ultF >= 4 Then
        dataT = wsT.Range(wsT.Cells(4, 1), wsT.Cells(ultF, 4)).Value
        For i = 1 To UBound(dataT, 1)
            idRef = Trim(CStr(dataT(i, 1)))
            If idRef <> "" Then
                If Not dictRefParams.Exists(idRef) Then
                    dictRefParams(idRef) = Array(1, val(dataT(i, 4)))
                Else
                    Dim arr: arr = dictRefParams(idRef)
                    arr(0) = arr(0) + 1: arr(1) = arr(1) + val(dataT(i, 4))
                    dictRefParams(idRef) = arr
                End If
            End If
        Next i
    End If
    
    ' 2. Mapear Lotes -> Total de Operaciones Completadas
    Set dictOpsComp = CreateObject("Scripting.Dictionary")
    ultF = wsA.Cells(wsA.Rows.count, 1).End(xlUp).Row
    If ultF >= 4 Then
        dataA = wsA.Range(wsA.Cells(4, 1), wsA.Cells(ultF, 16)).Value
        For i = 1 To UBound(dataA, 1)
            idRef = Trim(CStr(dataA(i, 4)))
            idLote = Trim(CStr(dataA(i, 6)))
            If idRef <> "" And idLote <> "" Then
                key = idRef & "|" & idLote
                dictOpsComp(key) = dictOpsComp(key) + val(dataA(i, 16))
            End If
        Next i
    End If
    
    ' 3. Construir el Reporte
    Application.ScreenUpdating = False
    wsR.Rows("3:" & wsR.Rows.count).ClearContents
    wsR.Rows("3:" & wsR.Rows.count).Interior.ColorIndex = xlNone
    
    wsR.Range("A3:K3").Value = Array("Lote", "Referencia", "Color", "Talla", "Cantidad", "Tareas/Unidad", "Ops. Requeridas", "Ops. Completadas", "Avance %", "Tiempo Est. Horas", "Estado")
    wsR.Range("A3:K3").Font.Bold = True: wsR.Range("A3:K3").Interior.color = RGB(220, 220, 220)
    
    ultF = wsL.Cells(wsL.Rows.count, 1).End(xlUp).Row
    If ultF >= 4 Then
        wsL.Range("A4:K" & ultF).Sort Key1:=wsL.Range("B4"), Order1:=xlAscending, Key2:=wsL.Range("A4"), Order2:=xlAscending, Header:=xlNo
        
        dataL = wsL.Range(wsL.Cells(4, 1), wsL.Cells(ultF, 5)).Value
        fila = 4
        For i = 1 To UBound(dataL, 1)
            idLote = Trim(CStr(dataL(i, 1))): idRef = Trim(CStr(dataL(i, 2)))
            color = dataL(i, 3): talla = dataL(i, 4): cantLote = val(dataL(i, 5))
            
            nTareas = 0: tUnitTotal = 0
            If dictRefParams.Exists(idRef) Then nTareas = dictRefParams(idRef)(0): tUnitTotal = dictRefParams(idRef)(1)
            
            opsReq = CDbl(cantLote) * nTareas
            key = idRef & "|" & idLote
            opsComp = 0: If dictOpsComp.Exists(key) Then opsComp = dictOpsComp(key)
            
            If opsReq > 0 Then
                avance = Round(opsComp / opsReq, 4)
            Else
                avance = 0
            End If
            
            tEstHoras = (CDbl(cantLote) * tUnitTotal) / 3600
            
            If avance >= 1 Then estado = "COMPLETADO" Else If avance > 0 Then estado = "EN PROCESO" Else estado = "PENDIENTE"
            
            wsR.Range("A" & fila & ":K" & fila).Value = Array(idLote, idRef, color, talla, cantLote, nTareas, opsReq, opsComp, avance, tEstHoras, estado)
            If estado = "COMPLETADO" Then
                wsR.Range("A" & fila & ":K" & fila).Interior.color = RGB(198, 239, 206)
            ElseIf estado = "EN PROCESO" Then
                wsR.Range("A" & fila & ":K" & fila).Interior.color = RGB(255, 235, 156)
            End If
            fila = fila + 1
        Next i
        If fila > 4 Then
            Dim rngLote As Range: Set rngLote = wsR.Range("I4:I" & fila - 1)
            rngLote.NumberFormat = "0.00%"
            rngLote.FormatConditions.Delete
            rngLote.FormatConditions.AddDatabar
            wsR.Range("J4:J" & fila - 1).NumberFormat = "0.00"
        End If
    End If
    Call AjustarAnchoColumnasInteligente(wsR): wsR.Activate: Application.ScreenUpdating = True
End Sub

'==============================================================================
' SUB: GenerarReporteTrazabilidadLotesPendientes (NUEVO v4.1)

'            completadas para cada lote activo en el sistema.

    Dim dictTasks As Object, dictAsig As Object
    Dim i As Long, ultF As Long, filaDest As Long
    Dim idRef As String, idLote As String, color As String, talla As String, key As String
    Dim dataL As Variant, dataT As Variant, dataA As Variant

    Dim estado As String

    Set wsL = ThisWorkbook.Sheets("LOTES")
    Set wsA = ThisWorkbook.Sheets("ASIGNACIONES_DIARIAS")
    Set wsTask = ThisWorkbook.Sheets("TAREAS")

            If idRef <> "" Then
                If Not dictTasks.Exists(idRef) Then Set dictTasks(idRef) = New Collection
                dictTasks(idRef).Add Array(dataT(i, 2), dataT(i, 3)) ' [0]=NumTarea, [1]=NombreTarea
            End If
        Next i
    End If

    ' 2. Cargar Asignaciones Actuales (lo que SE ESTÁ haciendo)
    Set dictAsig = CreateObject("Scripting.Dictionary")
    ultF = wsA.Cells(wsA.Rows.Count, 1).End(xlUp).Row
    If ultF >= 4 Then
        ' Columnas: ID_Ref(4), ID_Lote(6), Color(7), Talla(8), Num_Tarea(9), Estado(15)
        dataA = wsA.Range("A4:O" & ultF).Value
        For i = 1 To UBound(dataA, 1)
            key = dataA(i, 4) & "|" & dataA(i, 6) & "|" & dataA(i, 7) & "|" & dataA(i, 8) & "|" & dataA(i, 9)

        Next i

    filaDest = 4
    ultF = wsL.Cells(wsL.Rows.Count, 1).End(xlUp).Row
    If ultF >= 4 Then
        dataL = wsL.Range("A4:D" & ultF).Value
        For i = 1 To UBound(dataL, 1)
            idLote = dataL(i, 1): idRef = dataL(i, 2): color = dataL(i, 3): talla = dataL(i, 4)
            
            If dictTasks.Exists(idRef) Then
                Set colTasks = dictTasks(idRef)
                For Each taskItem In colTasks
                    key = idRef & "|" & idLote & "|" & color & "|" & talla & "|" & taskItem(0)
                    
                    estado = ""
                    If Not dictAsig.Exists(key) Then
                        estado = "SIN ASIGNAR"
                    ElseIf UCase(dictAsig(key)) <> "COMPLETADA" Then
                        estado = "PENDIENTE (" & dictAsig(key) & ")"
                    End If
                    
                    ' Si la tarea no está completada, la listamos
                    If estado <> "" Then
                        wsTraz.Cells(filaDest, 1).Value = idRef
                        wsTraz.Cells(filaDest, 2).Value = ObtenerNombreReferencia(idRef)
                        wsTraz.Cells(filaDest, 3).Value = idLote
                        wsTraz.Cells(filaDest, 4).Value = color
                        wsTraz.Cells(filaDest, 5).Value = talla
                        wsTraz.Cells(filaDest, 6).Value = taskItem(0)
                        wsTraz.Cells(filaDest, 7).Value = taskItem(1)
                        wsTraz.Cells(filaDest, 8).Value = estado

                        filaDest = filaDest + 1
                    End If
                Next taskItem
            End If
        Next i
    End If

    Call AjustarAnchoColumnasInteligente(wsTraz)
    wsTraz.Activate
    Application.ScreenUpdating = True
    MsgBox "Reporte de trazabilidad generado con éxito. Se encontraron " & (filaDest - 4) & " tareas pendientes.", vbInformation
End Sub


'==============================================================================
' SUB: GenerarReporteEficiencia
' Propósito: Compara tiempos estándar vs reales para medir desempeño por operaria.
'==============================================================================
Public Sub GenerarReporteEficiencia()
    Dim wsR As Worksheet, wsA As Worksheet, dictP As Object, colOp As Collection, dictActivas As Object
    Dim i As Long, ultF As Long, fila As Long, dLab As Long
    Dim fIni As Date, fFin As Date, fStr As String, fIter As Date
    Dim data As Variant, key As String, k As Variant, d As Variant
    Dim tProd As Double, capOp As Double, efic As Double, idOp As String
    
    fStr = InputBox("Inicio (dd/mm/yyyy):", "Reporte Eficiencia", Format(DateSerial(Year(Date), Month(Date), 1), "dd/mm/yyyy"))
    If fStr = "" Or Not IsDate(fStr) Then Exit Sub Else fIni = CDate(fStr)
    
    fStr = InputBox("Fin (dd/mm/yyyy):", "Reporte Eficiencia", Format(DateSerial(Year(fIni), Month(fIni) + 1, 0), "dd/mm/yyyy"))
    If fStr = "" Or Not IsDate(fStr) Then Exit Sub Else fFin = CDate(fStr)
    
    If fFin < fIni Then MsgBox "La fecha de fin no puede ser anterior a la fecha de inicio.", vbCritical: Exit Sub

    dLab = 0
    For fIter = fIni To fFin
        If EsDiaLaboral(fIter) Then dLab = dLab + 1
    Next fIter

    Set wsR = ThisWorkbook.Sheets("EFICIENCIA_OPERARIAS")
    Set wsA = ThisWorkbook.Sheets("ASIGNACIONES_DIARIAS")
    Set dictP = CreateObject("Scripting.Dictionary")
    
    Set dictActivas = CreateObject("Scripting.Dictionary")
    Set colOp = ObtenerAllOperarias(True)
    For Each k In colOp: dictActivas(CStr(k)) = True: Next k

    ultF = wsA.Cells(wsA.Rows.count, 1).End(xlUp).Row: If ultF < 4 Then Exit Sub
    data = wsA.Range(wsA.Cells(4, 1), wsA.Cells(ultF, 21)).Value
    For i = 1 To UBound(data, 1)
        If IsDate(data(i, 1)) Then
            If data(i, 1) >= fIni And data(i, 1) <= fFin Then
                idOp = Trim(CStr(data(i, 2)))
                
                If dictActivas.Exists(idOp) Then
                    If Not dictP.Exists(idOp) Then
                        capOp = ObtenerHorasDisponiblesOperaria(idOp) * dLab
                        dictP(idOp) = Array(ObtenerNombreOperaria(idOp), capOp, 0)
                    End If
                    
                    If data(i, 15) = "Completada" Then
                        d = dictP(idOp)
                        tProd = val(data(i, 14))
                        d(2) = d(2) + tProd
                        dictP(idOp) = d
                    End If
                End If
            End If
        End If
    Next i
    
    Application.ScreenUpdating = False
    wsR.Rows("4:" & wsR.Rows.count).ClearContents
    wsR.Rows("4:" & wsR.Rows.count).Interior.ColorIndex = xlNone
    wsR.Rows("4:" & wsR.Rows.count).Font.ColorIndex = xlAutomatic
    
    wsR.Range("A1").Value = "REPORTE DE EFICIENCIA DE OPERARIAS"
    wsR.Range("A1").Font.Bold = True: wsR.Range("A1").Font.Size = 14
    wsR.Range("A2").Value = "Periodo: " & Format(fIni, "dd/mm/yyyy") & " al " & Format(fFin, "dd/mm/yyyy")
    wsR.Range("A2").Font.Italic = True
    
    wsR.Range("A3:D3").Value = Array("Operaria", "Disponibilidad (Min)", "Producción (Min)", "Eficiencia %")
    wsR.Range("A3:D3").Font.Bold = True: wsR.Range("A3:D3").Interior.color = RGB(220, 220, 220)
    
    fila = 4
    For Each k In dictP.Keys
        d = dictP(k)
        efic = IIf(d(1) > 0, d(2) / d(1), 0)
        wsR.Cells(fila, 1).Value = d(0): wsR.Cells(fila, 2).Value = d(1): wsR.Cells(fila, 3).Value = d(2): wsR.Cells(fila, 4).Value = efic
        If efic < 0.8 Then
            wsR.Cells(fila, 4).Font.color = vbRed
        ElseIf efic > 0.95 Then
            wsR.Cells(fila, 4).Font.color = RGB(0, 128, 0)
        End If
        fila = fila + 1
    Next k
    If fila > 4 Then wsR.Range("D4:D" & fila - 1).NumberFormat = "0.00%"
    
    Call AjustarAnchoColumnasInteligente(wsR)
    wsR.Activate
    Application.ScreenUpdating = True
    MsgBox "Análisis de eficiencia completado.", vbInformation
End Sub



'==============================================================================
' SUB: GenerarInformeGerencialProduccion
' Propósito: Calcula el reporte de avance general por cada referencia
'==============================================================================
Sub GenerarInformeGerencialProduccion()
    Dim wsInforme As Worksheet, wsL As Worksheet, wsA As Worksheet, wsT As Worksheet, wsRef As Worksheet
    Dim dictRefData As Object
    Dim dictTiempoTarea As Object
    Dim dictTiempoTotalRef As Object
    Dim dictMinutosEjecutadosPorLote As Object
    Dim dictTareasPorLote As Object
    Dim dictLoteCantidad As Object
    
    Dim i As Long, j As Long, ultFila As Long
    Dim idRef As String, idLote As String, numTarea As String
    Dim cantLote As Double, uComp As Double
    Dim tiempoUnitarioMin As Double, tiempoTotalRefMin As Double
    Dim keyTarea As Variant
    Dim keyLote As Variant, keyTareaLote As Variant
    Dim minutosEjecutados As Double
    Dim fila As Long
    Dim k As Variant
    
    Set wsInforme = ThisWorkbook.Sheets("INFORME_CONSOLIDADO")
    Set wsL = ThisWorkbook.Sheets("LOTES")
    Set wsA = ThisWorkbook.Sheets("ASIGNACIONES_DIARIAS")
    Set wsT = ThisWorkbook.Sheets("TAREAS")
    Set wsRef = ThisWorkbook.Sheets("REFERENCIAS")
    
    Set dictRefData = CreateObject("Scripting.Dictionary")
    Set dictTiempoTarea = CreateObject("Scripting.Dictionary")
    Set dictTiempoTotalRef = CreateObject("Scripting.Dictionary")
    Set dictMinutosEjecutadosPorLote = CreateObject("Scripting.Dictionary")
    Set dictTareasPorLote = CreateObject("Scripting.Dictionary")
    Set dictLoteCantidad = CreateObject("Scripting.Dictionary")
    
    Application.ScreenUpdating = False
    
    '--------------------------------------------------------------------------
    ' PASO 1: Cargar tiempos unitarios por tarea (en minutos)
    '--------------------------------------------------------------------------
    ultFila = wsT.Cells(wsT.Rows.count, 1).End(xlUp).Row
    For i = 4 To ultFila
        idRef = Trim(CStr(wsT.Cells(i, 1).Value))
        numTarea = Trim(CStr(wsT.Cells(i, 2).Value))
        tiempoUnitarioMin = val(wsT.Cells(i, 4).Value) / 60
        
        If idRef <> "" And numTarea <> "" Then
            keyTarea = idRef & "|" & numTarea
            dictTiempoTarea(keyTarea) = tiempoUnitarioMin
        End If
    Next i
    
    '--------------------------------------------------------------------------
    ' PASO 2: Calcular tiempo total por referencia
    '--------------------------------------------------------------------------
    For Each keyTarea In dictTiempoTarea.Keys
        idRef = Split(CStr(keyTarea), "|")(0)
        If Not dictTiempoTotalRef.Exists(idRef) Then
            dictTiempoTotalRef(idRef) = dictTiempoTarea(keyTarea)
        Else
            dictTiempoTotalRef(idRef) = dictTiempoTotalRef(idRef) + dictTiempoTarea(keyTarea)
        End If
    Next keyTarea
    
    '--------------------------------------------------------------------------
    ' PASO 3: Cargar datos de LOTES
    '--------------------------------------------------------------------------
    ultFila = wsL.Cells(wsL.Rows.count, 1).End(xlUp).Row
    For i = 4 To ultFila
        idLote = Trim(CStr(wsL.Cells(i, 1).Value))
        idRef = Trim(CStr(wsL.Cells(i, 2).Value))
        cantLote = val(wsL.Cells(i, 5).Value)
        
        If idRef <> "" And idLote <> "" And cantLote > 0 Then
            ' Guardar cantidad del lote
            keyLote = idRef & "|" & idLote
            dictLoteCantidad(keyLote) = cantLote
            
            ' Inicializar estructura de tareas por lote
            If Not dictTareasPorLote.Exists(keyLote) Then
                Set dictTareasPorLote(keyLote) = New Collection
            End If
            
            ' Inicializar acumulador de minutos por lote
            If Not dictMinutosEjecutadosPorLote.Exists(keyLote) Then
                dictMinutosEjecutadosPorLote(keyLote) = 0
            End If
            
            ' Inicializar datos de referencia
            If Not dictRefData.Exists(idRef) Then
                Dim nombreRef As String
                Dim celdaRef As Range
                Set celdaRef = wsRef.Columns(1).Find(idRef, LookIn:=xlValues, LookAt:=xlWhole)
                If Not celdaRef Is Nothing Then
                    nombreRef = wsRef.Cells(celdaRef.Row, 3).Value
                Else
                    nombreRef = ""
                End If
                
                ' Array sin comentarios después del guion bajo
                dictRefData(idRef) = Array(nombreRef, 0, 0, 0, 0)
            End If
            
            ' Acumular cantidad total
            Dim arrTemp As Variant
            arrTemp = dictRefData(idRef)
            arrTemp(1) = arrTemp(1) + cantLote
            ' Minutos requeridos = cantidad x tiempo total de la referencia
            If dictTiempoTotalRef.Exists(idRef) Then
                arrTemp(2) = arrTemp(2) + (cantLote * dictTiempoTotalRef(idRef))
            End If
            dictRefData(idRef) = arrTemp
        End If
    Next i
    
    '--------------------------------------------------------------------------
    ' PASO 4: Procesar ASIGNACIONES_DIARIAS
    '--------------------------------------------------------------------------
    ultFila = wsA.Cells(wsA.Rows.count, 1).End(xlUp).Row
    For i = 4 To ultFila
        idRef = Trim(CStr(wsA.Cells(i, 4).Value))
        idLote = Trim(CStr(wsA.Cells(i, 6).Value))
        numTarea = Trim(CStr(wsA.Cells(i, 9).Value))
        uComp = val(wsA.Cells(i, 16).Value)
        
        If idRef <> "" And idLote <> "" And numTarea <> "" And uComp > 0 Then
            keyLote = idRef & "|" & idLote
            keyTarea = idRef & "|" & numTarea
            
            ' Obtener tiempo unitario de esta tarea
            tiempoUnitarioMin = 0
            If dictTiempoTarea.Exists(keyTarea) Then
                tiempoUnitarioMin = dictTiempoTarea(keyTarea)
            End If
            
            ' Calcular minutos ejecutados
            minutosEjecutados = uComp * tiempoUnitarioMin
            
            ' Acumular minutos ejecutados por lote
            If dictMinutosEjecutadosPorLote.Exists(keyLote) Then
                dictMinutosEjecutadosPorLote(keyLote) = dictMinutosEjecutadosPorLote(keyLote) + minutosEjecutados
            End If
            
            ' Registrar tarea en el lote
            If dictTareasPorLote.Exists(keyLote) Then
                Dim colTareas As Collection
                Set colTareas = dictTareasPorLote(keyLote)
                
                Dim tareaExiste As Boolean
                tareaExiste = False
                Dim tItem As Variant
                For Each tItem In colTareas
                    If CStr(tItem) = numTarea Then
                        tareaExiste = True
                        Exit For
                    End If
                Next tItem
                
                If Not tareaExiste Then
                    colTareas.Add numTarea
                    Set dictTareasPorLote(keyLote) = colTareas
                End If
            End If
            
            ' Acumular minutos ejecutados en dictRefData
            If dictRefData.Exists(idRef) Then
                arrTemp = dictRefData(idRef)
                arrTemp(3) = arrTemp(3) + minutosEjecutados
                dictRefData(idRef) = arrTemp
            End If
        End If
    Next i
    
    '--------------------------------------------------------------------------
    ' PASO 5: Calcular Unidades Terminadas (lotes 100% completos)
    '--------------------------------------------------------------------------
    Dim lotesKeys As Variant
    lotesKeys = dictLoteCantidad.Keys
    
    For Each keyLote In lotesKeys
        idRef = Split(keyLote, "|")(0)
        idLote = Split(keyLote, "|")(1)
        cantLote = dictLoteCantidad(keyLote)
        
        ' Obtener todas las tareas que DEBERÍA tener esta referencia
        Dim tareasRequeridas As New Collection
        Dim tk As Variant
        For Each tk In dictTiempoTarea.Keys
            If Left(CStr(tk), Len(idRef)) = idRef And Mid(CStr(tk), Len(idRef) + 1, 1) = "|" Then
                Dim tareaNumReq As String
                tareaNumReq = Split(CStr(tk), "|")(1)
                tareasRequeridas.Add tareaNumReq
            End If
        Next tk
        
        If tareasRequeridas.count = 0 Then GoTo SiguienteLote
        
        ' Verificar si TODAS las tareas requeridas están completas al 100%
        Dim loteCompleto As Boolean
        loteCompleto = True
        
        Dim tareaReq As Variant
        For Each tareaReq In tareasRequeridas
            Dim keyTareaReq As String
            keyTareaReq = idRef & "|" & tareaReq
            
            ' Obtener tiempo unitario de esta tarea
            Dim tiempoUnitarioTarea As Double
            tiempoUnitarioTarea = 0
            If dictTiempoTarea.Exists(keyTareaReq) Then
                tiempoUnitarioTarea = dictTiempoTarea(keyTareaReq)
            End If
            
            ' Minutos necesarios para completar esta tarea al 100%
            Dim minutosNecesarios As Double
            minutosNecesarios = cantLote * tiempoUnitarioTarea
            
            ' Calcular minutos reales ejecutados para esta tarea específica
            Dim minutosRealesTarea As Double
            minutosRealesTarea = 0
            
            For j = 4 To wsA.Cells(wsA.Rows.count, 1).End(xlUp).Row
                If Trim(CStr(wsA.Cells(j, 4).Value)) = idRef And _
                   Trim(CStr(wsA.Cells(j, 6).Value)) = idLote And _
                   Trim(CStr(wsA.Cells(j, 9).Value)) = tareaReq Then
                    
                    Dim uCompTarea As Double
                    uCompTarea = val(wsA.Cells(j, 16).Value)
                    minutosRealesTarea = minutosRealesTarea + (uCompTarea * tiempoUnitarioTarea)
                End If
            Next j
            
            If minutosRealesTarea < minutosNecesarios - 0.001 Then
                loteCompleto = False
                Exit For
            End If
        Next tareaReq
        
        If loteCompleto And tareasRequeridas.count > 0 Then
            If dictRefData.Exists(idRef) Then
                arrTemp = dictRefData(idRef)
                arrTemp(4) = arrTemp(4) + cantLote
                dictRefData(idRef) = arrTemp
            End If
        End If
            
SiguienteLote:
        Next keyLote
        
        '--------------------------------------------------------------------------
        ' PASO 6: Generar informe visual
        '--------------------------------------------------------------------------
        wsInforme.Rows("4:" & wsInforme.Rows.count).Clear
        wsInforme.Rows("4:" & wsInforme.Rows.count).Interior.color = xlNone
        
        wsInforme.Range("B4").Value = "INFORME DETALLADO DE PRODUCCIÓN"
        wsInforme.Range("B4").Font.Size = 14
        wsInforme.Range("B4").Font.Bold = True
        
        wsInforme.Range("B5").Value = Format(Date, "mmmm d \d\e yyyy")
        wsInforme.Range("B5").Font.Italic = True
        
        fila = 7
        
        For Each k In dictRefData.Keys
            arrTemp = dictRefData(k)
            
            Dim totalCantidad As Double
            Dim totalMinutosReq As Double
            Dim totalMinutosEjec As Double
            Dim unidTerm As Double
            Dim avanceRef As Double
            
            totalCantidad = CDbl(arrTemp(1))
            totalMinutosReq = CDbl(arrTemp(2))
            totalMinutosEjec = CDbl(arrTemp(3))
            unidTerm = CDbl(arrTemp(4))
            
            avanceRef = 0
            If totalMinutosReq > 0 Then
                avanceRef = Round(totalMinutosEjec / totalMinutosReq, 4)
            End If
            
            wsInforme.Cells(fila, 2).Value = k
            wsInforme.Cells(fila, 2).Font.Bold = True
            fila = fila + 1
            
            wsInforme.Cells(fila, 2).Value = "Descripción: " & arrTemp(0)
            fila = fila + 1
            
            wsInforme.Cells(fila, 2).Value = "Cantidad Recibida: " & totalCantidad
            fila = fila + 1
            
            wsInforme.Cells(fila, 2).Value = "Avance:"
            With wsInforme.Cells(fila, 3)
                .Value = avanceRef
                .NumberFormat = "0.00%"
                .FormatConditions.Delete
                .FormatConditions.AddDatabar
            End With
            fila = fila + 1
            
            wsInforme.Cells(fila, 2).Value = "Unidades terminadas: " & Format(unidTerm, "#,##0")
            fila = fila + 2
        Next k
        
        wsInforme.Columns("B").AutoFit
        wsInforme.Activate
        Application.ScreenUpdating = True
        MsgBox "Informe de producción generado (corregido).", vbInformation

End Sub
