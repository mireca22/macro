' FUNCIONES GENERALES DE VALIDACION Y FORMATEO
Public Function IsNumericValue(valor As Variant) As Boolean
    On Error Resume Next
    IsNumericValue = IsNumeric(valor)
    On Error GoTo 0
End Function

Public Function ConvertirANumero(valor As Variant) As Long
    On Error Resume Next
    ConvertirANumero = CLng(valor)
    On Error GoTo 0
End Function

Public Function FormatearSegundosAMinutos(segundos As Long) As String
    Dim minutos As Long, segs As Long
    minutos = segundos \ 60
    segs = segundos Mod 60
    FormatearSegundosAMinutos = Format(minutos, "00") & ":" & Format(segs, "00")
End Function

Public Function ObtenerDiaDelaSemana(fecha As Date) As String
    Dim dias As String
    dias = Split("Domingo,Lunes,Martes,Miercoles,Jueves,Viernes,Sabado", ",")
    ObtenerDiaDelaSemana = dias(Weekday(fecha) - 1)
End Function

Public Function CalcularPorcentajeCompletado(completadas As Long, total As Long) As String
    If total = 0 Then
        CalcularPorcentajeCompletado = "0%"
    Else
        CalcularPorcentajeCompletado = Format(completadas / total, "0%")
    End If
End Function

Public Sub LimpiarControles(frm As Object)
    Dim ctrl As Control
    For Each ctrl In frm.Controls
        Select Case TypeName(ctrl)
            Case "TextBox": ctrl.Value = ""
            Case "ComboBox": ctrl.ListIndex = -1: ctrl.Value = ""
            Case "CheckBox": ctrl.Value = False
            Case "SpinButton": ctrl.Value = 0
        End Select
    Next ctrl
End Sub

Public Function ObtenerProximoColorFila(fila As Long) As Long
    If fila Mod 2 = 0 Then
        ObtenerProximoColorFila = RGB(240, 240, 240)
    Else
        ObtenerProximoColorFila = RGB(255, 255, 255)
    End If
End Function

' FORMATEO Y PRESENTACION VISUAL
Public Sub AjustarAnchoColumnasInteligente(ws As Worksheet)
    Dim col As Range
    Dim maxAncho As Double
    maxAncho = 50
    On Error Resume Next
    ws.UsedRange.Columns.AutoFit
    For Each col In ws.UsedRange.Columns
        If col.ColumnWidth > maxAncho Then
            col.ColumnWidth = maxAncho
            col.WrapText = True
        Else
            col.WrapText = False
        End If
    Next col
    Dim lastRow As Long
    lastRow = ws.UsedRange.Rows.Count + ws.UsedRange.Row - 1
    If lastRow > 3 Then
        ws.Rows(3, lastRow).EntireRow.AutoFit
    End If
    On Error GoTo 0
End Sub

Public Sub AplicarFormatoAsignaciones()
    Dim ws As Worksheet
    Dim fila As Long, ultFila As Long
    Set ws = ThisWorkbook.Sheets("ASIGNACIONES_DIARIAS")
    Application.ScreenUpdating = False
    Call AjustarAnchoColumnasInteligente(ws)
    ultFila = ws.Cells(ws.Rows.Count, 1).End(xlUp).Row
    For fila = 4 To ultFila
        ws.Rows(fila).Interior.Color = ObtenerProximoColorFila(fila)
    Next fila
    ' Centrar columnas clave
    ws.Columns("B:B").HorizontalAlignment = xlCenter
    ws.Columns("C:C").HorizontalAlignment = xlCenter
    ws.Columns("E:E").HorizontalAlignment = xlCenter
    ws.Columns("G:G").HorizontalAlignment = xlCenter
    ws.Columns("H:H").HorizontalAlignment = xlCenter
    ws.Columns("J:K").HorizontalAlignment = xlCenter
    ws.Columns("N:O").HorizontalAlignment = xlCenter
    ws.Columns("L:M").NumberFormat = "#,0"
    Application.ScreenUpdating = True
    MsgBox "✅ Formato aplicado.", vbInformation
End Sub

' ESTADISTICAS Y REPORTES
Public Function ObtenerEstadisticasHoy() As Variant
    Dim ws As Worksheet
    Dim fila As Long
    Dim stats(1 To 4) As Variant
    Dim totalMin As Double
    Dim completadas As Long, pendientes As Long, contador As Long
    Set ws = ThisWorkbook.Sheets("ASIGNACIONES_DIARIAS")
    For fila = 4 To ws.Cells(ws.Rows.Count, 1).End(xlUp).Row
        If Int(ws.Cells(fila, 1).Value) = Int(Date) Then
            contador = contador + 1
            totalMin = totalMin + ws.Cells(fila, 14).Value
            If ws.Cells(fila, 15).Value = "Completada" Then
                completadas = completadas + 1
            Else
                pendientes = pendientes + 1
            End If
        End If
    Next fila
    stats(1) = contador
    stats(2) = CLng(totalMin)
    stats(3) = completadas
    stats(4) = pendientes
    ObtenerEstadisticasHoy = stats
End Function

' ✅ GENERADOR PRINCIPAL DE REPORTES - HEADERS EN FILA 3, DATOS EN FILA 4+
Public Sub GenerarReporteEnHoja(fecha As Date, tipoReporte As String)
    Dim wsResumen As Worksheet, wsAsig As Worksheet, wsOp As Worksheet
    Dim dictOp As Object
    Dim i As Long, ultFila As Long
    Dim idOp As String, nomOp As String
    Dim cap As Double, asig As Double
    Dim plan As Long, ejec As Long
    Dim fechaFila As Date
    Dim esMes As Boolean
    Dim diasLaborales As Integer

    Set wsResumen = ThisWorkbook.Sheets("RESUMENDIARIO")
    Set wsAsig = ThisWorkbook.Sheets("ASIGNACIONES_DIARIAS")
    Set wsOp = ThisWorkbook.Sheets("OPERARIAS")
    Set dictOp = CreateObject("Scripting.Dictionary")

    esMes = (tipoReporte = "MENSUAL")
    diasLaborales = 22

    Application.ScreenUpdating = False
    Application.EnableEvents = False

    ' 1. Cargar Operarias
    ultFila = wsOp.Cells(wsOp.Rows.Count, 1).End(xlUp).Row
    For i = 4 To ultFila
        idOp = wsOp.Cells(i, 1).Value
        nomOp = wsOp.Cells(i, 2).Value
        cap = Val(wsOp.Cells(i, 3).Value)
        If esMes Then
            cap = cap * diasLaborales
        End If
        If idOp <> "" Then
            dictOp(idOp) = Array(nomOp, cap, 0, 0, 0)
        End If
    Next i

    ' 2. Leer ASIGNACIONES_DIARIAS completo en array
    Dim collDetalles As Collection
    Set collDetalles = New Collection
    Dim rowData As Variant
    Dim incluirFila As Boolean
    ultFila = wsAsig.Cells(wsAsig.Rows.Count, 1).End(xlUp).Row

    If ultFila > 4 Then
        Dim dataArr As Variant
        Dim totalFilas As Long
        dataArr = wsAsig.Range(wsAsig.Cells(4, 1), wsAsig.Cells(ultFila, 21)).Value
        totalFilas = UBound(dataArr, 1)

        For i = 1 To totalFilas
            If IsDate(dataArr(i, 1)) Then
                fechaFila = dataArr(i, 1)
                incluirFila = False
                If esMes Then
                    If Month(fechaFila) = Month(fecha) And Year(fechaFila) = Year(fecha) Then
                        incluirFila = True
                    End If
                Else
                    If Int(fechaFila) = Int(fecha) Then
                        incluirFila = True
                    End If
                End If

                If incluirFila Then
                    idOp = dataArr(i, 2)
                    Dim minAsig As Double, minReal As Double
                    minAsig = Val(dataArr(i, 14))
                    minReal = Val(dataArr(i, 20))

                    If dictOp.Exists(idOp) Then
                        Dim arrOp As Variant
                        arrOp = dictOp(idOp)
                        arrOp(2) = arrOp(2) + minAsig
                        If dataArr(i, 15) = "Completada" Then
                            arrOp(3) = arrOp(3) + 1
                        End If
                        arrOp(4) = arrOp(4) + minReal
                        dictOp(idOp) = arrOp
                    End If
                End If
            End If
        Next i
    End If

    ' 3. Escribir reporte - ✅ HEADERS EN FILA 3
    wsResumen.Cells.Clear
    wsResumen.Range("A1:B2").Clear

    ' Encabezados de título
    wsResumen.Range("A3").Value = "Operaria"
    wsResumen.Range("B3").Value = "Capacidad min"
    wsResumen.Range("C3").Value = "Asignado min"
    wsResumen.Range("D3").Value = "Ocupación %"
    wsResumen.Range("E3").Value = "Disponible min"
    wsResumen.Range("F3").Value = "Estado"

    With wsResumen.Range("A3:F3")
        .Font.Bold = True
        .Interior.Color = RGB(220, 220, 220)
        .Borders.LineStyle = xlContinuous
    End With

    ' ✅ Datos comienzan en FILA 4
    Dim fila As Long
    Dim k As Variant
    fila = 4
    For Each k In dictOp.Keys
        Dim d As Variant
        d = dictOp(k)

        Dim ocu As Double, dispon As Double, estado As String
        ocu = IIf(d(1) > 0, d(2) / d(1), 0)
        dispon = d(1) - d(2)

        If dispon > 0 Then
            estado = "DISPONIBLE"
        ElseIf ocu <= 1 Then
            estado = "ÓPTIMA"
        Else
            estado = "SOBRECARGA"
        End If

        wsResumen.Cells(fila, 1).Value = d(0)
        wsResumen.Cells(fila, 2).Value = d(1)
        wsResumen.Cells(fila, 3).Value = d(2)
        wsResumen.Cells(fila, 4).Value = Format(ocu, "0%")
        wsResumen.Cells(fila, 5).Value = dispon
        wsResumen.Cells(fila, 6).Value = estado

        ' Colorear estado
        Select Case estado
            Case "DISPONIBLE"
                wsResumen.Cells(fila, 6).Interior.Color = RGB(0, 176, 80)
            Case "ÓPTIMA"
                wsResumen.Cells(fila, 6).Interior.Color = RGB(0, 176, 80)
            Case "SOBRECARGA"
                wsResumen.Cells(fila, 6).Interior.Color = RGB(255, 0, 0)
        End Select

        fila = fila + 1
    Next k

    wsResumen.Columns.AutoFit
    wsResumen.Activate
    Application.EnableEvents = True
    Application.ScreenUpdating = True
    MsgBox "✅ Reporte diario generado correctamente.", vbInformation
End Sub

' ✅ GENERADOR REPORTE EFICIENCIA - HEADERS EN FILA 3, DATOS EN FILA 4+
Public Sub GenerarReporteEficiencia()
    Dim wsReporte As Worksheet, wsAsig As Worksheet
    Dim dictPerf As Object
    Dim i As Long, ultFila As Long
    Dim fechaInicio As Date, fechaFin As Date
    Dim fechaStr As String

    fechaStr = InputBox("Fecha de INICIO del reporte (dd/mm/yyyy):", "Reporte de Eficiencia", Format(Date - 30, "dd/mm/yyyy"))
    If fechaStr = "" Then Exit Sub
    If Not IsDate(fechaStr) Then Exit Sub
    fechaInicio = CDate(fechaStr)

    fechaStr = InputBox("Fecha de FIN del reporte (dd/mm/yyyy):", "Reporte de Eficiencia", Format(Date, "dd/mm/yyyy"))
    If fechaStr = "" Then Exit Sub
    If Not IsDate(fechaStr) Then Exit Sub
    fechaFin = CDate(fechaStr)

    Set wsReporte = ThisWorkbook.Sheets("ANALISISTIEMPOS")
    Set wsAsig = ThisWorkbook.Sheets("ASIGNACIONES_DIARIAS")
    Set dictPerf = CreateObject("Scripting.Dictionary")

    Application.ScreenUpdating = False
    Application.EnableEvents = False

    ' Leer toda la hoja en un array
    ultFila = wsAsig.Cells(wsAsig.Rows.Count, 1).End(xlUp).Row
    If ultFila > 4 Then
        Dim dataArr As Variant
        Dim totalFilas As Long
        dataArr = wsAsig.Range(wsAsig.Cells(4, 1), wsAsig.Cells(ultFila, 21)).Value
        totalFilas = UBound(dataArr, 1)

        For i = 1 To totalFilas
            If IsDate(dataArr(i, 1)) Then
                Dim fechaAsig As Date
                fechaAsig = dataArr(i, 1)
                If fechaAsig >= fechaInicio And fechaAsig <= fechaFin Then
                    Dim idOp As String, nomOp As String
                    Dim idRef As String, nomRef As String
                    Dim numTarea As String, nomTarea As String
                    Dim key As String

                    idOp = dataArr(i, 2)
                    nomOp = dataArr(i, 3)
                    idRef = dataArr(i, 4)
                    nomRef = dataArr(i, 5)
                    numTarea = dataArr(i, 9)
                    nomTarea = dataArr(i, 10)
                    key = idOp & "|" & idRef & "|" & numTarea

                    Dim minAsig As Double, unidComp As Long, tiempoUnitSeg As Long, minProd As Double
                    minAsig = Val(dataArr(i, 14))
                    unidComp = CLng(Val(dataArr(i, 16)))
                    tiempoUnitSeg = CLng(Val(dataArr(i, 12)))
                    minProd = unidComp * tiempoUnitSeg / 60

                    If Not dictPerf.Exists(key) Then
                        dictPerf(key) = Array(nomOp, nomRef, nomTarea, 0, 0)
                    End If
                    Dim arrP As Variant
                    arrP = dictPerf(key)
                    arrP(3) = arrP(3) + minAsig
                    arrP(4) = arrP(4) + minProd
                    dictPerf(key) = arrP
                End If
            End If
        Next i
    End If

    ' ✅ Escribir reporte - HEADERS EN FILA 3
    wsReporte.Cells.Clear
    wsReporte.Range("A3:G3").Value = Array("Operaria", "Referencia", "Tarea", "Min. Asignados", _
                                           "Min. Producidos Est.", "Desviación min", "Eficiencia")
    With wsReporte.Range("A3:G3")
        .Font.Bold = True
        .Interior.Color = RGB(220, 220, 220)
        .Borders.LineStyle = xlContinuous
    End With

    ' ✅ Datos comienzan en FILA 4
    Dim fila As Long
    Dim k As Variant
    fila = 4
    For Each k In dictPerf.Keys
        Dim d As Variant
        d = dictPerf(k)
        wsReporte.Cells(fila, 1).Value = d(0)
        wsReporte.Cells(fila, 2).Value = d(1)
        wsReporte.Cells(fila, 3).Value = d(2)
        wsReporte.Cells(fila, 4).Value = d(3)
        wsReporte.Cells(fila, 5).Value = d(4)
        wsReporte.Cells(fila, 6).Value = d(4) - d(3)
        If d(3) > 0 Then
            wsReporte.Cells(fila, 7).Value = d(4) / d(3)
        End If
        fila = fila + 1
    Next k

    wsReporte.Range("G4:G" & fila).NumberFormat = "0.0%"
    wsReporte.Columns.AutoFit
    wsReporte.Activate
    Application.EnableEvents = True
    Application.ScreenUpdating = True
    MsgBox "✅ Reporte de Eficiencia generado con " & dictPerf.Count & " registros.", vbInformation
End Sub

' ✅ GENERADOR AVANCE PRODUCCION - HEADERS EN FILA 3, DATOS EN FILA 4+
Public Sub GenerarReporteAvanceProduccion()
    Dim wsAvance As Worksheet, wsRecep As Worksheet
    Dim wsTareas As Worksheet, wsAsig As Worksheet
    Dim dictUnidades As Object, dictAsignado As Object
    Dim i As Long, ultFila As Long
    Dim idRef As String, key As String, cant As Long

    Set wsAvance = ThisWorkbook.Sheets("AVANCEPRODUCCION")
    Set wsRecep = ThisWorkbook.Sheets("LOTES")
    Set wsTareas = ThisWorkbook.Sheets("TAREAS")
    Set wsAsig = ThisWorkbook.Sheets("ASIGNACIONES_DIARIAS")
    Set dictUnidades = CreateObject("Scripting.Dictionary")
    Set dictAsignado = CreateObject("Scripting.Dictionary")

    Application.ScreenUpdating = False

    ' 1. Total por Referencia desde LOTES
    ultFila = wsRecep.Cells(wsRecep.Rows.Count, 1).End(xlUp).Row
    For i = 4 To ultFila
        idRef = wsRecep.Cells(i, 2).Value
        cant = wsRecep.Cells(i, 5).Value
        If dictUnidades.Exists(idRef) Then
            dictUnidades(idRef) = dictUnidades(idRef) + cant
        Else
            dictUnidades(idRef) = cant
        End If
    Next i

    ' 2. Lo asignado por RefTarea
    ultFila = wsAsig.Cells(wsAsig.Rows.Count, 1).End(xlUp).Row
    For i = 4 To ultFila
        idRef = wsAsig.Cells(i, 4).Value
        Dim numTarea As String
        numTarea = wsAsig.Cells(i, 9).Value
        cant = wsAsig.Cells(i, 11).Value
        key = idRef & "|" & numTarea
        If dictAsignado.Exists(key) Then
            dictAsignado(key) = dictAsignado(key) + cant
        Else
            dictAsignado(key) = cant
        End If
    Next i

    ' 3. ✅ Generar Reporte - HEADERS EN FILA 3
    wsAvance.Cells.Clear
    wsAvance.Range("A3:I3").Value = Array("ID Referencia", "Nombre Referencia", "Nombre Tarea", _
                                          "Tiempo Unit seg", "Total Unidades Req.", "Unidades Asignadas", _
                                          "Unidades Pendientes", "Tiempo Pendiente min", "Estado Asignacion")
    With wsAvance.Range("A3:I3")
        .Font.Bold = True
        .Interior.Color = RGB(220, 220, 220)
        .Borders.LineStyle = xlContinuous
    End With

    ' ✅ Datos comienzan en FILA 4
    Dim filaReporte As Long
    filaReporte = 4
    ultFila = wsTareas.Cells(wsTareas.Rows.Count, 1).End(xlUp).Row
    For i = 4 To ultFila
        idRef = wsTareas.Cells(i, 1).Value
        Dim numT As String, nomT As String, tiempoT As Long
        numT = wsTareas.Cells(i, 2).Value
        nomT = wsTareas.Cells(i, 3).Value
        tiempoT = wsTareas.Cells(i, 4).Value

        Dim totalReq As Long, totalAsig As Long, pendiente As Long
        totalReq = IIf(dictUnidades.Exists(idRef), dictUnidades(idRef), 0)
        key = idRef & "|" & numT
        totalAsig = IIf(dictAsignado.Exists(key), dictAsignado(key), 0)
        pendiente = IIf(totalReq - totalAsig > 0, totalReq - totalAsig, 0)

        Dim tiempoPendMin As Double
        tiempoPendMin = pendiente * tiempoT / 60

        Dim estado As String
        If totalReq = 0 Then
            estado = "Sin Tallas"
        ElseIf totalAsig >= totalReq Then
            estado = "Completo"
        ElseIf totalAsig = 0 Then
            estado = "En Curso"
        Else
            estado = "Pendiente"
        End If

        wsAvance.Range("A" & filaReporte & ":I" & filaReporte).Value = Array(idRef, _
                                                                             ObtenerNombreReferencia(idRef), nomT, tiempoT, totalReq, totalAsig, pendiente, tiempoPendMin, estado)
        filaReporte = filaReporte + 1
    Next i

    wsAvance.Columns.AutoFit
    wsAvance.Activate
    Application.ScreenUpdating = True
    MsgBox "✅ Reporte de Avance de Producción generado.", vbInformation
End Sub

' ✅ GENERADOR REPORTE LOTES - HEADERS EN FILA 3, DATOS EN FILA 4+
Public Sub GenerarReporteLotes()
    Dim wsReporte As Worksheet, wsRecep As Worksheet
    Dim wsAsig As Worksheet, wsTareas As Worksheet, wsRef As Worksheet
    Dim dictRefTasks As Object, dictRefTime As Object, dictOpsDone As Object
    Dim i As Long, ultFila As Long
    Dim idRef As String, idLote As String, color As String, talla As String, key As String
    Dim cantLote As Long, tasksCount As Long, timeUnit As Long
    Dim totalOpsReq As Long, opsDone As Long
    Dim avance As Double, tiempoTotalHoras As Double, estado As String

    Set wsReporte = ThisWorkbook.Sheets("REPORTELOTES")
    Set wsRecep = ThisWorkbook.Sheets("LOTES")
    Set wsAsig = ThisWorkbook.Sheets("ASIGNACIONES_DIARIAS")
    Set wsTareas = ThisWorkbook.Sheets("TAREAS")
    Set wsRef = ThisWorkbook.Sheets("REFERENCIAS")
    Set dictRefTasks = CreateObject("Scripting.Dictionary")
    Set dictRefTime = CreateObject("Scripting.Dictionary")
    Set dictOpsDone = CreateObject("Scripting.Dictionary")

    Application.ScreenUpdating = False

    ' 1. Tareas por Ref
    ultFila = wsTareas.Cells(wsTareas.Rows.Count, 1).End(xlUp).Row
    For i = 4 To ultFila
        idRef = wsTareas.Cells(i, 1).Value
        If idRef <> "" Then
            If dictRefTasks.Exists(idRef) Then
                dictRefTasks(idRef) = dictRefTasks(idRef) + 1
            Else
                dictRefTasks(idRef) = 1
            End If
        End If
    Next i

    ' 2. Tiempo total por Ref
    ultFila = wsRef.Cells(wsRef.Rows.Count, 1).End(xlUp).Row
    For i = 4 To ultFila
        idRef = wsRef.Cells(i, 1).Value
        If idRef <> "" Then
            dictRefTime(idRef) = Val(wsRef.Cells(i, 5).Value)
        End If
    Next i

    ' 3. Unidades completadas por lote
    ultFila = wsAsig.Cells(wsAsig.Rows.Count, 1).End(xlUp).Row
    For i = 4 To ultFila
        idRef = wsAsig.Cells(i, 4).Value
        idLote = wsAsig.Cells(i, 6).Value
        color = wsAsig.Cells(i, 7).Value
        talla = wsAsig.Cells(i, 8).Value
        Dim comp As Long
        comp = Val(wsAsig.Cells(i, 16).Value)
        key = idLote & "|" & idRef & "|" & color & "|" & talla
        If dictOpsDone.Exists(key) Then
            dictOpsDone(key) = dictOpsDone(key) + comp
        Else
            dictOpsDone(key) = comp
        End If
    Next i

    ' 4. ✅ Reporte final - HEADERS EN FILA 3
    wsReporte.Cells.Clear
    wsReporte.Range("A3:K3").Value = Array("Lote", "Referencia", "Color", "Talla", "Cantidad", _
                                           "TareasUnidad", "Ops. Requeridas", "Ops. Completadas", _
                                           "Avance %", "Tiempo Est. Horas", "Estado")
    With wsReporte.Range("A3:K3")
        .Font.Bold = True
        .Interior.Color = RGB(220, 220, 220)
        .Borders.LineStyle = xlContinuous
    End With

    ' ✅ Datos comienzan en FILA 4
    Dim filaR As Long
    filaR = 4
    ultFila = wsRecep.Cells(wsRecep.Rows.Count, 1).End(xlUp).Row
    For i = 4 To ultFila
        idLote = wsRecep.Cells(i, 1).Value
        idRef = wsRecep.Cells(i, 2).Value
        color = wsRecep.Cells(i, 3).Value
        talla = wsRecep.Cells(i, 4).Value
        cantLote = Val(wsRecep.Cells(i, 5).Value)

        If idLote <> "" Then
            tasksCount = IIf(dictRefTasks.Exists(idRef), dictRefTasks(idRef), 0)
            timeUnit = IIf(dictRefTime.Exists(idRef), dictRefTime(idRef), 0)
            totalOpsReq = cantLote * tasksCount
            key = idLote & "|" & idRef & "|" & color & "|" & talla
            opsDone = IIf(dictOpsDone.Exists(key), dictOpsDone(key), 0)
            avance = IIf(totalOpsReq > 0, opsDone / totalOpsReq, 0)
            tiempoTotalHoras = cantLote * timeUnit / 3600

            If avance >= 1 Then
                estado = "COMPLETADO"
                wsReporte.Cells(filaR, 11).Font.Color = RGB(0, 150, 0)
            ElseIf avance > 0 Then
                estado = "EN PROCESO"
                wsReporte.Cells(filaR, 11).Font.Color = RGB(255, 140, 0)
            Else
                estado = "PENDIENTE"
            End If

            wsReporte.Range("A" & filaR & ":K" & filaR).Value = Array(idLote, idRef, color, talla, cantLote, _
                                                                      tasksCount, totalOpsReq, opsDone, avance, Round(tiempoTotalHoras, 2), estado)
            filaR = filaR + 1
        End If
    Next i

    wsReporte.Range("I4:I" & filaR).NumberFormat = "0.0%"
    wsReporte.Columns.AutoFit
    wsReporte.Activate
    Application.ScreenUpdating = True
    MsgBox "✅ Reporte de Lotes generado exitosamente.", vbInformation
End Sub

' ✅ GENERADOR INFORME PRODUCCION - HEADERS EN FILA 3, DATOS EN FILA 4+
Public Sub GenerarInformeProduccion()
    Dim wsProd As Worksheet, wsRecep As Worksheet
    Dim dictAvance As Object
    Dim i As Long, ultFila As Long
    Dim idRef As String, nomRef As String, key As String
    Dim unidTotal As Long, unidCompletadas As Long, diasProduccion As Integer

    Set wsProd = ThisWorkbook.Sheets("PRODUCCIONFINALIZADA")

    ' Leer estado de LOTES y agrupar
    ultFila = wsRecep.Cells(wsRecep.Rows.Count, 1).End(xlUp).Row
    For i = 4 To ultFila
        idRef = wsRecep.Cells(i, 2).Value
        If idRef <> "" Then
            If Not dictAvance.Exists(idRef) Then
                dictAvance(idRef) = Array(ObtenerNombreReferencia(idRef), 0, 0, 0, 0)
            End If
            Dim arrRef As Variant
            arrRef = dictAvance(idRef)
            arrRef(1) = arrRef(1) + Val(wsRecep.Cells(i, 5).Value)
            arrRef(2) = arrRef(2) + Val(wsRecep.Cells(i, 8).Value)
            arrRef(3) = arrRef(3) + 1
            dictAvance(idRef) = arrRef
        End If
    Next i

    ' ✅ Escribir Informe - HEADERS EN FILA 3
    wsProd.Cells.Clear
    wsProd.Range("A3:F3").Value = Array("Fecha Cierre", "ID Referencia", "Nombre Referencia", _
                                        "Total Unidades", "Unidades Completadas", "Porcentaje")
    With wsProd.Range("A3:F3")
        .Font.Bold = True
        .Interior.Color = RGB(220, 220, 220)
        .Borders.LineStyle = xlContinuous
    End With

    ' ✅ Datos comienzan en FILA 4
    Dim filaInforme As Long
    filaInforme = 4
    Dim k As Variant
    For Each k In dictAvance.Keys
        Dim d As Variant
        d = dictAvance(k)
        Dim porcentaje As Double
        If d(1) > 0 Then
            porcentaje = d(2) / d(1)
        Else
            porcentaje = 0
        End If

    wsProd.Activate

' FUNCIONES AUXILIARES PARAMETROS, FECHAS ESTIMADAS
Public Function ObtenerParametro(nombreParam As String) As Variant
    Dim ws As Worksheet
    Dim celda As Range
    On Error Resume Next
    Set ws = ThisWorkbook.Sheets("PARAMETROS")
    Set celda = ws.Columns("A").Find(nombreParam, LookIn:=xlValues, LookAt:=xlWhole)
    If Not celda Is Nothing Then
        ObtenerParametro = celda.Offset(0, 1).Value
    Else
        ObtenerParametro = 0
    End If
    On Error GoTo 0
End Function

Public Function CalcularMinutosDiariosBase() As Long
    Dim horas As Double, dias As Double
    horas = ObtenerParametro("HorasSemanalesLegales")
    dias = ObtenerParametro("DiasLaboralesSemana")
    If horas = 0 Then horas = 44
    If dias = 0 Then dias = 6
    CalcularMinutosDiariosBase = CLng(horas * 60 / dias)
End Function

Public Function GetCapacidadDiariaTotal() As Long
    Dim wsOp As Worksheet
    Dim i As Long, ultFila As Long
    Dim capacidadTotal As Long
    Set wsOp = ThisWorkbook.Sheets("OPERARIAS")
    ultFila = wsOp.Cells(wsOp.Rows.Count, 1).End(xlUp).Row
    For i = 4 To ultFila
        If wsOp.Cells(i, 4).Value = "Activa" Then
            capacidadTotal = capacidadTotal + Val(wsOp.Cells(i, 3).Value)
        End If
    Next i
    GetCapacidadDiariaTotal = capacidadTotal
End Function

Public Function CalcularFechaFinEstimada(idRef As String, cantidadTotal As Long, fechaInicio As Date) As Date
    Dim tiempoUnitTotalSeg As Long
    Dim minutosRequeridos As Double
    Dim capacidadDiariaMin As Long
    Dim diasNecesarios As Double
    Dim fechaFin As Date
    Dim diasContados As Integer

    tiempoUnitTotalSeg = ObtenerTiempoTotalUnidad(idRef)
    If tiempoUnitTotalSeg = 0 Then
        MsgBox "La referencia " & idRef & " no tiene tareas o tiempos definidos.", vbExclamation
        CalcularFechaFinEstimada = 0
        Exit Function
    End If

    minutosRequeridos = CDbl(cantidadTotal) * CDbl(tiempoUnitTotalSeg) / 60
    capacidadDiariaMin = GetCapacidadDiariaTotal()

    If capacidadDiariaMin = 0 Then
        MsgBox "No hay operarias activas. No se puede estimar la fecha.", vbExclamation
        CalcularFechaFinEstimada = 0
        Exit Function
    End If

    diasNecesarios = minutosRequeridos / capacidadDiariaMin
    fechaFin = fechaInicio
    diasContados = 0

    Do While diasContados < diasNecesarios
        Do While Not EsDiaLaboral(fechaFin)
            fechaFin = fechaFin + 1
        Loop
        diasContados = diasContados + 1
        If diasContados < diasNecesarios Then
            fechaFin = fechaFin + 1
        End If
    Loop

    CalcularFechaFinEstimada = fechaFin
End Function

' CALENDARIO DIAS NO LABORALES Y FESTIVOS COLOMBIA
Public Sub ActualizarCalendarioNoLaboral()
    Dim ws As Worksheet
    Dim ultFila As Long
    Dim fechaInicio As Date, fechaFin As Date, fechaIter As Date
    Dim descripcion As String
    Dim contador As Long

    Set ws = ThisWorkbook.Sheets("DIASNOLABORALES")
    ultFila = ws.Cells(ws.Rows.Count, 1).End(xlUp).Row

    If ultFila > 3 Then
        fechaInicio = ws.Cells(ultFila, 1).Value + 1
    Else
        fechaInicio = DateSerial(Year(Date), 1, 1)
    End If

    fechaFin = DateSerial(Year(Date), 12, 31)
    If fechaInicio > fechaFin Then
        MsgBox "INFO: El calendario ya está actualizado hasta " & fechaFin, vbInformation
        Exit Sub
    End If

    Application.ScreenUpdating = False
    contador = 0

    For fechaIter = fechaInicio To fechaFin
        descripcion = ""
        If Weekday(fechaIter, vbMonday) = 6 Then
            descripcion = "Sábado"
        End If
        If Weekday(fechaIter, vbMonday) = 7 Then
            descripcion = "Domingo"
        End If

        Dim festivo As String
        festivo = ObtenerFestivoColombia(fechaIter)
        If festivo <> "" Then
            If descripcion <> "" Then
                descripcion = descripcion & " - " & festivo
            Else
                descripcion = festivo
            End If
        End If

        If descripcion <> "" Then
            ultFila = ultFila + 1
            ws.Cells(ultFila, 1).Value = fechaIter
            ws.Cells(ultFila, 2).Value = Format(fechaIter, "dddd")
            ws.Cells(ultFila, 3).Value = descripcion
            contador = contador + 1
        End If
    Next fechaIter

    ws.Columns.AutoFit
    Application.ScreenUpdating = True
    MsgBox "✅ Calendario actualizado. Se agregaron " & contador & " días no laborales.", vbInformation
End Sub

Public Function EsDiaLaboral(fecha As Date) As Boolean
    Dim ws As Worksheet
    Dim res As Variant
    Set ws = ThisWorkbook.Sheets("DIASNOLABORALES")
    res = Application.Match(CLng(fecha), ws.Columns(1), 0)
    EsDiaLaboral = IsError(res)
End Function

Public Function ObtenerFestivoColombia(fecha As Date) As String
    Dim d As Integer, m As Integer, y As Integer
    Dim pascua As Date
    d = Day(fecha): m = Month(fecha): y = Year(fecha)

    ' Festivos fijos
    If d = 1 And m = 1 Then ObtenerFestivoColombia = "Año Nuevo": Exit Function
    If d = 1 And m = 5 Then ObtenerFestivoColombia = "Día del Trabajo": Exit Function
    If d = 20 And m = 7 Then ObtenerFestivoColombia = "Independencia": Exit Function
    If d = 7 And m = 8 Then ObtenerFestivoColombia = "Batalla de Boyacá": Exit Function
    If d = 8 And m = 12 Then ObtenerFestivoColombia = "Inmaculada Concepción": Exit Function
    If d = 25 And m = 12 Then ObtenerFestivoColombia = "Navidad": Exit Function

    ' Ley Emiliani
    If EsFestivoEmiliani(fecha, DateSerial(y, 1, 6)) Then ObtenerFestivoColombia = "Reyes Magos": Exit Function
    If EsFestivoEmiliani(fecha, DateSerial(y, 3, 19)) Then ObtenerFestivoColombia = "San José": Exit Function
    If EsFestivoEmiliani(fecha, DateSerial(y, 6, 29)) Then ObtenerFestivoColombia = "San Pedro y San Pablo": Exit Function
    If EsFestivoEmiliani(fecha, DateSerial(y, 8, 15)) Then ObtenerFestivoColombia = "Asunción": Exit Function
    If EsFestivoEmiliani(fecha, DateSerial(y, 10, 12)) Then ObtenerFestivoColombia = "Día de la Raza": Exit Function
    If EsFestivoEmiliani(fecha, DateSerial(y, 11, 1)) Then ObtenerFestivoColombia = "Todos los Santos": Exit Function
    If EsFestivoEmiliani(fecha, DateSerial(y, 11, 11)) Then ObtenerFestivoColombia = "Independencia Cartagena": Exit Function

    ' Semana Santa
    pascua = CalcularPascua(y)
    If fecha = pascua - 3 Then ObtenerFestivoColombia = "Jueves Santo": Exit Function
    If fecha = pascua - 2 Then ObtenerFestivoColombia = "Viernes Santo": Exit Function
    If EsFestivoEmiliani(fecha, pascua + 43) Then ObtenerFestivoColombia = "Ascensión": Exit Function
    If EsFestivoEmiliani(fecha, pascua + 64) Then ObtenerFestivoColombia = "Corpus Christi": Exit Function
    If EsFestivoEmiliani(fecha, pascua + 71) Then ObtenerFestivoColombia = "Sagrado Corazón": Exit Function

    ObtenerFestivoColombia = ""
End Function

Public Function EsFestivoEmiliani(fechaEvaluar As Date, fechaReal As Date) As Boolean
    Dim fechaTraslado As Date
    If Weekday(fechaReal, vbMonday) = 1 Then
        fechaTraslado = fechaReal
    Else
        If Weekday(fechaReal, vbMonday) = 7 Then
            fechaTraslado = fechaReal + 1
        Else
            fechaTraslado = fechaReal + (8 - Weekday(fechaReal, vbMonday))
        End If
    End If
    EsFestivoEmiliani = (fechaEvaluar = fechaTraslado)
End Function

Public Function CalcularPascua(ByVal Anio As Integer) As Date
    Dim a As Integer, b As Integer, c As Integer, d As Integer, e As Integer
    a = Anio Mod 19
    b = Anio Mod 4
    c = Anio Mod 7
    d = (19 * a + 24) Mod 30
    e = (2 * b + 4 * c - 6 * d + 5) Mod 7
    CalcularPascua = DateSerial(Anio, 3, 22) + d + e
End Function
