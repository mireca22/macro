Public Function IsNumericValue(valor As Variant) As Boolean
    On Error Resume Next
    IsNumericValue = IsNumeric(valor)
    On Error GoTo 0
End Function

Public Function ConvertirANumero(valor As Variant) As Long
    On Error Resume Next
    ConvertirANumero = CLng(valor)
    On Error GoTo 0
End Function

Public Function FormatearSegundosAMinutos(segundos As Long) As String
    Dim minutos As Long, segs As Long
    minutos = segundos \ 60
    segs    = segundos Mod 60
    FormatearSegundosAMinutos = Format(minutos, "00") & ":" & Format(segs, "00")
End Function

Public Function ObtenerDiaDelaSemana(fecha As Date) As String
    Dim dias As Variant
    dias = Split("Domingo,Lunes,Martes,Miercoles,Jueves,Viernes,Sabado", ",")
    ObtenerDiaDelaSemana = dias(Weekday(fecha) - 1)
End Function

Public Function CalcularPorcentajeCompletado(completadas As Long, total As Long) As String
    If total = 0 Then
        CalcularPorcentajeCompletado = "0%"
    Else
        CalcularPorcentajeCompletado = Format(completadas / total, "0%")
    End If
End Function

Public Sub LimpiarControles(frm As Object)
    Dim ctrl As Control
    For Each ctrl In frm.Controls
        Select Case TypeName(ctrl)
            Case "TextBox":    ctrl.Value = ""
            Case "ComboBox":   ctrl.ListIndex = -1: ctrl.Value = ""
            Case "CheckBox":   ctrl.Value = False
            Case "SpinButton": ctrl.Value = 0
        End Select
    Next ctrl
End Sub

Public Function ObtenerProximoColorFila(fila As Long) As Long
    If fila Mod 2 = 0 Then
        ObtenerProximoColorFila = RGB(240, 240, 240)
    Else
        ObtenerProximoColorFila = RGB(255, 255, 255)
    End If
End Function

' ==============================================================================
' FORMATEO Y PRESENTACION VISUAL
' ==============================================================================
Public Sub AjustarAnchoColumnasInteligente(ws As Worksheet)
    Dim col      As Range
    Dim maxAncho As Double
    maxAncho = 50
    On Error Resume Next
    ws.UsedRange.Columns.AutoFit
    For Each col In ws.UsedRange.Columns
        If col.ColumnWidth > maxAncho Then
            col.ColumnWidth = maxAncho
            col.WrapText    = True
        Else
            col.WrapText = False
        End If
    Next col
    Dim lastRow As Long
    lastRow = ws.UsedRange.Rows.Count + ws.UsedRange.Row - 1
    If lastRow > 3 Then ws.Rows("3:" & lastRow).EntireRow.AutoFit
    On Error GoTo 0
End Sub

Public Sub AplicarFormatoAsignaciones()

    Dim fila    As Long, ultFila As Long
    Set ws = ThisWorkbook.Sheets("ASIGNACIONES_DIARIAS")
    Application.ScreenUpdating = False

    ws.Columns("E:E").HorizontalAlignment = xlCenter
    ws.Columns("G:H").HorizontalAlignment = xlCenter
    ws.Columns("J:K").HorizontalAlignment = xlCenter
    ws.Columns("N:O").HorizontalAlignment = xlCenter
    ws.Columns("L:M").NumberFormat = "#,0"
    Application.ScreenUpdating = True
    MsgBox "Formato aplicado.", vbInformation
End Sub

Public Function ObtenerEstadisticasHoy() As Variant
    Dim ws          As Worksheet
    Dim fila        As Long
    Dim stats(1 To 4) As Variant
    Dim totalMin    As Double
    Dim completadas As Long, pendientes As Long, contador As Long
    Set ws = ThisWorkbook.Sheets("ASIGNACIONES_DIARIAS")
    For fila = 4 To ws.Cells(ws.Rows.Count, 1).End(xlUp).Row
        If Int(ws.Cells(fila, 1).Value) = Int(Date) Then
            contador  = contador + 1
            totalMin  = totalMin + ws.Cells(fila, 14).Value
            If ws.Cells(fila, 15).Value = "Completada" Then
                completadas = completadas + 1
            Else
                pendientes = pendientes + 1
            End If
        End If
    Next fila
    stats(1) = contador
    stats(2) = CLng(totalMin)
    stats(3) = completadas
    stats(4) = pendientes
    ObtenerEstadisticasHoy = stats
End Function

' ==============================================================================
' GENERADOR PRINCIPAL DE REPORTES
' CORRECCION v3.2.2: activarHoja opcional (default False)
'   False = desde formulario -> NO cambia hoja activa (se queda en INICIO)
'   True  = desde dashboard  -> SI activa RESUMEN_DIARIO
' ==============================================================================
Public Sub GenerarReporteEnHoja(fecha As Date, tipoReporte As String, _
    Optional activarHoja As Boolean = False)

    Dim wsResumen   As Worksheet, wsAsig As Worksheet, wsOp As Worksheet
    Dim dictOp      As Object
    Dim i           As Long, ultFila As Long
    Dim idOp        As String, nomOp As String
    Dim cap         As Double
    Dim fechaFila   As Date
    Dim esMes       As Boolean
    Dim diasLaborales As Integer

    Set wsResumen = ThisWorkbook.Sheets("RESUMEN_DIARIO")
    Set wsAsig    = ThisWorkbook.Sheets("ASIGNACIONES_DIARIAS")
    Set wsOp      = ThisWorkbook.Sheets("OPERARIAS")
    Set dictOp    = CreateObject("Scripting.Dictionary")
    esMes         = (tipoReporte = "MENSUAL")
    diasLaborales = 22

    Application.ScreenUpdating = False
    Application.EnableEvents   = False

    ' 1. Cargar Operarias
    ultFila = wsOp.Cells(wsOp.Rows.Count, 1).End(xlUp).Row
    For i = 4 To ultFila
        idOp  = wsOp.Cells(i, 1).Value
        nomOp = wsOp.Cells(i, 2).Value
        cap   = Val(wsOp.Cells(i, 3).Value)
        If esMes Then cap = cap * diasLaborales
        If idOp <> "" Then dictOp(idOp) = Array(nomOp, cap, 0, 0, 0)
    Next i

    ' 2. Leer ASIGNACIONES_DIARIAS en array
    Dim dataArr As Variant, incluir As Boolean
    ultFila = wsAsig.Cells(wsAsig.Rows.Count, 1).End(xlUp).Row
    If ultFila >= 4 Then
        dataArr = wsAsig.Range(wsAsig.Cells(4, 1), wsAsig.Cells(ultFila, 21)).Value
        For i = 1 To UBound(dataArr, 1)
            If IsDate(dataArr(i, 1)) Then
                fechaFila = dataArr(i, 1)
                incluir   = False
                If esMes Then
                    If Month(fechaFila) = Month(fecha) And Year(fechaFila) = Year(fecha) Then incluir = True
                Else
                    If Int(fechaFila) = Int(fecha) Then incluir = True
                End If
                If incluir Then
                    idOp = dataArr(i, 2)
                    If dictOp.Exists(idOp) Then
                        Dim arrOp As Variant
                        arrOp    = dictOp(idOp)
                        arrOp(2) = arrOp(2) + Val(dataArr(i, 14))
                        If dataArr(i, 15) = "Completada" Then arrOp(3) = arrOp(3) + 1
                        arrOp(4)     = arrOp(4) + Val(dataArr(i, 20))
                        dictOp(idOp) = arrOp
                    End If
                End If
            End If
        Next i
    End If

    ' 3. Escribir reporte — headers fila 3, datos fila 4
    wsResumen.Cells.Clear
    With wsResumen.Range("A3:F3")
        .Value = Array("Operaria", "Capacidad min", "Asignado min", "Ocupacion %", "Disponible min", "Estado")
        .Font.Bold = True
        .Interior.Color = RGB(220, 220, 220)
        .Borders.LineStyle = xlContinuous
    End With

    Dim fila As Long, k As Variant
    fila = 4
    For Each k In dictOp.Keys
        Dim d As Variant, ocu As Double, dispon As Double, estado As String
        d      = dictOp(k)
        ocu    = IIf(d(1) > 0, d(2) / d(1), 0)
        dispon = d(1) - d(2)
        If ocu >= 1 Then
            estado = "SOBRECARGA"
        ElseIf dispon > 0 Then
            estado = "DISPONIBLE"
        Else
            estado = "OPTIMA"
        End If
        wsResumen.Cells(fila, 1).Value = d(0)
        wsResumen.Cells(fila, 2).Value = d(1)
        wsResumen.Cells(fila, 3).Value = d(2)
        wsResumen.Cells(fila, 4).Value = Format(ocu, "0%")
        wsResumen.Cells(fila, 5).Value = dispon
        wsResumen.Cells(fila, 6).Value = estado
        Select Case estado
            Case "DISPONIBLE", "OPTIMA": wsResumen.Cells(fila, 6).Interior.Color = RGB(0, 176, 80)
            Case "SOBRECARGA":            wsResumen.Cells(fila, 6).Interior.Color = RGB(255, 0, 0)
        End Select
        fila = fila + 1
    Next k

    wsResumen.Columns.AutoFit
    If activarHoja Then wsResumen.Activate  ' CORRECCION v3.2.2

    Application.EnableEvents   = True
    Application.ScreenUpdating = True
End Sub

' ==============================================================================
' REPORTE EFICIENCIA
' ==============================================================================
Public Sub GenerarReporteEficiencia()
    Dim wsReporte   As Worksheet, wsAsig As Worksheet
    Dim dictPerf    As Object
    Dim i           As Long, ultFila As Long
    Dim fechaInicio As Date, fechaFin As Date, fechaStr As String

    fechaStr = InputBox("Fecha de INICIO del reporte (dd/mm/yyyy):", "Reporte de Eficiencia", Format(Date - 30, "dd/mm/yyyy"))
    If fechaStr = "" Then Exit Sub
    If Not IsDate(fechaStr) Then Exit Sub
    fechaInicio = CDate(fechaStr)
    fechaStr = InputBox("Fecha de FIN del reporte (dd/mm/yyyy):", "Reporte de Eficiencia", Format(Date, "dd/mm/yyyy"))
    If fechaStr = "" Then Exit Sub
    If Not IsDate(fechaStr) Then Exit Sub
    fechaFin = CDate(fechaStr)

    Set wsReporte = ThisWorkbook.Sheets("ANALISIS_TIEMPOS")
    Set wsAsig    = ThisWorkbook.Sheets("ASIGNACIONES_DIARIAS")
    Set dictPerf  = CreateObject("Scripting.Dictionary")

    Application.ScreenUpdating = False
    Application.EnableEvents   = False

    ultFila = wsAsig.Cells(wsAsig.Rows.Count, 1).End(xlUp).Row
    If ultFila >= 4 Then
        Dim dataArr As Variant
        dataArr = wsAsig.Range(wsAsig.Cells(4, 1), wsAsig.Cells(ultFila, 21)).Value
        For i = 1 To UBound(dataArr, 1)
            If IsDate(dataArr(i, 1)) Then
                Dim fechaAsig As Date
                fechaAsig = dataArr(i, 1)
                If fechaAsig >= fechaInicio And fechaAsig <= fechaFin Then
                    Dim key As String
                    key = dataArr(i, 2) & "|" & dataArr(i, 4) & "|" & dataArr(i, 9)
                    Dim minAsig As Double, minProd As Double
                    minAsig = Val(dataArr(i, 14))
                    minProd = CLng(Val(dataArr(i, 16))) * CLng(Val(dataArr(i, 12))) / 60
                    If Not dictPerf.Exists(key) Then
                        dictPerf(key) = Array(dataArr(i, 3), dataArr(i, 5), dataArr(i, 10), 0, 0)
                    End If
                    Dim arrP As Variant
                    arrP    = dictPerf(key)
                    arrP(3) = arrP(3) + minAsig
                    arrP(4) = arrP(4) + minProd
                    dictPerf(key) = arrP
                End If
            End If
        Next i
    End If

    wsReporte.Cells.Clear
    With wsReporte.Range("A3:G3")
        .Value = Array("Operaria", "Referencia", "Tarea", "Min. Asignados", "Min. Producidos Est.", "Desviacion min", "Eficiencia")
        .Font.Bold = True: .Interior.Color = RGB(220, 220, 220): .Borders.LineStyle = xlContinuous
    End With

    Dim fila As Long, k As Variant
    fila = 4
    For Each k In dictPerf.Keys
        Dim d As Variant
        d = dictPerf(k)
        wsReporte.Cells(fila, 1).Value = d(0)
        wsReporte.Cells(fila, 2).Value = d(1)
        wsReporte.Cells(fila, 3).Value = d(2)
        wsReporte.Cells(fila, 4).Value = d(3)
        wsReporte.Cells(fila, 5).Value = d(4)
        wsReporte.Cells(fila, 6).Value = d(4) - d(3)
        If d(3) > 0 Then wsReporte.Cells(fila, 7).Value = d(4) / d(3)
        fila = fila + 1
    Next k

    wsReporte.Range("G4:G" & fila).NumberFormat = "0.0%"
    wsReporte.Columns.AutoFit
    wsReporte.Activate
    Application.EnableEvents   = True
    Application.ScreenUpdating = True
    MsgBox "Reporte de Eficiencia generado con " & dictPerf.Count & " registros.", vbInformation
End Sub

' ==============================================================================
' REPORTE AVANCE PRODUCCION
' ==============================================================================
Public Sub GenerarReporteAvanceProduccion()
    Dim wsAvance    As Worksheet, wsLotes As Worksheet
    Dim wsTareas    As Worksheet, wsAsig  As Worksheet
    Dim dictUnidades As Object, dictAsignado As Object
    Dim i As Long, ultFila As Long
    Dim idRef As String, key As String, cant As Long

    Set wsAvance    = ThisWorkbook.Sheets("AVANCE_PRODUCCION")
    Set wsLotes     = ThisWorkbook.Sheets("LOTES")
    Set wsTareas    = ThisWorkbook.Sheets("TAREAS")
    Set wsAsig      = ThisWorkbook.Sheets("ASIGNACIONES_DIARIAS")
    Set dictUnidades = CreateObject("Scripting.Dictionary")
    Set dictAsignado = CreateObject("Scripting.Dictionary")

    Application.ScreenUpdating = False

    ' 1. Total por Referencia desde LOTES (fila 4)
    ultFila = wsLotes.Cells(wsLotes.Rows.Count, 1).End(xlUp).Row
    For i = 4 To ultFila
        idRef = wsLotes.Cells(i, 2).Value
        cant  = wsLotes.Cells(i, 5).Value
        If dictUnidades.Exists(idRef) Then
            dictUnidades(idRef) = dictUnidades(idRef) + cant
        Else
            dictUnidades(idRef) = cant
        End If
    Next i

    ' 2. Lo asignado por Ref+Tarea (fila 4)
    ultFila = wsAsig.Cells(wsAsig.Rows.Count, 1).End(xlUp).Row
    For i = 4 To ultFila
        idRef = wsAsig.Cells(i, 4).Value
        key   = idRef & "|" & wsAsig.Cells(i, 9).Value
        cant  = wsAsig.Cells(i, 11).Value
        If dictAsignado.Exists(key) Then
            dictAsignado(key) = dictAsignado(key) + cant
        Else
            dictAsignado(key) = cant
        End If
    Next i

    ' 3. Reporte — headers fila 3, datos fila 4
    wsAvance.Cells.Clear
    With wsAvance.Range("A3:I3")
        .Value = Array("ID Referencia", "Nombre Referencia", "Nombre Tarea", "Tiempo Unit seg", _
            "Total Unidades Req.", "Unidades Asignadas", "Unidades Pendientes", "Tiempo Pendiente min", "Estado Asignacion")
        .Font.Bold = True: .Interior.Color = RGB(220, 220, 220): .Borders.LineStyle = xlContinuous
    End With

    Dim fila As Long
    fila    = 4
    ultFila = wsTareas.Cells(wsTareas.Rows.Count, 1).End(xlUp).Row
    For i = 4 To ultFila
        idRef = wsTareas.Cells(i, 1).Value
        Dim numT As String, nomT As String, tiempoT As Long
        numT    = wsTareas.Cells(i, 2).Value
        nomT    = wsTareas.Cells(i, 3).Value
        tiempoT = wsTareas.Cells(i, 4).Value
        Dim totalReq As Long, totalAsig As Long, pendiente As Long
        totalReq  = IIf(dictUnidades.Exists(idRef), dictUnidades(idRef), 0)
        key       = idRef & "|" & numT
        totalAsig = IIf(dictAsignado.Exists(key), dictAsignado(key), 0)
        pendiente = IIf(totalReq - totalAsig > 0, totalReq - totalAsig, 0)
        Dim tiempoPend As Double, estadoT As String
        tiempoPend = pendiente * tiempoT / 60
        If totalReq = 0 Then
            estadoT = "Sin Tallas"
        ElseIf totalAsig >= totalReq Then
            estadoT = "Completo"
        ElseIf totalAsig = 0 Then
            estadoT = "En Curso"
        Else
            estadoT = "Pendiente"
        End If
        wsAvance.Range("A" & fila & ":I" & fila).Value = Array(idRef, _
            ObtenerNombreReferencia(idRef), nomT, tiempoT, totalReq, totalAsig, pendiente, tiempoPend, estadoT)
        fila = fila + 1
    Next i

    wsAvance.Columns.AutoFit
    wsAvance.Activate
    Application.ScreenUpdating = True
    MsgBox "Reporte de Avance de Produccion generado.", vbInformation
End Sub

' ==============================================================================
' REPORTE LOTES
' ==============================================================================
Public Sub GenerarReporteLotes() ' MEJORADO: Mide el avance basado en tiempo invertido vs tiempo estimado.
    Dim wsReporte   As Worksheet, wsLotes As Worksheet, wsAsig As Worksheet
    Dim dictRefTime As Object, dictTimeDone As Object
    Dim i As Long, ultFila As Long
    Dim idRef As String, idLote As String
    Dim cantLote As Long, timeUnitSeg As Long
    Dim tiempoEstTotalHrs As Double, tiempoRealInvertidoHrs As Double
    Dim avance As Double, estadoL As String
    Dim dataArr As Variant

    Set wsReporte   = ThisWorkbook.Sheets("REPORTE_LOTES")
    Set wsLotes     = ThisWorkbook.Sheets("LOTES")
    Set wsAsig      = ThisWorkbook.Sheets("ASIGNACIONES_DIARIAS")
    Set dictRefTime  = CreateObject("Scripting.Dictionary")
    Set dictTimeDone  = CreateObject("Scripting.Dictionary")

    Application.ScreenUpdating = False

    ' 1. Cargar tiempo total por unidad para cada Referencia
    ' Esta información está en la columna E de la hoja REFERENCIAS
    ultFila = ThisWorkbook.Sheets("REFERENCIAS").Cells(ThisWorkbook.Sheets("REFERENCIAS").Rows.Count, 1).End(xlUp).Row
    If ultFila >= 4 Then
        dataArr = ThisWorkbook.Sheets("REFERENCIAS").Range("A4:E" & ultFila).Value
        For i = 1 To UBound(dataArr, 1)
            idRef = CStr(dataArr(i, 1))
            If idRef <> "" Then dictRefTime(idRef) = Val(dataArr(i, 5))
        Next i
    End If

    ' 2. Sumar el tiempo real invertido por Lote desde ASIGNACIONES_DIARIAS
    ultFila = wsAsig.Cells(wsAsig.Rows.Count, 1).End(xlUp).Row
    If ultFila >= 4 Then
        ' Columnas: F(Lote), T(Tiempo Real Min)
        dataArr = wsAsig.Range("A4:T" & ultFila).Value
        For i = 1 To UBound(dataArr, 1)
            idLote = Trim(CStr(dataArr(i, 6)))
            If idLote <> "" Then
                Dim tiempoRealMin As Double
                tiempoRealMin = Val(dataArr(i, 20))
                If dictTimeDone.Exists(idLote) Then
                    dictTimeDone(idLote) = dictTimeDone(idLote) + tiempoRealMin
                Else
                    dictTimeDone(idLote) = tiempoRealMin
                End If
            End If
        Next i
    End If

    ' 3. Generar Reporte
    wsReporte.Cells.Clear
    With wsReporte.Range("A3:I3")
        .Value = Array("Lote", "Referencia", "Color", "Talla", "Cantidad", "Tiempo Estimado (hrs)", "Tiempo Invertido (hrs)", "Avance por Tiempo", "Estado")
        .Font.Bold = True: .Interior.Color = RGB(220, 220, 220): .Borders.LineStyle = xlContinuous
    End With

    Dim fila As Long
    fila    = 4
    ultFila = wsLotes.Cells(wsLotes.Rows.Count, 1).End(xlUp).Row
    If ultFila >= 4 Then
        dataArr = wsLotes.Range("A4:K" & ultFila).Value
        For i = 1 To UBound(dataArr, 1)
            idLote   = Trim(CStr(dataArr(i, 1)))
            idRef    = Trim(CStr(dataArr(i, 2)))
            cantLote = Val(dataArr(i, 5))
            estadoL  = CStr(dataArr(i, 10))

            If idLote <> "" Then
                timeUnitSeg = IIf(dictRefTime.Exists(idRef), dictRefTime(idRef), 0)
                tiempoEstTotalHrs = (cantLote * timeUnitSeg) / 3600

                tiempoRealInvertidoHrs = IIf(dictTimeDone.Exists(idLote), dictTimeDone(idLote) / 60, 0)

                If tiempoEstTotalHrs > 0 Then
                    avance = tiempoRealInvertidoHrs / tiempoEstTotalHrs
                Else
                    avance = 0
                End If

                If estadoL = "Completada" Then
                    wsReporte.Cells(fila, 9).Font.Color = RGB(0, 150, 0)
                ElseIf estadoL = "En Proceso" Then
                    wsReporte.Cells(fila, 9).Font.Color = RGB(255, 140, 0)
                End If

                wsReporte.Cells(fila, 1).Value = idLote
                wsReporte.Cells(fila, 2).Value = idRef
                wsReporte.Cells(fila, 3).Value = CStr(dataArr(i, 3)) ' Color
                wsReporte.Cells(fila, 4).Value = CStr(dataArr(i, 4)) ' Talla
                wsReporte.Cells(fila, 5).Value = cantLote
                wsReporte.Cells(fila, 6).Value = Round(tiempoEstTotalHrs, 2)
                wsReporte.Cells(fila, 7).Value = Round(tiempoRealInvertidoHrs, 2)
                wsReporte.Cells(fila, 8).Value = avance
                wsReporte.Cells(fila, 9).Value = estadoL

                fila = fila + 1
            End If
        Next i
    End If

    wsReporte.Range("H4:H" & fila).NumberFormat = "0.0%"
    wsReporte.Columns.AutoFit
    wsReporte.Activate
    Application.ScreenUpdating = True
    MsgBox "Reporte de Lotes (Mejorado) generado exitosamente.", vbInformation
End Sub


' ==============================================================================
' INFORME PRODUCCION FINALIZADA
' ==============================================================================
Public Sub GenerarInformeProduccion()
    Dim wsProd      As Worksheet, wsLotes As Worksheet
    Dim dictAvance  As Object
    Dim i As Long, ultFila As Long
    Dim idRef As String

    Set wsProd     = ThisWorkbook.Sheets("PRODUCCION_FINALIZADA")
    Set wsLotes    = ThisWorkbook.Sheets("LOTES")
    Set dictAvance = CreateObject("Scripting.Dictionary")

    Application.ScreenUpdating = False

    ultFila = wsLotes.Cells(wsLotes.Rows.Count, 1).End(xlUp).Row
    For i = 4 To ultFila
        idRef = wsLotes.Cells(i, 2).Value
        If idLote <> "" Then
            If Not dictAvance.Exists(idRef) Then
                dictAvance(idRef) = Array(ObtenerNombreReferencia(idRef), 0, 0, 0, 0)
            End If
            Dim arrRef As Variant
            arrRef    = dictAvance(idRef)
            arrRef(1) = arrRef(1) + Val(wsLotes.Cells(i, 5).Value)
            arrRef(2) = arrRef(2) + Val(wsLotes.Cells(i, 8).Value)
            arrRef(3) = arrRef(3) + 1
            dictAvance(idRef) = arrRef
        End If
    Next i
    
    wsProd.Cells.Clear
    With wsProd.Range("A3:F3")
        .Value = Array("Fecha Cierre", "ID Referencia", "Nombre Referencia", "Total Unidades", "Unidades Completadas", "Porcentaje")
        .Font.Bold = True: .Interior.Color = RGB(220, 220, 220): .Borders.LineStyle = xlContinuous
    End With

    Dim fila As Long, k As Variant
    fila = 4
    For Each k In dictAvance.Keys
        Dim d As Variant, pct As Double
        d   = dictAvance(k)
        pct = IIf(d(1) > 0, d(2) / d(1), 0)
        wsProd.Range("A" & fila & ":F" & fila).Value = Array(Date, k, d(0), d(1), d(2), Format(pct, "0%"))
        fila = fila + 1
    Next k

    wsProd.Columns.AutoFit
    wsProd.Activate
    Application.ScreenUpdating = True
    MsgBox "Informe de Produccion generado.", vbInformation
End Sub

' ==============================================================================
' REPORTE TRAZABILIDAD DE LOTE (NUEVO)
' Muestra el avance de un lote específico, tarea por tarea.
' ==============================================================================
Public Sub GenerarReporteTrazabilidadLote()
    Dim wsReporte   As Worksheet
    Dim wsTareas    As Worksheet
    Dim wsAsig      As Worksheet
    Dim wsLotes     As Worksheet
    Dim dictTareas  As Object
    Dim i           As Long, ultFila As Long
    Dim idLoteInput As String
    Dim idRef       As String, cantLote As Long, colorLote As String, tallaLote As String
    Dim dataArr     As Variant
    Dim key         As String

    ' 1. Solicitar ID del Lote al usuario
    idLoteInput = InputBox("Ingrese el ID del Lote para generar su trazabilidad:", "Reporte de Trazabilidad de Lote")
    If idLoteInput = "" Then Exit Sub
    idLoteInput = Trim(idLoteInput)

    Application.ScreenUpdating = False

    ' 2. Encontrar el Lote y obtener sus datos base
    Set wsLotes = ThisWorkbook.Sheets("LOTES")
    ultFila = wsLotes.Cells(wsLotes.Rows.Count, 1).End(xlUp).Row
    Dim loteEncontrado As Boolean
    loteEncontrado = False
    If ultFila >= 4 Then
        dataArr = wsLotes.Range("A4:E" & ultFila).Value
        For i = 1 To UBound(dataArr, 1)
            If Trim(CStr(dataArr(i, 1))) = idLoteInput Then
                idRef = CStr(dataArr(i, 2))
                colorLote = CStr(dataArr(i, 3))
                tallaLote = CStr(dataArr(i, 4))
                cantLote = CLng(dataArr(i, 5))
                loteEncontrado = True
                Exit For
            End If
        Next i
    End If

    If Not loteEncontrado Then
        MsgBox "El Lote '" & idLoteInput & "' no fue encontrado en la hoja LOTES.", vbCritical
        Application.ScreenUpdating = True
        Exit Sub
    End If

    ' 3. Cargar todas las tareas para la referencia del lote
    Set dictTareas = CreateObject("Scripting.Dictionary")
    Set wsTareas = ThisWorkbook.Sheets("TAREAS")
    ultFila = wsTareas.Cells(wsTareas.Rows.Count, 1).End(xlUp).Row
    If ultFila >= 4 Then
        dataArr = wsTareas.Range("A4:D" & ultFila).Value
        For i = 1 To UBound(dataArr, 1)
            If Trim(CStr(dataArr(i, 1))) = idRef Then
                key = CStr(dataArr(i, 2)) ' Numero de Tarea
                ' Array: NombreTarea, UnidadesAsignadas, UnidadesCompletadas
                dictTareas(key) = Array(CStr(dataArr(i, 3)), 0, 0)
            End If
        Next i
    End If

    ' 4. Recorrer las asignaciones para sumar unidades
    Set wsAsig = ThisWorkbook.Sheets("ASIGNACIONES_DIARIAS")
    ultFila = wsAsig.Cells(wsAsig.Rows.Count, 1).End(xlUp).Row
    If ultFila >= 4 Then
        ' Leer columnas F(Lote), I(Tarea), K(Asignadas), P(Completadas)
        dataArr = wsAsig.Range("A4:P" & ultFila).Value
        For i = 1 To UBound(dataArr, 1)
            If Trim(CStr(dataArr(i, 6))) = idLoteInput Then
                key = CStr(dataArr(i, 9)) ' Numero de Tarea
                If dictTareas.Exists(key) Then
                    Dim tareaData As Variant
                    tareaData = dictTareas(key)
                    tareaData(1) = tareaData(1) + CLng(Val(dataArr(i, 11))) ' Sumar Asignadas
                    tareaData(2) = tareaData(2) + CLng(Val(dataArr(i, 16))) ' Sumar Completadas
                    dictTareas(key) = tareaData
                End If
            End If
        Next i
    End If

    ' 5. Generar el reporte en una nueva hoja o una predefinida
    Dim hojaReporteNombre As String
    hojaReporteNombre = "TRAZABILIDAD_LOTE"
    On Error Resume Next
    Set wsReporte = ThisWorkbook.Sheets(hojaReporteNombre)
    If wsReporte Is Nothing Then
        Set wsReporte = ThisWorkbook.Sheets.Add(After:=ThisWorkbook.Sheets(ThisWorkbook.Sheets.Count))
        wsReporte.Name = hojaReporteNombre
    End If
    On Error GoTo 0
    wsReporte.Cells.Clear
    wsReporte.Tab.Color = RGB(255, 140, 0) ' Naranja

    ' Escribir encabezados del reporte
    wsReporte.Range("A1").Value = "Reporte de Trazabilidad del Lote: " & idLoteInput
    wsReporte.Range("A1").Font.Bold = True
    wsReporte.Range("A1").Font.Size = 14
    wsReporte.Range("A2").Value = "Referencia: " & idRef & " (" & ObtenerNombreReferencia(idRef) & ")"
    wsReporte.Range("C2").Value = "Color: " & colorLote
    wsReporte.Range("D2").Value = "Talla: " & tallaLote
    wsReporte.Range("E2").Value = "Cantidad Total del Lote: " & cantLote

    With wsReporte.Range("A4:G4")
        .Value = Array("N° Tarea", "Nombre Tarea", "Total Requerido", "Unidades Asignadas", "Unidades Completadas", "Avance Tarea", "Estado")
        .Font.Bold = True
        .Interior.Color = RGB(220, 220, 220)
        .Borders.LineStyle = xlContinuous
    End With

    ' 6. Escribir los datos en la hoja
    Dim fila As Long, k As Variant
    fila = 5
    For Each k In dictTareas.Keys
        Dim d As Variant
        d = dictTareas(k)
        Dim avance As Double, estado As String
        Dim completadas As Long
        completadas = d(2)

        If cantLote > 0 Then
            avance = completadas / cantLote
        Else
            avance = 0
        End If

        If avance >= 1 Then
            estado = "Completada"
        ElseIf avance > 0 Then
            estado = "En Proceso"
        Else
            estado = "Pendiente"
        End If

        wsReporte.Cells(fila, 1).Value = k ' N° Tarea
        wsReporte.Cells(fila, 2).Value = d(0) ' Nombre Tarea
        wsReporte.Cells(fila, 3).Value = cantLote ' Total Requerido
        wsReporte.Cells(fila, 4).Value = d(1) ' Unidades Asignadas
        wsReporte.Cells(fila, 5).Value = completadas ' Unidades Completadas
        wsReporte.Cells(fila, 6).Value = avance ' Avance Tarea
        wsReporte.Cells(fila, 7).Value = estado ' Estado

        If estado = "Completada" Then
            wsReporte.Range("G" & fila).Interior.Color = RGB(198, 239, 206) ' Verde
            wsReporte.Range("G" & fila).Font.Color = RGB(0, 97, 0)
        ElseIf estado = "En Proceso" Then
            wsReporte.Range("G" & fila).Interior.Color = RGB(255, 235, 156) ' Amarillo
            wsReporte.Range("G" & fila).Font.Color = RGB(156, 87, 0)
        End If

        fila = fila + 1
    Next k

    wsReporte.Range("F5:F" & fila).NumberFormat = "0.0%"
    wsReporte.Columns.AutoFit
    wsReporte.Activate
    Application.ScreenUpdating = True
    MsgBox "Reporte de Trazabilidad para el lote '" & idLoteInput & "' generado exitosamente.", vbInformation
End Sub

' ==============================================================================
' CALENDARIO DIAS NO LABORALES - COLOMBIA
' RESTAURADO v3.2.3
' ==============================================================================
Public Sub ActualizarCalendarioNoLaboral()
    Dim ws          As Worksheet
    Dim ultFila     As Long
    Dim fechaInicio As Date, fechaFin As Date, fechaIter As Date
    Dim descripcion As String
    Dim contador    As Long

    Set ws  = ThisWorkbook.Sheets("DIAS_NO_LABORALES")
    ultFila = ws.Cells(ws.Rows.Count, 1).End(xlUp).Row

    If ultFila > 3 Then
        fechaInicio = ws.Cells(ultFila, 1).Value + 1
    Else
        fechaInicio = DateSerial(Year(Date), 1, 1)
    End If
    fechaFin = DateSerial(Year(Date), 12, 31)

    If fechaInicio > fechaFin Then
        MsgBox "INFO: El calendario ya esta actualizado hasta " & Format(fechaFin, "dd/mm/yyyy"), vbInformation
        Exit Sub
    End If

    Application.ScreenUpdating = False
    contador = 0

    For fechaIter = fechaInicio To fechaFin
        descripcion = ""

        ' Fines de semana
        If Weekday(fechaIter, vbMonday) = 6 Then descripcion = "Sabado"
        If Weekday(fechaIter, vbMonday) = 7 Then descripcion = "Domingo"

        ' Festivos Colombia
        Dim festivo As String
        festivo = ObtenerFestivoColombia(fechaIter)
        If festivo <> "" Then
            If descripcion <> "" Then
                descripcion = descripcion & " - " & festivo
            Else
                descripcion = festivo
            End If
        End If

        If descripcion <> "" Then
            ultFila = ultFila + 1
            ws.Cells(ultFila, 1).Value = fechaIter
            ws.Cells(ultFila, 2).Value = Format(fechaIter, "dddd")
            ws.Cells(ultFila, 3).Value = descripcion
            contador = contador + 1
        End If
    Next fechaIter

    ws.Columns("A:C").AutoFit
    Application.ScreenUpdating = True
    MsgBox "Calendario actualizado. Se agregaron " & contador & " dias no laborales.", vbInformation
End Sub

Public Function EsDiaLaboral(fecha As Date) As Boolean
    Dim ws  As Worksheet
    Dim res As Variant
    Set ws = ThisWorkbook.Sheets("DIAS_NO_LABORALES")
    res = Application.Match(CLng(fecha), ws.Columns(1), 0)
    EsDiaLaboral = IsError(res)
End Function

' ==============================================================================
' FESTIVOS COLOMBIA - Ley 51/1983 (Emiliani) + Semana Santa
' ==============================================================================
Public Function ObtenerFestivoColombia(fecha As Date) As String
    Dim d As Integer, m As Integer, y As Integer
    Dim pascua As Date
    d = Day(fecha): m = Month(fecha): y = Year(fecha)

    ' Festivos fijos
    If d = 1  And m = 1  Then ObtenerFestivoColombia = "Ano Nuevo":                    Exit Function
    If d = 1  And m = 5  Then ObtenerFestivoColombia = "Dia del Trabajo":               Exit Function
    If d = 20 And m = 7  Then ObtenerFestivoColombia = "Independencia de Colombia":     Exit Function
    If d = 7  And m = 8  Then ObtenerFestivoColombia = "Batalla de Boyaca":             Exit Function
    If d = 8  And m = 12 Then ObtenerFestivoColombia = "Inmaculada Concepcion":         Exit Function
    If d = 25 And m = 12 Then ObtenerFestivoColombia = "Navidad":                       Exit Function

    ' Festivos Ley Emiliani (traslado al lunes)
    If EsFestivoEmiliani(fecha, DateSerial(y, 1, 6))   Then ObtenerFestivoColombia = "Reyes Magos":                Exit Function
    If EsFestivoEmiliani(fecha, DateSerial(y, 3, 19))  Then ObtenerFestivoColombia = "San Jose":                   Exit Function
    If EsFestivoEmiliani(fecha, DateSerial(y, 6, 29))  Then ObtenerFestivoColombia = "San Pedro y San Pablo":      Exit Function
    If EsFestivoEmiliani(fecha, DateSerial(y, 8, 15))  Then ObtenerFestivoColombia = "Asuncion de la Virgen":      Exit Function
    If EsFestivoEmiliani(fecha, DateSerial(y, 10, 12)) Then ObtenerFestivoColombia = "Dia de la Raza":             Exit Function
    If EsFestivoEmiliani(fecha, DateSerial(y, 11, 1))  Then ObtenerFestivoColombia = "Todos los Santos":           Exit Function
    If EsFestivoEmiliani(fecha, DateSerial(y, 11, 11)) Then ObtenerFestivoColombia = "Independencia de Cartagena": Exit Function

    ' Semana Santa y moviles (base: Pascua)
    pascua = CalcularPascua(y)
    If fecha = pascua - 3 Then ObtenerFestivoColombia = "Jueves Santo":                 Exit Function
    If fecha = pascua - 2 Then ObtenerFestivoColombia = "Viernes Santo":                Exit Function
    If EsFestivoEmiliani(fecha, pascua + 43) Then ObtenerFestivoColombia = "Ascension del Senor":     Exit Function
    If EsFestivoEmiliani(fecha, pascua + 64) Then ObtenerFestivoColombia = "Corpus Christi":           Exit Function
    If EsFestivoEmiliani(fecha, pascua + 71) Then ObtenerFestivoColombia = "Sagrado Corazon de Jesus": Exit Function

    ObtenerFestivoColombia = ""
End Function

Public Function EsFestivoEmiliani(fechaEvaluar As Date, fechaReal As Date) As Boolean
    Dim fechaTraslado As Date
    Dim diaSemana As Integer
    diaSemana = Weekday(fechaReal, vbMonday)
    If diaSemana = 1 Then
        fechaTraslado = fechaReal
    Else
        fechaTraslado = fechaReal + (8 - diaSemana)
    End If
    EsFestivoEmiliani = (fechaEvaluar = fechaTraslado)
End Function

' Algoritmo de Butcher para Domingo de Pascua
Public Function CalcularPascua(ByVal Anio As Integer) As Date
    Dim a As Integer, b As Integer, c As Integer, d As Integer, e As Integer
    a = Anio Mod 19
    b = Anio Mod 4
    c = Anio Mod 7
    d = (19 * a + 24) Mod 30
    e = (2 * b + 4 * c + 6 * d + 5) Mod 7
    CalcularPascua = DateSerial(Anio, 3, 22) + d + e
End Function

        End If
    Next i
    GetCapacidadDiariaTotal = capTotal
End Function

Public Function CalcularFechaFinEstimada(idRef As String, cantidadTotal As Long, fechaInicio As Date) As Date
    Dim tiempoUnitTotalSeg  As Long
    Dim minutosRequeridos   As Double
    Dim capacidadDiariaMin  As Long
    Dim diasNecesarios      As Double
    Dim fechaFin            As Date
    Dim diasContados        As Integer

    tiempoUnitTotalSeg = ObtenerTiempoTotalUnidad(idRef)
    If tiempoUnitTotalSeg = 0 Then
        MsgBox "La referencia " & idRef & " no tiene tareas o tiempos definidos.", vbExclamation
        CalcularFechaFinEstimada = 0
        Exit Function
    End If

    minutosRequeridos  = cantidadTotal * tiempoUnitTotalSeg / 60
    capacidadDiariaMin = GetCapacidadDiariaTotal()
    If capacidadDiariaMin = 0 Then capacidadDiariaMin = 528

    diasNecesarios = Application.WorksheetFunction.RoundUp(minutosRequeridos / capacidadDiariaMin, 0)
    fechaFin       = fechaInicio
    diasContados   = 0

    Do While diasContados < diasNecesarios
        fechaFin = fechaFin + 1
        If EsDiaLaboral(fechaFin) Then diasContados = diasContados + 1
        If diasContados > 1000 Then Exit Do
    Loop

    CalcularFechaFinEstimada = fechaFin
End Function

' ==============================================================================
' NUEVO: REPORTE MATRIZ DE ASIGNACION (CARGA DE TRABAJO)
' Muestra una matriz de Fechas (Filas) vs Operarias (Columnas)
' ==============================================================================
Public Sub GenerarReporteMatrizAsignacion()
    Dim wsMatriz    As Worksheet, wsAsig As Worksheet, wsOp As Worksheet
    Dim fechaStr    As String
    Dim fechaInicio As Date, fechaFin As Date
    Dim dictAsig    As Object ' Key: "Fecha|IdOp" -> Value: MinutosAsignados
    Dim dictOp      As Object ' Key: IdOp -> Array(Nombre, CapacidadMin)
    Dim colOps      As Collection
    Dim i           As Long, ultFila As Long
    Dim f           As Date, opId As Variant, minutos As Double
    Dim data        As Variant

    ' 1. Solicitar rango de fechas
    fechaStr = InputBox("Ingrese fecha INICIO (dd/mm/yyyy):", "Reporte Matriz", Format(Date, "dd/mm/yyyy"))
    If fechaStr = "" Or Not IsDate(fechaStr) Then Exit Sub
    fechaInicio = CDate(fechaStr)

    fechaStr = InputBox("Ingrese fecha FIN (dd/mm/yyyy):", "Reporte Matriz", Format(Date + 7, "dd/mm/yyyy"))
    If fechaStr = "" Or Not IsDate(fechaStr) Then Exit Sub
    fechaFin = CDate(fechaStr)

    If fechaFin < fechaInicio Then
        MsgBox "La fecha fin no puede ser menor a la de inicio.", vbExclamation
        Exit Sub
    End If

    Application.ScreenUpdating = False
    Application.Calculation = xlCalculationManual

    ' 2. Preparar Hoja
    On Error Resume Next
    Set wsMatriz = ThisWorkbook.Sheets("MATRIZ_ASIGNACION")
    If wsMatriz Is Nothing Then
        Set wsMatriz = ThisWorkbook.Sheets.Add(After:=ThisWorkbook.Sheets(ThisWorkbook.Sheets.Count))
        wsMatriz.Name = "MATRIZ_ASIGNACION"
    End If
    On Error GoTo 0
    wsMatriz.Cells.Clear
    wsMatriz.Tab.Color = RGB(0, 112, 192)

    ' 3. Cargar Operarias Activas
    Set dictOp = CreateObject("Scripting.Dictionary")
    Set colOps = New Collection
    Set wsOp = ThisWorkbook.Sheets("OPERARIAS")
    ultFila = wsOp.Cells(wsOp.Rows.Count, 1).End(xlUp).Row
    For i = 4 To ultFila
        opId = Trim(CStr(wsOp.Cells(i, 1).Value))
        If opId <> "" And wsOp.Cells(i, 4).Value = "Activa" Then
            ' CORRECCION: El valor en la hoja ya son MINUTOS, no multiplicar por 60
            dictOp(opId) = Array(wsOp.Cells(i, 2).Value, Val(wsOp.Cells(i, 3).Value))
            colOps.Add opId
        End If
    Next i

    ' 4. Cargar Asignaciones del rango
    Set dictAsig = CreateObject("Scripting.Dictionary")
    Set wsAsig = ThisWorkbook.Sheets("ASIGNACIONES_DIARIAS")
    ultFila = wsAsig.Cells(wsAsig.Rows.Count, 1).End(xlUp).Row
    If ultFila >= 4 Then
        data = wsAsig.Range("A4:N" & ultFila).Value ' Col 1: Fecha, Col 2: IDOp, Col 14: Minutos
        For i = 1 To UBound(data, 1)
            f = Int(CDate(data(i, 1)))
            If f >= fechaInicio And f <= fechaFin Then
                opId = Trim(CStr(data(i, 2)))
                minutos = Val(data(i, 14))
                Dim key As String
                key = Format(f, "yyyy-mm-dd") & "|" & opId
                dictAsig(key) = dictAsig(key) + minutos
            End If
        Next i
    End If

    ' 5. Construir Encabezados
    wsMatriz.Range("A3").Value = "FECHA / OPERARIA"
    wsMatriz.Range("A3").Font.Bold = True
    Dim col As Long
    col = 2
    For Each opId In colOps
        wsMatriz.Cells(3, col).Value = dictOp(opId)(0) & " (" & dictOp(opId)(1) & " min)"
        wsMatriz.Cells(3, col).Font.Bold = True
        wsMatriz.Cells(3, col).Interior.Color = RGB(220, 220, 220)
        col = col + 1
    Next opId

    ' 6. Llenar Matriz
    Dim fila As Long
    fila = 4
    For f = fechaInicio To fechaFin
        wsMatriz.Cells(fila, 1).Value = f
        wsMatriz.Cells(fila, 1).NumberFormat = "ddd dd/mm/yyyy"
        
        col = 2
        For Each opId In colOps
            Dim keyAsig As String
            keyAsig = Format(f, "yyyy-mm-dd") & "|" & opId
            Dim minAsignados As Double, capDiaria As Double
            minAsignados = IIf(dictAsig.Exists(keyAsig), dictAsig(keyAsig), 0)
            capDiaria = dictOp(opId)(1)
            
            If capDiaria > 0 Then
                Dim pct As Double
                pct = minAsignados / capDiaria
                wsMatriz.Cells(fila, col).Value = minAsignados & " / " & capDiaria & " min (" & Format(pct, "0%") & ")"
                ' Formato condicional manual
                If pct > 1 Then
                    wsMatriz.Cells(fila, col).Font.Color = RGB(255, 0, 0)
                    wsMatriz.Cells(fila, col).Font.Bold = True
                ElseIf pct >= 0.9 Then
                    wsMatriz.Cells(fila, col).Font.Color = RGB(255, 150, 0)
                ElseIf pct > 0 Then
                    wsMatriz.Cells(fila, col).Font.Color = RGB(0, 128, 0)
                End If
            Else
                wsMatriz.Cells(fila, col).Value = minAsignados & " min"
            End If
            col = col + 1
        Next opId
        fila = fila + 1
    Next f

    ' 7. Fila de Totales
    wsMatriz.Cells(fila, 1).Value = "TOTALES (Asig / Disp)"
    wsMatriz.Cells(fila, 1).Font.Bold = True
    Dim numDias As Long
    numDias = (fechaFin - fechaInicio) + 1
    
    col = 2
    For Each opId In colOps
        Dim totalAsig As Double, totalDisp As Double
        ' Recorrer fechas para sumar solo laborales si se quisiera, 
        ' pero aqui sumaremos linealmente segun el rango
        totalDisp = dictOp(opId)(1) * numDias
        totalAsig = 0
        For f = fechaInicio To fechaFin
            keyAsig = Format(f, "yyyy-mm-dd") & "|" & opId
            totalAsig = totalAsig + IIf(dictAsig.Exists(keyAsig), dictAsig(keyAsig), 0)
        Next f
        
        Dim pctTotal As Double
        pctTotal = IIf(totalDisp > 0, totalAsig / totalDisp, 0)
        wsMatriz.Cells(fila, col).Value = totalAsig & " / " & totalDisp & " (" & Format(pctTotal, "0%") & ")"
        wsMatriz.Cells(fila, col).Font.Bold = True
        wsMatriz.Cells(fila, col).Borders(xlEdgeTop).LineStyle = xlContinuous
        col = col + 1
    Next opId

    wsMatriz.Columns.AutoFit
    wsMatriz.Activate
    
    Application.Calculation = xlCalculationAutomatic
    Application.ScreenUpdating = True
    MsgBox "Reporte Matriz de Asignacion generado exitosamente.", vbInformation
End Sub
