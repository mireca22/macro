' FUNCIONES GENERALES DE VALIDACION Y FORMATEO
Public Function IsNumericValue(valor As Variant) As Boolean
    On Error Resume Next
    IsNumericValue = IsNumeric(valor)
    On Error GoTo 0
End Function

Public Function ConvertirANumero(valor As Variant) As Long
    On Error Resume Next
    ConvertirANumero = CLng(valor)
    On Error GoTo 0
End Function

Public Function FormatearSegundosAMinutos(segundos As Long) As String
    Dim minutos As Long, segs As Long
    minutos = segundos \ 60
    segs    = segundos Mod 60
    FormatearSegundosAMinutos = Format(minutos, "00") & ":" & Format(segs, "00")
End Function

Public Function ObtenerDiaDelaSemana(fecha As Date) As String
    Dim dias As Variant
    dias = Split("Domingo,Lunes,Martes,Miercoles,Jueves,Viernes,Sabado", ",")
    ObtenerDiaDelaSemana = dias(Weekday(fecha) - 1)
End Function

Public Function CalcularPorcentajeCompletado(completadas As Long, total As Long) As String
    If total = 0 Then
        CalcularPorcentajeCompletado = "0%"
    Else
        CalcularPorcentajeCompletado = Format(completadas / total, "0%")
    End If
End Function

Public Sub LimpiarControles(frm As Object)
    Dim ctrl As Control
    For Each ctrl In frm.Controls
        Select Case TypeName(ctrl)
            Case "TextBox":    ctrl.Value = ""
            Case "ComboBox":   ctrl.ListIndex = -1: ctrl.Value = ""
            Case "CheckBox":   ctrl.Value = False
            Case "SpinButton": ctrl.Value = 0
        End Select
    Next ctrl
End Sub

Public Function ObtenerProximoColorFila(fila As Long) As Long
    If fila Mod 2 = 0 Then
        ObtenerProximoColorFila = RGB(240, 240, 240)
    Else
        ObtenerProximoColorFila = RGB(255, 255, 255)
    End If
End Function

' FORMATEO Y PRESENTACION VISUAL
Public Sub AjustarAnchoColumnasInteligente(ws As Worksheet)
    Dim col     As Range
    Dim maxAncho As Double
    maxAncho = 50
    On Error Resume Next
    ws.UsedRange.Columns.AutoFit
    For Each col In ws.UsedRange.Columns
        If col.ColumnWidth > maxAncho Then
            col.ColumnWidth = maxAncho
            col.WrapText    = True
        Else
            col.WrapText = False
        End If
    Next col
    Dim lastRow As Long
    lastRow = ws.UsedRange.Rows.Count + ws.UsedRange.Row - 1
    If lastRow > 3 Then ws.Rows("3:" & lastRow).EntireRow.AutoFit
    On Error GoTo 0
End Sub

Public Sub AplicarFormatoAsignaciones()
    Dim ws      As Worksheet
    Dim fila    As Long, ultFila As Long
    Set ws = ThisWorkbook.Sheets("ASIGNACIONES_DIARIAS")
    Application.ScreenUpdating = False
    Call AjustarAnchoColumnasInteligente(ws)
    ultFila = ws.Cells(ws.Rows.Count, 1).End(xlUp).Row
    For fila = 4 To ultFila
        ws.Rows(fila).Interior.Color = ObtenerProximoColorFila(fila)
    Next fila
    ws.Columns("B:C").HorizontalAlignment = xlCenter
    ws.Columns("E:E").HorizontalAlignment = xlCenter
    ws.Columns("G:H").HorizontalAlignment = xlCenter
    ws.Columns("J:K").HorizontalAlignment = xlCenter
    ws.Columns("N:O").HorizontalAlignment = xlCenter
    ws.Columns("L:M").NumberFormat = "#,0"
    Application.ScreenUpdating = True
    MsgBox "Formato aplicado.", vbInformation
End Sub

' ESTADISTICAS
Public Function ObtenerEstadisticasHoy() As Variant
    Dim ws          As Worksheet
    Dim fila        As Long
    Dim stats(1 To 4) As Variant
    Dim totalMin    As Double
    Dim completadas As Long, pendientes As Long, contador As Long
    Set ws = ThisWorkbook.Sheets("ASIGNACIONES_DIARIAS")
    For fila = 4 To ws.Cells(ws.Rows.Count, 1).End(xlUp).Row
        If Int(ws.Cells(fila, 1).Value) = Int(Date) Then
            contador  = contador + 1
            totalMin  = totalMin + ws.Cells(fila, 14).Value
            If ws.Cells(fila, 15).Value = "Completada" Then
                completadas = completadas + 1
            Else
                pendientes = pendientes + 1
            End If
        End If
    Next fila
    stats(1) = contador
    stats(2) = CLng(totalMin)
    stats(3) = completadas
    stats(4) = pendientes
    ObtenerEstadisticasHoy = stats
End Function

' ==============================================================================
' GENERADOR PRINCIPAL DE REPORTES
' CORRECCIÓN v3.2.2: parametro opcional activarHoja (default False)
'   False = llamada desde formulario/modulo -> NO cambia hoja activa
'   True  = llamada manual desde dashboard  -> SI activa RESUMENDIARIO
' ==============================================================================
Public Sub GenerarReporteEnHoja(fecha As Date, tipoReporte As String, _
    Optional activarHoja As Boolean = False)

    Dim wsResumen   As Worksheet, wsAsig As Worksheet, wsOp As Worksheet
    Dim dictOp      As Object
    Dim i           As Long, ultFila As Long
    Dim idOp        As String, nomOp As String
    Dim cap         As Double, asig As Double
    Dim fechaFila   As Date
    Dim esMes       As Boolean
    Dim diasLaborales As Integer

    Set wsResumen = ThisWorkbook.Sheets("RESUMENDIARIO")
    Set wsAsig    = ThisWorkbook.Sheets("ASIGNACIONES_DIARIAS")
    Set wsOp      = ThisWorkbook.Sheets("OPERARIAS")
    Set dictOp    = CreateObject("Scripting.Dictionary")

    esMes         = (tipoReporte = "MENSUAL")
    diasLaborales = 22

    Application.ScreenUpdating = False
    Application.EnableEvents   = False

    ' 1. Cargar Operarias
    ultFila = wsOp.Cells(wsOp.Rows.Count, 1).End(xlUp).Row
    For i = 4 To ultFila
        idOp  = wsOp.Cells(i, 1).Value
        nomOp = wsOp.Cells(i, 2).Value
        cap   = Val(wsOp.Cells(i, 3).Value)
        If esMes Then cap = cap * diasLaborales
        If idOp <> "" Then dictOp(idOp) = Array(nomOp, cap, 0, 0, 0)
    Next i

    ' 2. Leer ASIGNACIONES_DIARIAS en array
    Dim dataArr   As Variant
    Dim incluir   As Boolean
    ultFila = wsAsig.Cells(wsAsig.Rows.Count, 1).End(xlUp).Row

    If ultFila >= 4 Then
        dataArr = wsAsig.Range(wsAsig.Cells(4, 1), wsAsig.Cells(ultFila, 21)).Value
        For i = 1 To UBound(dataArr, 1)
            If IsDate(dataArr(i, 1)) Then
                fechaFila = dataArr(i, 1)
                incluir   = False
                If esMes Then
                    If Month(fechaFila) = Month(fecha) And Year(fechaFila) = Year(fecha) Then incluir = True
                Else
                    If Int(fechaFila) = Int(fecha) Then incluir = True
                End If

                If incluir Then
                    idOp = dataArr(i, 2)
                    Dim minAsig As Double, minReal As Double
                    minAsig = Val(dataArr(i, 14))
                    minReal = Val(dataArr(i, 20))
                    If dictOp.Exists(idOp) Then
                        Dim arrOp As Variant
                        arrOp    = dictOp(idOp)
                        arrOp(2) = arrOp(2) + minAsig
                        If dataArr(i, 15) = "Completada" Then arrOp(3) = arrOp(3) + 1
                        arrOp(4)    = arrOp(4) + minReal
                        dictOp(idOp) = arrOp
                    End If
                End If
            End If
        Next i
    End If

    ' 3. Escribir reporte — headers fila 3, datos fila 4
    wsResumen.Cells.Clear
    With wsResumen.Range("A3:F3")
        .Value = Array("Operaria", "Capacidad min", "Asignado min", "Ocupacion %", "Disponible min", "Estado")
        .Font.Bold     = True
        .Interior.Color = RGB(220, 220, 220)
        .Borders.LineStyle = xlContinuous
    End With

    Dim fila As Long, k As Variant
    fila = 4
    For Each k In dictOp.Keys
        Dim d       As Variant
        Dim ocu     As Double, dispon As Double, estado As String
        d      = dictOp(k)
        ocu    = IIf(d(1) > 0, d(2) / d(1), 0)
        dispon = d(1) - d(2)
        If dispon > 0 Then
            estado = "DISPONIBLE"
        ElseIf ocu <= 1 Then
            estado = "OPTIMA"
        Else
            estado = "SOBRECARGA"
        End If
        wsResumen.Cells(fila, 1).Value = d(0)
        wsResumen.Cells(fila, 2).Value = d(1)
        wsResumen.Cells(fila, 3).Value = d(2)
        wsResumen.Cells(fila, 4).Value = Format(ocu, "0%")
        wsResumen.Cells(fila, 5).Value = dispon
        wsResumen.Cells(fila, 6).Value = estado
        Select Case estado
            Case "DISPONIBLE", "OPTIMA":   wsResumen.Cells(fila, 6).Interior.Color = RGB(0, 176, 80)
            Case "SOBRECARGA":             wsResumen.Cells(fila, 6).Interior.Color = RGB(255, 0, 0)
        End Select
        fila = fila + 1
    Next k

    wsResumen.Columns.AutoFit

    ' CORRECCIÓN v3.2.2: solo activa la hoja si se pide explicitamente
    If activarHoja Then wsResumen.Activate

    Application.EnableEvents   = True
    Application.ScreenUpdating = True
End Sub

' REPORTE EFICIENCIA
Public Sub GenerarReporteEficiencia()
    Dim wsReporte   As Worksheet, wsAsig As Worksheet
    Dim dictPerf    As Object
    Dim i           As Long, ultFila As Long
    Dim fechaInicio As Date, fechaFin As Date
    Dim fechaStr    As String

    fechaStr = InputBox("Fecha de INICIO del reporte (dd/mm/yyyy):", "Reporte de Eficiencia", Format(Date - 30, "dd/mm/yyyy"))
    If fechaStr = "" Then Exit Sub
    If Not IsDate(fechaStr) Then Exit Sub
    fechaInicio = CDate(fechaStr)

    fechaStr = InputBox("Fecha de FIN del reporte (dd/mm/yyyy):", "Reporte de Eficiencia", Format(Date, "dd/mm/yyyy"))
    If fechaStr = "" Then Exit Sub
    If Not IsDate(fechaStr) Then Exit Sub
    fechaFin = CDate(fechaStr)

    Set wsReporte = ThisWorkbook.Sheets("ANALISISTIEMPOS")
    Set wsAsig    = ThisWorkbook.Sheets("ASIGNACIONES_DIARIAS")
    Set dictPerf  = CreateObject("Scripting.Dictionary")

    Application.ScreenUpdating = False
    Application.EnableEvents   = False

    ultFila = wsAsig.Cells(wsAsig.Rows.Count, 1).End(xlUp).Row
    If ultFila >= 4 Then
        Dim dataArr As Variant
        dataArr = wsAsig.Range(wsAsig.Cells(4, 1), wsAsig.Cells(ultFila, 21)).Value
        For i = 1 To UBound(dataArr, 1)
            If IsDate(dataArr(i, 1)) Then
                Dim fechaAsig As Date
                fechaAsig = dataArr(i, 1)
                If fechaAsig >= fechaInicio And fechaAsig <= fechaFin Then
                    Dim idOp2 As String, nomOp2 As String
                    Dim idRef2 As String, nomRef2 As String
                    Dim numTarea2 As String, nomTarea2 As String, key2 As String
                    idOp2    = dataArr(i, 2):  nomOp2   = dataArr(i, 3)
                    idRef2   = dataArr(i, 4):  nomRef2  = dataArr(i, 5)
                    numTarea2 = dataArr(i, 9): nomTarea2 = dataArr(i, 10)
                    key2     = idOp2 & "|" & idRef2 & "|" & numTarea2
                    Dim minAsig2 As Double, unidComp2 As Long, tiempoUnit2 As Long, minProd2 As Double
                    minAsig2   = Val(dataArr(i, 14))
                    unidComp2  = CLng(Val(dataArr(i, 16)))
                    tiempoUnit2 = CLng(Val(dataArr(i, 12)))
                    minProd2   = unidComp2 * tiempoUnit2 / 60
                    If Not dictPerf.Exists(key2) Then dictPerf(key2) = Array(nomOp2, nomRef2, nomTarea2, 0, 0)
                    Dim arrP As Variant
                    arrP    = dictPerf(key2)
                    arrP(3) = arrP(3) + minAsig2
                    arrP(4) = arrP(4) + minProd2
                    dictPerf(key2) = arrP
                End If
            End If
        Next i
    End If

    wsReporte.Cells.Clear
    With wsReporte.Range("A3:G3")
        .Value = Array("Operaria", "Referencia", "Tarea", "Min. Asignados", "Min. Producidos Est.", "Desviacion min", "Eficiencia")
        .Font.Bold = True: .Interior.Color = RGB(220, 220, 220): .Borders.LineStyle = xlContinuous
    End With

    Dim fila2 As Long, k2 As Variant
    fila2 = 4
    For Each k2 In dictPerf.Keys
        Dim d2 As Variant
        d2 = dictPerf(k2)
        wsReporte.Cells(fila2, 1).Value = d2(0)
        wsReporte.Cells(fila2, 2).Value = d2(1)
        wsReporte.Cells(fila2, 3).Value = d2(2)
        wsReporte.Cells(fila2, 4).Value = d2(3)
        wsReporte.Cells(fila2, 5).Value = d2(4)
        wsReporte.Cells(fila2, 6).Value = d2(4) - d2(3)
        If d2(3) > 0 Then wsReporte.Cells(fila2, 7).Value = d2(4) / d2(3)
        fila2 = fila2 + 1
    Next k2

    wsReporte.Range("G4:G" & fila2).NumberFormat = "0.0%"
    wsReporte.Columns.AutoFit
    wsReporte.Activate
    Application.EnableEvents   = True
    Application.ScreenUpdating = True
    MsgBox "Reporte de Eficiencia generado con " & dictPerf.Count & " registros.", vbInformation
End Sub

' AVANCE PRODUCCION
Public Sub GenerarReporteAvanceProduccion()
    Dim wsAvance    As Worksheet, wsLotes As Worksheet
    Dim wsTareas    As Worksheet, wsAsig  As Worksheet
    Dim dictUnidades As Object, dictAsignado As Object
    Dim i           As Long, ultFila As Long
    Dim idRef3      As String, key3 As String, cant3 As Long

    Set wsAvance    = ThisWorkbook.Sheets("AVANCEPRODUCCION")
    Set wsLotes     = ThisWorkbook.Sheets("LOTES")
    Set wsTareas    = ThisWorkbook.Sheets("TAREAS")
    Set wsAsig      = ThisWorkbook.Sheets("ASIGNACIONES_DIARIAS")
    Set dictUnidades = CreateObject("Scripting.Dictionary")
    Set dictAsignado = CreateObject("Scripting.Dictionary")

    Application.ScreenUpdating = False

    ultFila = wsLotes.Cells(wsLotes.Rows.Count, 1).End(xlUp).Row
    For i = 4 To ultFila
        idRef3 = wsLotes.Cells(i, 2).Value
        cant3  = wsLotes.Cells(i, 5).Value
        If dictUnidades.Exists(idRef3) Then
            dictUnidades(idRef3) = dictUnidades(idRef3) + cant3
        Else
            dictUnidades(idRef3) = cant3
        End If
    Next i

    ultFila = wsAsig.Cells(wsAsig.Rows.Count, 1).End(xlUp).Row
    For i = 4 To ultFila
        idRef3 = wsAsig.Cells(i, 4).Value
        Dim numT3 As String
        numT3  = wsAsig.Cells(i, 9).Value
        cant3  = wsAsig.Cells(i, 11).Value
        key3   = idRef3 & "|" & numT3
        If dictAsignado.Exists(key3) Then
            dictAsignado(key3) = dictAsignado(key3) + cant3
        Else
            dictAsignado(key3) = cant3
        End If
    Next i

    wsAvance.Cells.Clear
    With wsAvance.Range("A3:I3")
        .Value = Array("ID Referencia", "Nombre Referencia", "Nombre Tarea", "Tiempo Unit seg", _
                       "Total Unidades Req.", "Unidades Asignadas", "Unidades Pendientes", _
                       "Tiempo Pendiente min", "Estado Asignacion")
        .Font.Bold = True: .Interior.Color = RGB(220, 220, 220): .Borders.LineStyle = xlContinuous
    End With

    Dim filaR3 As Long
    filaR3  = 4
    ultFila = wsTareas.Cells(wsTareas.Rows.Count, 1).End(xlUp).Row
    For i = 4 To ultFila
        idRef3 = wsTareas.Cells(i, 1).Value
        Dim numT4 As String, nomT4 As String, tiempoT4 As Long
        numT4   = wsTareas.Cells(i, 2).Value
        nomT4   = wsTareas.Cells(i, 3).Value
        tiempoT4 = wsTareas.Cells(i, 4).Value
        Dim totalReq4 As Long, totalAsig4 As Long, pendiente4 As Long
        totalReq4  = IIf(dictUnidades.Exists(idRef3), dictUnidades(idRef3), 0)
        key3       = idRef3 & "|" & numT4
        totalAsig4 = IIf(dictAsignado.Exists(key3), dictAsignado(key3), 0)
        pendiente4 = IIf(totalReq4 - totalAsig4 > 0, totalReq4 - totalAsig4, 0)
        Dim tiempoPend4 As Double, estado4 As String
        tiempoPend4 = pendiente4 * tiempoT4 / 60
        If totalReq4 = 0 Then
            estado4 = "Sin Tallas"
        ElseIf totalAsig4 >= totalReq4 Then
            estado4 = "Completo"
        ElseIf totalAsig4 = 0 Then
            estado4 = "En Curso"
        Else
            estado4 = "Pendiente"
        End If
        wsAvance.Range("A" & filaR3 & ":I" & filaR3).Value = Array(idRef3, _
            ObtenerNombreReferencia(idRef3), nomT4, tiempoT4, totalReq4, totalAsig4, pendiente4, tiempoPend4, estado4)
        filaR3 = filaR3 + 1
    Next i

    wsAvance.Columns.AutoFit
    wsAvance.Activate
    Application.ScreenUpdating = True
    MsgBox "Reporte de Avance de Produccion generado.", vbInformation
End Sub

' REPORTE LOTES
Public Sub GenerarReporteLotes()
    Dim wsReporte   As Worksheet, wsLotes As Worksheet
    Dim wsAsig      As Worksheet, wsTareas As Worksheet, wsRef As Worksheet
    Dim dictRefTasks As Object, dictRefTime As Object, dictOpsDone As Object
    Dim i           As Long, ultFila As Long
    Dim idRef5      As String, idLote5 As String, color5 As String, talla5 As String, key5 As String
    Dim cantLote5   As Long, tasksCount5 As Long, timeUnit5 As Long
    Dim totalOpsReq5 As Long, opsDone5 As Long
    Dim avance5     As Double, tiempoHoras5 As Double, estado5 As String

    Set wsReporte   = ThisWorkbook.Sheets("REPORTELOTES")
    Set wsLotes     = ThisWorkbook.Sheets("LOTES")
    Set wsAsig      = ThisWorkbook.Sheets("ASIGNACIONES_DIARIAS")
    Set wsTareas    = ThisWorkbook.Sheets("TAREAS")
    Set wsRef       = ThisWorkbook.Sheets("REFERENCIAS")
    Set dictRefTasks = CreateObject("Scripting.Dictionary")
    Set dictRefTime  = CreateObject("Scripting.Dictionary")
    Set dictOpsDone  = CreateObject("Scripting.Dictionary")

    Application.ScreenUpdating = False

    ultFila = wsTareas.Cells(wsTareas.Rows.Count, 1).End(xlUp).Row
    For i = 4 To ultFila
        idRef5 = wsTareas.Cells(i, 1).Value
        If idRef5 <> "" Then
            If dictRefTasks.Exists(idRef5) Then
                dictRefTasks(idRef5) = dictRefTasks(idRef5) + 1
            Else
                dictRefTasks(idRef5) = 1
            End If
        End If
    Next i

    ultFila = wsRef.Cells(wsRef.Rows.Count, 1).End(xlUp).Row
    For i = 4 To ultFila
        idRef5 = wsRef.Cells(i, 1).Value
        If idRef5 <> "" Then dictRefTime(idRef5) = Val(wsRef.Cells(i, 5).Value)
    Next i

    ultFila = wsAsig.Cells(wsAsig.Rows.Count, 1).End(xlUp).Row
    For i = 4 To ultFila
        idRef5  = wsAsig.Cells(i, 4).Value
        idLote5 = wsAsig.Cells(i, 6).Value
        color5  = wsAsig.Cells(i, 7).Value
        talla5  = wsAsig.Cells(i, 8).Value
        Dim comp5 As Long
        comp5 = Val(wsAsig.Cells(i, 16).Value)
        key5  = idLote5 & "|" & idRef5 & "|" & color5 & "|" & talla5
        If dictOpsDone.Exists(key5) Then
            dictOpsDone(key5) = dictOpsDone(key5) + comp5
        Else
            dictOpsDone(key5) = comp5
        End If
    Next i

    wsReporte.Cells.Clear
    With wsReporte.Range("A3:K3")
        .Value = Array("Lote", "Referencia", "Color", "Talla", "Cantidad", "TareasUnidad", _
                       "Ops. Requeridas", "Ops. Completadas", "Avance %", "Tiempo Est. Horas", "Estado")
        .Font.Bold = True: .Interior.Color = RGB(220, 220, 220): .Borders.LineStyle = xlContinuous
    End With

    Dim filaR5 As Long
    filaR5  = 4
    ultFila = wsLotes.Cells(wsLotes.Rows.Count, 1).End(xlUp).Row
    For i = 4 To ultFila
        idLote5  = wsLotes.Cells(i, 1).Value
        idRef5   = wsLotes.Cells(i, 2).Value
        color5   = wsLotes.Cells(i, 3).Value
        talla5   = wsLotes.Cells(i, 4).Value
        cantLote5 = Val(wsLotes.Cells(i, 5).Value)
        If idLote5 <> "" Then
            tasksCount5  = IIf(dictRefTasks.Exists(idRef5), dictRefTasks(idRef5), 0)
            timeUnit5    = IIf(dictRefTime.Exists(idRef5),  dictRefTime(idRef5),  0)
            totalOpsReq5 = cantLote5 * tasksCount5
            key5         = idLote5 & "|" & idRef5 & "|" & color5 & "|" & talla5
            opsDone5     = IIf(dictOpsDone.Exists(key5), dictOpsDone(key5), 0)
            avance5      = IIf(totalOpsReq5 > 0, opsDone5 / totalOpsReq5, 0)
            tiempoHoras5 = cantLote5 * timeUnit5 / 3600
            If avance5 >= 1 Then
                estado5 = "COMPLETADO"
                wsReporte.Cells(filaR5, 11).Font.Color = RGB(0, 150, 0)
            ElseIf avance5 > 0 Then
                estado5 = "EN PROCESO"
                wsReporte.Cells(filaR5, 11).Font.Color = RGB(255, 140, 0)
            Else
                estado5 = "PENDIENTE"
            End If
            wsReporte.Range("A" & filaR5 & ":K" & filaR5).Value = Array(idLote5, idRef5, color5, talla5, cantLote5, _
                tasksCount5, totalOpsReq5, opsDone5, avance5, Round(tiempoHoras5, 2), estado5)
            filaR5 = filaR5 + 1
        End If
    Next i

    wsReporte.Range("I4:I" & filaR5).NumberFormat = "0.0%"
    wsReporte.Columns.AutoFit
    wsReporte.Activate
    Application.ScreenUpdating = True
    MsgBox "Reporte de Lotes generado exitosamente.", vbInformation
End Sub

    Dim wsProd      As Worksheet, wsLotes As Worksheet
    Dim dictAvance  As Object

    Dim idRef6      As String

    Set wsProd     = ThisWorkbook.Sheets("PRODUCCIONFINALIZADA")
    Set wsLotes    = ThisWorkbook.Sheets("LOTES")
    Set dictAvance = CreateObject("Scripting.Dictionary")

    Application.ScreenUpdating = False

    ultFila = wsLotes.Cells(wsLotes.Rows.Count, 1).End(xlUp).Row
    For i = 4 To ultFila
        idRef6 = wsLotes.Cells(i, 2).Value
        If idRef6 <> "" Then
            If Not dictAvance.Exists(idRef6) Then
                dictAvance(idRef6) = Array(ObtenerNombreReferencia(idRef6), 0, 0, 0, 0)
            End If
            Dim arrRef6 As Variant
            arrRef6    = dictAvance(idRef6)
            arrRef6(1) = arrRef6(1) + Val(wsLotes.Cells(i, 5).Value)
            arrRef6(2) = arrRef6(2) + Val(wsLotes.Cells(i, 8).Value)
            arrRef6(3) = arrRef6(3) + 1
            dictAvance(idRef6) = arrRef6
        End If
    Next i

    wsProd.Cells.Clear
    With wsProd.Range("A3:F3")
        .Value = Array("Fecha Cierre", "ID Referencia", "Nombre Referencia", "Total Unidades", "Unidades Completadas", "Porcentaje")
        .Font.Bold = True: .Interior.Color = RGB(220, 220, 220): .Borders.LineStyle = xlContinuous
    End With

    Dim filaI6 As Long, k6 As Variant
    filaI6 = 4
    For Each k6 In dictAvance.Keys
        Dim d6 As Variant
        d6 = dictAvance(k6)
        Dim pct6 As Double
        pct6 = IIf(d6(1) > 0, d6(2) / d6(1), 0)
        wsProd.Range("A" & filaI6 & ":F" & filaI6).Value = Array(Date, k6, d6(0), d6(1), d6(2), Format(pct6, "0%"))
        filaI6 = filaI6 + 1
    Next k6

    wsProd.Columns.AutoFit
    wsProd.Activate
    Application.ScreenUpdating = True
    MsgBox "Informe de Produccion generado.", vbInformation
End Sub

' PARAMETROS Y FECHAS
Public Function ObtenerParametro(nombreParam As String) As Variant
    Dim ws      As Worksheet
    Dim celda   As Range
    On Error Resume Next
    Set ws    = ThisWorkbook.Sheets("PARAMETROS")
    Set celda = ws.Columns("A").Find(nombreParam, LookIn:=xlValues, LookAt:=xlWhole)
    If Not celda Is Nothing Then
        ObtenerParametro = celda.Offset(0, 1).Value
    Else
        ObtenerParametro = 0
    End If
    On Error GoTo 0
End Function

Public Function CalcularMinutosDiariosBase() As Long
    Dim horas As Double, dias As Double
    horas = ObtenerParametro("HorasSemanalesLegales")
    dias  = ObtenerParametro("DiasLaboralesSemana")
    If horas = 0 Then horas = 44
    If dias  = 0 Then dias  = 6
    CalcularMinutosDiariosBase = CLng(horas * 60 / dias)
End Function

Public Function GetCapacidadDiariaTotal() As Long
    Dim wsOp    As Worksheet
    Dim i       As Long, ultFila As Long
    Dim capTotal As Long
    Set wsOp  = ThisWorkbook.Sheets("OPERARIAS")
    ultFila   = wsOp.Cells(wsOp.Rows.Count, 1).End(xlUp).Row
    For i = 4 To ultFila
        If wsOp.Cells(i, 4).Value = "Activa" Then
            capTotal = capTotal + Val(wsOp.Cells(i, 3).Value)
        End If
    Next i
    GetCapacidadDiariaTotal = capTotal
End Function

Public Function CalcularFechaFinEstimada(idRef As String, cantidadTotal As Long, fechaInicio As Date) As Date
    Dim tiempoUnitTotalSeg  As Long
    Dim minutosRequeridos   As Double
    Dim capacidadDiariaMin  As Long
    Dim diasNecesarios      As Double
    Dim fechaFin            As Date
    Dim diasContados        As Integer

    tiempoUnitTotalSeg = ObtenerTiempoTotalUnidad(idRef)
    If tiempoUnitTotalSeg = 0 Then
        MsgBox "La referencia " & idRef & " no tiene tareas o tiempos definidos.", vbExclamation
        CalcularFechaFinEstimada = 0
        Exit Function
    End If

    minutosRequeridos  = cantidadTotal * tiempoUnitTotalSeg / 60
    capacidadDiariaMin = GetCapacidadDiariaTotal()
    If capacidadDiariaMin = 0 Then capacidadDiariaMin = 480

    diasNecesarios = Application.WorksheetFunction.RoundUp(minutosRequeridos / capacidadDiariaMin, 0)
    fechaFin       = fechaInicio
    diasContados   = 0

    Do While diasContados < diasNecesarios
        fechaFin = fechaFin + 1
        If EsDiaLaboral(fechaFin) Then diasContados = diasContados + 1
        If diasContados > 1000 Then Exit Do
    Loop

    CalcularFechaFinEstimada = fechaFin
End Function
