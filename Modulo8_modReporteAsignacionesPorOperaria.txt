'==============================================================================
' MODULO 8 - modReporteAsignacionesPorOperaria v1.4
' Propósito: Generar reporte imprimible de asignaciones por operaria y fecha
' MEJORAS v1.4:
'   - Corrección de errores al eliminar controles inexistentes
'   - Manejo seguro de creación de hoja y controles
'   - Indicador de carga (spinner visual y mensaje)
'   - Formato de operaria: "ID - Nombre"
'==============================================================================

'==============================================================================
' SUB: GenerarReporteAsignacionesPorOperaria
' Propósito: Crea o actualiza la hoja "REPORTE_ASIGNACIONES_OPERARIA" con las
'            asignaciones filtradas por operaria y fecha seleccionada.
'==============================================================================
Public Sub GenerarReporteAsignacionesPorOperaria(Optional fechaSeleccionada As Variant = Null, Optional operariaSeleccionada As String = "")
    Dim wsReporte As Worksheet
    Dim wsAsig As Worksheet
    Dim ultFila As Long
    Dim i As Long, filaDestino As Long
    Dim data As Variant
    Dim fechaFiltro As Date
    Dim idOperariaFiltro As String
    Dim hayDatos As Boolean
    Dim nombreOperaria As String
    
    Application.ScreenUpdating = False
    
    ' Validar o crear la hoja de reporte
    On Error Resume Next
    Set wsReporte = ThisWorkbook.Sheets("REPORTE_ASIGNACIONES_OPERARIA")
    If wsReporte Is Nothing Then
        Set wsReporte = ThisWorkbook.Sheets.Add(After:=ThisWorkbook.Sheets(ThisWorkbook.Sheets.Count))
        wsReporte.Name = "REPORTE_ASIGNACIONES_OPERARIA"
    End If
    On Error GoTo 0
    
    ' Limpiar contenido anterior (desde fila 7 en adelante)
    If wsReporte.Rows.Count > 6 Then
        wsReporte.Rows("7:" & wsReporte.Rows.Count).ClearContents
        wsReporte.Rows("7:" & wsReporte.Rows.Count).Interior.ColorIndex = xlNone
    End If
    
    ' Configurar encabezados de la hoja
    wsReporte.Range("A1").Value = "REPORTE DE ASIGNACIONES POR OPERARIA"
    wsReporte.Range("A1").Font.Size = 18
    wsReporte.Range("A1").Font.Bold = True
    
    wsReporte.Range("A3").Value = "Fecha:"
    wsReporte.Range("A3").Font.Bold = True
    wsReporte.Range("B3").Value = "Seleccione una fecha"
    
    wsReporte.Range("A4").Value = "Operaria:"
    wsReporte.Range("A4").Font.Bold = True
    wsReporte.Range("B4").Value = "Seleccione una operaria"
    
    wsReporte.Range("A6").Value = "ID_Lote"
    wsReporte.Range("B6").Value = "Nombre_Tarea"
    wsReporte.Range("C6").Value = "Cantidad_Unidades_Asignadas"
    wsReporte.Range("D6").Value = "Tiempo_Total_Asignado_Minutos"
    
    ' Formato de encabezados de tabla
    With wsReporte.Range("A6:D6")
        .Font.Bold = True
        .Interior.Color = RGB(0, 122, 204)
        .Font.Color = RGB(255, 255, 255)
        .Borders(xlEdgeBottom).LineStyle = xlContinuous
        .Borders(xlEdgeBottom).Weight = xlThick
    End With
    
    ' Ajustar ancho de columnas
    wsReporte.Columns("A").ColumnWidth = 25
    wsReporte.Columns("B").ColumnWidth = 40
    wsReporte.Columns("C").ColumnWidth = 25
    wsReporte.Columns("D").ColumnWidth = 30
    
    ' Configurar controles de la hoja (siempre se configuran al abrir)
    Call ConfigurarControlesReporte(wsReporte)
    
    ' Si no se proporcionaron filtros, salir
    If IsNull(fechaSeleccionada) And operariaSeleccionada = "" Then
        wsReporte.Activate
        Application.ScreenUpdating = True
        Exit Sub
    End If
    
    ' Aplicar filtros
    fechaFiltro = CDate(fechaSeleccionada)
    idOperariaFiltro = ExtraerIDOperaria(operariaSeleccionada)
    nombreOperaria = ObtenerNombreOperaria(idOperariaFiltro)
    
    ' Actualizar los controles con los valores seleccionados
    Call ActualizarControlesConSeleccion(wsReporte, fechaFiltro, idOperariaFiltro, nombreOperaria)
    
    Set wsAsig = ThisWorkbook.Sheets("ASIGNACIONES_DIARIAS")
    ultFila = wsAsig.Cells(wsAsig.Rows.Count, 1).End(xlUp).Row
    
    If ultFila < 4 Then
        wsReporte.Range("A7").Value = "No hay asignaciones registradas en el sistema."
        wsReporte.Activate
        Application.ScreenUpdating = True
        Exit Sub
    End If
    
    ' Leer datos de asignaciones
    data = wsAsig.Range(wsAsig.Cells(4, 1), wsAsig.Cells(ultFila, 14)).Value
    
    filaDestino = 7
    hayDatos = False
    
    For i = 1 To UBound(data, 1)
        If IsDate(data(i, 1)) Then
            If Int(data(i, 1)) = Int(fechaFiltro) And Trim(CStr(data(i, 2))) = idOperariaFiltro Then
                wsReporte.Cells(filaDestino, 1).Value = data(i, 6)
                wsReporte.Cells(filaDestino, 2).Value = data(i, 10)
                wsReporte.Cells(filaDestino, 3).Value = data(i, 11)
                wsReporte.Cells(filaDestino, 4).Value = data(i, 14)
                
                wsReporte.Cells(filaDestino, 3).NumberFormat = "#,##0"
                wsReporte.Cells(filaDestino, 4).NumberFormat = "#,##0.00"
                
                filaDestino = filaDestino + 1
                hayDatos = True
            End If
        End If
    Next i
    
    If Not hayDatos Then
        wsReporte.Range("A7").Value = "No hay asignaciones para la operaria y fecha seleccionadas."
    Else
        ' Agregar fila de totales
        wsReporte.Cells(filaDestino, 2).Value = "TOTALES:"
        wsReporte.Cells(filaDestino, 2).Font.Bold = True
        wsReporte.Cells(filaDestino, 3).Formula = "=SUM(C7:C" & filaDestino - 1 & ")"
        wsReporte.Cells(filaDestino, 4).Formula = "=SUM(D7:D" & filaDestino - 1 & ")"
        wsReporte.Cells(filaDestino, 3).NumberFormat = "#,##0"
        wsReporte.Cells(filaDestino, 4).NumberFormat = "#,##0.00"
        
        With wsReporte.Range("A7:D" & filaDestino)
            .Borders(xlInsideHorizontal).LineStyle = xlContinuous
            .Borders(xlInsideHorizontal).Weight = xlThin
            .Borders(xlEdgeBottom).LineStyle = xlContinuous
            .Borders(xlEdgeBottom).Weight = xlThin
        End With
        
        Dim j As Long
        For j = 7 To filaDestino - 1
            If j Mod 2 = 0 Then
                wsReporte.Range("A" & j & ":D" & j).Interior.Color = RGB(240, 240, 240)
            End If
        Next j
        
        wsReporte.PageSetup.PrintArea = "$A$1:$D$" & filaDestino
        wsReporte.PageSetup.Orientation = xlPortrait
        wsReporte.PageSetup.FitToPagesWide = 1
        wsReporte.PageSetup.FitToPagesTall = False
        wsReporte.PageSetup.CenterHeader = "Asignaciones de: " & nombreOperaria & " - Fecha: " & Format(fechaFiltro, "dd/mm/yyyy")
    End If
    
    ' Ocultar indicador de carga si está visible
    Call OcultarIndicadorCarga(wsReporte)
    
    wsReporte.Activate
    Application.ScreenUpdating = True
    
    If hayDatos Then
        MsgBox "Reporte generado correctamente. " & (filaDestino - 7) & " asignaciones encontradas.", vbInformation
    End If
End Sub

'==============================================================================
' SUB: ConfigurarControlesReporte
' Propósito: Agrega o actualiza los controles desplegables en la hoja de reporte
'==============================================================================
Private Sub ConfigurarControlesReporte(ws As Worksheet)
    Dim oleObj As OLEObject
    Dim colFechas As Collection
    Dim cboFecha As Object
    Dim cboOperaria As Object
    Dim btnGenerar As Object
    Dim btnImprimir As Object
    Dim btnRefrescar As Object
    Dim lblMensaje As Object
    Dim fecha As Variant
    
    ' Eliminar controles existentes (con manejo de errores)
    On Error Resume Next
    If ObjectExists(ws, "cboFechaReporte") Then ws.OLEObjects("cboFechaReporte").Delete
    If ObjectExists(ws, "cboOperariaReporte") Then ws.OLEObjects("cboOperariaReporte").Delete
    If ObjectExists(ws, "btnGenerarReporte") Then ws.OLEObjects("btnGenerarReporte").Delete
    If ObjectExists(ws, "btnImprimirReporte") Then ws.OLEObjects("btnImprimirReporte").Delete
    If ObjectExists(ws, "lblMensajeCarga") Then ws.OLEObjects("lblMensajeCarga").Delete
    If ObjectExists(ws, "btnRefrescarOperarias") Then ws.OLEObjects("btnRefrescarOperarias").Delete
    On Error GoTo 0
    
    ' Obtener lista de fechas
    Set colFechas = ObtenerFechasConAsignaciones()
    
    ' Crear combo de fechas
    Set oleObj = ws.OLEObjects.Add(ClassType:="Forms.ComboBox.1", _
                                   Left:=ws.Cells(3, 2).Left, _
                                   Top:=ws.Cells(3, 2).Top, _
                                   Width:=120, Height:=22)
    oleObj.Name = "cboFechaReporte"
    Set cboFecha = oleObj.Object
    
    cboFecha.Clear
    For Each fecha In colFechas
        cboFecha.AddItem Format(fecha, "dd/mm/yyyy")
    Next fecha
    If cboFecha.ListCount > 0 Then cboFecha.ListIndex = 0
    
    ' Crear combo de operarias
    Set oleObj = ws.OLEObjects.Add(ClassType:="Forms.ComboBox.1", _
                                   Left:=ws.Cells(4, 2).Left, _
                                   Top:=ws.Cells(4, 2).Top, _
                                   Width:=200, Height:=22)
    oleObj.Name = "cboOperariaReporte"
    Set cboOperaria = oleObj.Object
    cboOperaria.Clear
    
    ' Crear boton GENERAR REPORTE
    Set oleObj = ws.OLEObjects.Add(ClassType:="Forms.CommandButton.1", _
                                   Left:=ws.Cells(3, 4).Left, _
                                   Top:=ws.Cells(3, 4).Top, _
                                   Width:=130, Height:=28)
    oleObj.Name = "btnGenerarReporte"
    Set btnGenerar = oleObj.Object
    btnGenerar.Caption = "GENERAR REPORTE"
    btnGenerar.BackColor = RGB(0, 122, 204)
    btnGenerar.ForeColor = RGB(255, 255, 255)
    
    ' Crear boton IMPRIMIR
    Set oleObj = ws.OLEObjects.Add(ClassType:="Forms.CommandButton.1", _
                                   Left:=ws.Cells(4, 4).Left, _
                                   Top:=ws.Cells(4, 4).Top, _
                                   Width:=130, Height:=28)
    oleObj.Name = "btnImprimirReporte"
    Set btnImprimir = oleObj.Object
    btnImprimir.Caption = "IMPRIMIR"
    btnImprimir.BackColor = RGB(44, 62, 80)
    btnImprimir.ForeColor = RGB(255, 255, 255)
    
    ' Crear boton REFRESCAR OPERARIAS
    Set oleObj = ws.OLEObjects.Add(ClassType:="Forms.CommandButton.1", _
                                   Left:=ws.Cells(3, 3).Left + 10, _
                                   Top:=ws.Cells(3, 2).Top, _
                                   Width:=140, Height:=22)
    oleObj.Name = "btnRefrescarOperarias"
    Set btnRefrescar = oleObj.Object
    btnRefrescar.Caption = "REFRESCAR OPERARIAS"
    btnRefrescar.BackColor = RGB(255, 140, 0)
    btnRefrescar.ForeColor = RGB(255, 255, 255)
    
    ' Crear etiqueta para mensaje de carga (inicialmente invisible)
    Set oleObj = ws.OLEObjects.Add(ClassType:="Forms.Label.1", _
                                   Left:=ws.Cells(4, 3).Left + 10, _
                                   Top:=ws.Cells(4, 2).Top, _
                                   Width:=150, Height:=22)
    oleObj.Name = "lblMensajeCarga"
    Set lblMensaje = oleObj.Object
    lblMensaje.Caption = "Cargando operarias..."
    lblMensaje.ForeColor = RGB(0, 112, 192)
    lblMensaje.Font.Bold = True
    lblMensaje.Visible = False

    ' Cargar operarias para la fecha seleccionada
    If cboFecha.ListCount > 0 Then
        Call CargarOperariasPorFecha(ws, CDate(cboFecha.Value))
    End If
End Sub

'==============================================================================
' FUNCIÓN: ObjectExists
' Propósito: Verifica si un OLEObject existe en una hoja
'==============================================================================
Private Function ObjectExists(ws As Worksheet, nombre As String) As Boolean
    On Error Resume Next
    ObjectExists = Not ws.OLEObjects(nombre) Is Nothing
    On Error GoTo 0
End Function

'==============================================================================
' SUB: MostrarIndicadorCarga
' Propósito: Muestra un indicador visual de carga
'==============================================================================
Private Sub MostrarIndicadorCarga(ws As Worksheet)
    Dim lbl As Object
    Dim cboOperaria As Object
    
    On Error Resume Next
    Set lbl = ws.OLEObjects("lblMensajeCarga").Object
    Set cboOperaria = ws.OLEObjects("cboOperariaReporte").Object
    
    If Not lbl Is Nothing Then
        lbl.Visible = True
        lbl.Caption = "Cargando operarias..."
        lbl.ForeColor = RGB(0, 112, 192)
    End If
    
    If Not cboOperaria Is Nothing Then
        cboOperaria.Clear
        cboOperaria.AddItem "Cargando, espere..."
        cboOperaria.Enabled = False
    End If
    
    ' Forzar actualización de pantalla
    DoEvents
    Application.ScreenUpdating = True
    
    On Error GoTo 0
End Sub

'==============================================================================
' SUB: OcultarIndicadorCarga
' Propósito: Oculta el indicador de carga
'==============================================================================
Private Sub OcultarIndicadorCarga(ws As Worksheet)
    Dim lbl As Object
    
    On Error Resume Next
    Set lbl = ws.OLEObjects("lblMensajeCarga").Object
    If Not lbl Is Nothing Then
        lbl.Visible = False
    End If
    On Error GoTo 0
End Sub

'==============================================================================
' SUB: CargarOperariasPorFecha
' Propósito: Actualiza el combo de operarias con las que tienen asignaciones
'            en la fecha especificada
'==============================================================================
Public Sub CargarOperariasPorFecha(ws As Worksheet, fecha As Date)
    Dim cboOperaria As Object
    Dim colOperarias As Collection
    Dim operaria As Variant
    
    ' Mostrar indicador de carga
    Call MostrarIndicadorCarga(ws)
    
    ' Pequeña pausa para que el usuario vea el mensaje
    Application.Wait Now + TimeValue("00:00:01")
    
    On Error Resume Next
    Set cboOperaria = ws.OLEObjects("cboOperariaReporte").Object
    If cboOperaria Is Nothing Then
        Call OcultarIndicadorCarga(ws)
        Exit Sub
    End If
    On Error GoTo 0
    
    ' Obtener operarias con asignaciones en la fecha especificada
    Set colOperarias = ObtenerOperariasPorFecha(fecha)
    
    ' Llenar combo
    cboOperaria.Clear
    For Each operaria In colOperarias
        cboOperaria.AddItem operaria
    Next operaria
    
    If cboOperaria.ListCount > 0 Then
        cboOperaria.ListIndex = 0
        cboOperaria.Enabled = True
    Else
        cboOperaria.AddItem "No hay operarias con asignaciones en esta fecha"
        cboOperaria.Enabled = True
    End If
    
    ' Ocultar indicador de carga
    Call OcultarIndicadorCarga(ws)
End Sub

'==============================================================================
' FUNCIÓN: ObtenerOperariasPorFecha
' Propósito: Retorna una colección de operarias (formato "ID - Nombre") que tienen
'            asignaciones en la fecha especificada
'==============================================================================
Public Function ObtenerOperariasPorFecha(fecha As Date) As Collection
    Dim ws As Worksheet
    Dim resultado As New Collection
    Dim ultFila As Long
    Dim i As Long
    Dim data As Variant
    Dim idOperaria As String
    Dim nombreOperaria As String
    Dim itemExiste As Boolean
    Dim item As Variant
    
    Set ws = ThisWorkbook.Sheets("ASIGNACIONES_DIARIAS")
    ultFila = ws.Cells(ws.Rows.Count, 1).End(xlUp).Row
    
    If ultFila < 4 Then
        Set ObtenerOperariasPorFecha = resultado
        Exit Function
    End If
    
    data = ws.Range(ws.Cells(4, 1), ws.Cells(ultFila, 3)).Value
    
    For i = 1 To UBound(data, 1)
        If IsDate(data(i, 1)) Then
            If Int(data(i, 1)) = Int(fecha) Then
                idOperaria = Trim(CStr(data(i, 2)))
                nombreOperaria = Trim(CStr(data(i, 3)))
                
                If idOperaria <> "" And nombreOperaria <> "" Then
                    itemExiste = False
                    For Each item In resultado
                        If item = idOperaria & " - " & nombreOperaria Then
                            itemExiste = True
                            Exit For
                        End If
                    Next item
                    
                    If Not itemExiste Then
                        resultado.Add idOperaria & " - " & nombreOperaria
                    End If
                End If
            End If
        End If
    Next i
    
    Set ObtenerOperariasPorFecha = resultado
End Function

'==============================================================================
' FUNCIÓN: ObtenerFechasConAsignaciones
' Propósito: Retorna una colección de fechas únicas que tienen asignaciones
'==============================================================================
Public Function ObtenerFechasConAsignaciones() As Collection
    Dim ws As Worksheet
    Dim colFechas As New Collection
    Dim ultFila As Long
    Dim i As Long
    Dim data As Variant
    Dim fechaActual As Date
    Dim fechaExiste As Boolean
    Dim item As Variant
    
    Set ws = ThisWorkbook.Sheets("ASIGNACIONES_DIARIAS")
    ultFila = ws.Cells(ws.Rows.Count, 1).End(xlUp).Row
    
    If ultFila >= 4 Then
        data = ws.Range(ws.Cells(4, 1), ws.Cells(ultFila, 1)).Value
        For i = 1 To UBound(data, 1)
            If IsDate(data(i, 1)) Then
                fechaActual = Int(CDate(data(i, 1)))
                fechaExiste = False
                For Each item In colFechas
                    If item = fechaActual Then
                        fechaExiste = True
                        Exit For
                    End If
                Next item
                If Not fechaExiste Then
                    colFechas.Add fechaActual
                End If
            End If
        Next i
    End If
    
    Set ObtenerFechasConAsignaciones = OrdenarCollectionFechas(colFechas, True)
End Function

'==============================================================================
' FUNCIÓN: OrdenarCollectionFechas
' Propósito: Ordena una colección de fechas ascendente o descendentemente
'==============================================================================
Private Function OrdenarCollectionFechas(col As Collection, Optional descendente As Boolean = False) As Collection
    Dim resultado As New Collection
    Dim i As Long, j As Long
    Dim tempArray() As Variant
    Dim temp As Variant
    
    If col.Count = 0 Then
        Set OrdenarCollectionFechas = resultado
        Exit Function
    End If
    
    ReDim tempArray(1 To col.Count)
    For i = 1 To col.Count
        tempArray(i) = col(i)
    Next i
    
    For i = 1 To col.Count - 1
        For j = i + 1 To col.Count
            If descendente Then
                If tempArray(i) < tempArray(j) Then
                    temp = tempArray(i)
                    tempArray(i) = tempArray(j)
                    tempArray(j) = temp
                End If
            Else
                If tempArray(i) > tempArray(j) Then
                    temp = tempArray(i)
                    tempArray(i) = tempArray(j)
                    tempArray(j) = temp
                End If
            End If
        Next j
    Next i
    
    For i = 1 To col.Count
        resultado.Add tempArray(i)
    Next i
    
    Set OrdenarCollectionFechas = resultado
End Function

'==============================================================================
' FUNCIÓN: ExtraerIDOperaria
' Propósito: Extrae solo el ID de una cadena "ID - Nombre"
'==============================================================================
Private Function ExtraerIDOperaria(valor As String) As String
    If InStr(valor, " - ") > 0 Then
        ExtraerIDOperaria = Trim(Split(valor, " - ")(0))
    Else
        ExtraerIDOperaria = Trim(valor)
    End If
End Function

'==============================================================================
' SUB: ActualizarControlesConSeleccion
' Propósito: Sincroniza los controles con los valores seleccionados
'==============================================================================
Private Sub ActualizarControlesConSeleccion(ws As Worksheet, fecha As Date, idOperaria As String, nombreOperaria As String)
    Dim cboFecha As Object
    Dim cboOperaria As Object
    Dim i As Integer
    
    On Error Resume Next
    Set cboFecha = ws.OLEObjects("cboFechaReporte").Object
    Set cboOperaria = ws.OLEObjects("cboOperariaReporte").Object
    
    If Not cboFecha Is Nothing Then
        For i = 0 To cboFecha.ListCount - 1
            If CDate(cboFecha.List(i)) = fecha Then
                cboFecha.ListIndex = i
                Exit For
            End If
        Next i
    End If
    
    If Not cboOperaria Is Nothing Then
        Call CargarOperariasPorFecha(ws, fecha)
        
        For i = 0 To cboOperaria.ListCount - 1
            If cboOperaria.List(i) = idOperaria & " - " & nombreOperaria Then
                cboOperaria.ListIndex = i
                Exit For
            End If
        Next i
    End If
    On Error GoTo 0
End Sub

'==============================================================================
' SUB: EjecutarGenerarReporte
' Propósito: Macro asociada al botón GENERAR REPORTE
'==============================================================================
Public Sub EjecutarGenerarReporte()
    Dim ws As Worksheet
    Dim cboFecha As Object
    Dim cboOperaria As Object
    Dim fechaSeleccionada As String
    Dim operariaSeleccionada As String
    
    On Error Resume Next
    Set ws = ThisWorkbook.Sheets("REPORTE_ASIGNACIONES_OPERARIA")
    If ws Is Nothing Then
        MsgBox "La hoja de reporte no existe. Ejecute GenerarReporteAsignacionesPorOperaria primero.", vbCritical
        Exit Sub
    End If
    
    Set cboFecha = ws.OLEObjects("cboFechaReporte").Object
    Set cboOperaria = ws.OLEObjects("cboOperariaReporte").Object
    
    If cboFecha Is Nothing Or cboOperaria Is Nothing Then
        MsgBox "Controles no encontrados. Reintente.", vbCritical
        Exit Sub
    End If
    
    fechaSeleccionada = cboFecha.Value
    operariaSeleccionada = cboOperaria.Value
    
    If fechaSeleccionada = "" Then
        MsgBox "Seleccione una fecha antes de generar el reporte.", vbExclamation
        Exit Sub
    End If
    
    If operariaSeleccionada = "" Or operariaSeleccionada = "No hay operarias con asignaciones en esta fecha" Or operariaSeleccionada = "Cargando, espere..." Then
        MsgBox "No hay operarias con asignaciones en la fecha seleccionada.", vbExclamation
        Exit Sub
    End If
    
    On Error GoTo 0
    
    Call GenerarReporteAsignacionesPorOperaria(CDate(fechaSeleccionada), operariaSeleccionada)
End Sub

'==============================================================================
' SUB: EjecutarImprimirReporte
' Propósito: Macro asociada al botón IMPRIMIR
'==============================================================================
Public Sub EjecutarImprimirReporte()
    Dim ws As Worksheet
    
    On Error Resume Next
    Set ws = ThisWorkbook.Sheets("REPORTE_ASIGNACIONES_OPERARIA")
    If ws Is Nothing Then
        MsgBox "La hoja de reporte no existe. Genere un reporte primero.", vbCritical
        Exit Sub
    End If
    
    If ws.Cells(7, 1).Value = "" Then
        MsgBox "No hay datos para imprimir. Genere un reporte primero.", vbExclamation
        Exit Sub
    End If
    
    On Error GoTo 0
    
    ws.PrintOut Preview:=True
End Sub

'==============================================================================
' SUB: EjecutarRefrescarOperarias
' Propósito: Macro asociada al botón REFRESCAR OPERARIAS
'==============================================================================
Public Sub EjecutarRefrescarOperarias()
    Dim ws As Worksheet
    Dim cboFecha As Object
    Dim fechaSeleccionada As String
    
    On Error Resume Next
    Set ws = ThisWorkbook.Sheets("REPORTE_ASIGNACIONES_OPERARIA")
    If ws Is Nothing Then Exit Sub
    
    Set cboFecha = ws.OLEObjects("cboFechaReporte").Object
    If cboFecha Is Nothing Then Exit Sub
    
    fechaSeleccionada = cboFecha.Value
    If fechaSeleccionada = "" Then
        MsgBox "Seleccione una fecha primero.", vbExclamation
        Exit Sub
    End If
    
    On Error GoTo 0
    
    Call CargarOperariasPorFecha(ws, CDate(fechaSeleccionada))
End Sub
