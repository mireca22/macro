Option Explicit

'==============================================================================
' MODULO 8 - modReporteAsignacionesPorOperaria v1.9 (ESTANDARIZADO)
' Propósito: Generar reporte imprimible de asignaciones por operaria y fecha
' CAMBIOS v1.9:
'   - Fecha: Ahora es un campo de texto libre para digitar (txtFechaReporte).
'   - Operarias: Lista a todas las operarias activas del sistema (Caché).
'==============================================================================

'==============================================================================
' SUB: GenerarReporteAsignacionesPorOperaria
' Propósito: Crea o actualiza la hoja "REPORTE_ASIGNACIONES_OPERARIA" con las
'            asignaciones filtradas por operaria y fecha seleccionada.
'==============================================================================
Public Sub GenerarReporteAsignacionesPorOperaria(Optional fechaSeleccionada As Variant = Null, Optional operariaSeleccionada As String = "")
    Dim wsReporte As Worksheet
    Dim wsAsig As Worksheet
    Dim ultFila As Long
    Dim i As Long, filaDestino As Long
    Dim data As Variant
    Dim fechaFiltro As Date
    Dim idOperariaFiltro As String
    Dim hayDatos As Boolean
    Dim nombreOperaria As String
    
    Application.ScreenUpdating = False
    
    ' Validar o crear la hoja de reporte
    On Error Resume Next
    Set wsReporte = ThisWorkbook.Sheets("REPORTE_ASIGNACIONES_OPERARIA")
    If wsReporte Is Nothing Then
        Set wsReporte = ThisWorkbook.Sheets.Add(After:=ThisWorkbook.Sheets(ThisWorkbook.Sheets.Count))
        wsReporte.Name = "REPORTE_ASIGNACIONES_OPERARIA"
    End If
    On Error GoTo 0
    
    ' Limpiar contenido y formato anterior completamente (desde fila 7 en adelante)
    ' Esto elimina bordes, colores y datos de reportes previos que pudieran ser más largos.
    wsReporte.Rows("7:" & wsReporte.Rows.Count).Clear
    
    ' Configurar encabezados de la hoja
    wsReporte.Range("A1").Value = "REPORTE DE ASIGNACIONES POR OPERARIA"
    wsReporte.Range("A1").Font.Size = 18
    wsReporte.Range("A1").Font.Bold = True
    
    wsReporte.Range("A3").Value = "Fecha (dd/mm/yyyy):"
    wsReporte.Range("A3").Font.Bold = True
    wsReporte.Range("B3").Value = ""
    
    wsReporte.Range("A4").Value = "Operaria:"
    wsReporte.Range("A4").Font.Bold = True
    wsReporte.Range("B4").Value = ""
    
    wsReporte.Range("A6").Value = "ID_Lote"
    wsReporte.Range("B6").Value = "Nombre_Tarea"
    wsReporte.Range("C6").Value = "Cantidad_Unidades_Asignadas"
    wsReporte.Range("D6").Value = "Tiempo_Total_Asignado_Minutos"
    
    ' Formato de encabezados de tabla
    With wsReporte.Range("A6:D6")
        .Font.Bold = True
        .Interior.Color = RGB(0, 122, 204)
        .Font.Color = RGB(255, 255, 255)
        .Borders(xlEdgeBottom).LineStyle = xlContinuous
        .Borders(xlEdgeBottom).Weight = xlThick
    End With
    
    
    ' Configurar controles de la hoja
    Call ConfigurarControlesReporte(wsReporte)
    
    ' Si no se proporcionaron filtros, salir
    If IsNull(fechaSeleccionada) And operariaSeleccionada = "" Then
        wsReporte.Activate
        Application.ScreenUpdating = True
        Exit Sub
    End If
    
    ' Aplicar filtros
    fechaFiltro = CDate(fechaSeleccionada)
    idOperariaFiltro = ExtraerIDOperaria(operariaSeleccionada)
    nombreOperaria = ObtenerNombreOperaria(idOperariaFiltro)
    
    ' Actualizar los controles con los valores seleccionados
    Call ActualizarControlesConSeleccion(wsReporte, fechaFiltro, idOperariaFiltro, nombreOperaria)
    
    Set wsAsig = ThisWorkbook.Sheets("ASIGNACIONES_DIARIAS")
    ultFila = wsAsig.Cells(wsAsig.Rows.Count, 1).End(xlUp).Row
    
    If ultFila < 4 Then
        wsReporte.Range("A7").Value = "No hay asignaciones registradas en el sistema."
        wsReporte.Activate
        Application.ScreenUpdating = True
        Exit Sub
    End If
    
    ' Leer datos de asignaciones
    data = wsAsig.Range(wsAsig.Cells(4, 1), wsAsig.Cells(ultFila, 14)).Value
    
    filaDestino = 7
    hayDatos = False
    
    For i = 1 To UBound(data, 1)
        If IsDate(data(i, 1)) Then
            If Int(data(i, 1)) = Int(fechaFiltro) And Trim(CStr(data(i, 2))) = idOperariaFiltro Then
                wsReporte.Cells(filaDestino, 1).Value = data(i, 6)
                wsReporte.Cells(filaDestino, 2).Value = data(i, 10)
                wsReporte.Cells(filaDestino, 3).Value = data(i, 11)
                wsReporte.Cells(filaDestino, 4).Value = data(i, 14)
                
                wsReporte.Cells(filaDestino, 3).NumberFormat = "#,##0"
                wsReporte.Cells(filaDestino, 4).NumberFormat = "#,##0.00"
                
                filaDestino = filaDestino + 1
                hayDatos = True
            End If
        End If
    Next i
    
    If Not hayDatos Then
        wsReporte.Range("A7").Value = "No hay asignaciones para la operaria y fecha seleccionadas."
    Else
        ' Agregar fila de totales
        wsReporte.Cells(filaDestino, 2).Value = "TOTALES:"
        wsReporte.Cells(filaDestino, 2).Font.Bold = True
        wsReporte.Cells(filaDestino, 3).Formula = "=SUM(C7:C" & filaDestino - 1 & ")"
        wsReporte.Cells(filaDestino, 4).Formula = "=SUM(D7:D" & filaDestino - 1 & ")"
        wsReporte.Cells(filaDestino, 3).NumberFormat = "#,##0"
        wsReporte.Cells(filaDestino, 4).NumberFormat = "#,##0.00"
        
        With wsReporte.Range("A7:D" & filaDestino)
            .Borders(xlInsideHorizontal).LineStyle = xlContinuous
            .Borders(xlInsideHorizontal).Weight = xlThin
            .Borders(xlEdgeBottom).LineStyle = xlContinuous
            .Borders(xlEdgeBottom).Weight = xlThin
        End With
        
        Dim j As Long
        For j = 7 To filaDestino - 1
            If j Mod 2 = 0 Then
                wsReporte.Range("A" & j & ":D" & j).Interior.Color = RGB(240, 240, 240)
            End If
        Next j
        
        ' Ajustar anchos de columna dinámicamente según el contenido
        Call AjustarAnchoColumnasInteligente(wsReporte)
        
        wsReporte.PageSetup.PrintArea = "$A$1:$D$" & filaDestino
        wsReporte.PageSetup.Orientation = xlPortrait
        wsReporte.PageSetup.PaperSize = xlPaperLetter ' Configurar a tamaño CARTA
        wsReporte.PageSetup.FitToPagesWide = 1
        wsReporte.PageSetup.FitToPagesTall = 1 ' Ajustar a una sola hoja (alto y ancho)
        wsReporte.PageSetup.CenterHeader = "Asignaciones de: " & nombreOperaria & " - Fecha: " & Format(fechaFiltro, "dd/mm/yyyy")
        wsReporte.PageSetup.Zoom = False ' Necesario para que FitToPages funcione
    End If
    
    wsReporte.Activate
    Application.ScreenUpdating = True
    
    If hayDatos Then
        MsgBox "Reporte generado correctamente. " & (filaDestino - 7) & " asignaciones encontradas.", vbInformation
    End If
End Sub

'==============================================================================
' SUB: ConfigurarControlesReporte
' Propósito: Agrega o actualiza los controles desplegables en la hoja de reporte
'==============================================================================
Private Sub ConfigurarControlesReporte(ws As Worksheet)
    Dim colOps As Collection, opId As Variant
    Dim txtFecha As Object, cboOperaria As Object, obj As OLEObject, oleObj As OLEObject

    ' --- LIMPIEZA SEGURA DE CONTROLES ---
    ws.Unprotect "admin123" ' Asegurar que la hoja no esté bloqueada
    Application.EnableEvents = False
    On Error Resume Next
    ' Borrar OLEObjects (Inputs y posibles botones obsoletos)
    For Each obj In ws.OLEObjects
        Select Case obj.Name
            Case "txtFechaReporte", "cboOperariaReporte", "btnGenerarReporte", "btnImprimirReporte"
                obj.Delete
        End Select
    Next obj
    ' Borrar Shapes (Botones nuevos)
    Call LimpiarBotonesReporte(ws)
    On Error GoTo 0
    DoEvents ' Pausa para procesar la eliminación de objetos en memoria
    Application.EnableEvents = True
    
    ' --- CREACIÓN DE CONTROLES (Patrón Robusto) ---
    
    ' 1. Crear campo de fecha (TextBox) para que el usuario digite
    Set oleObj = ws.OLEObjects.Add(ClassType:="Forms.TextBox.1", _
                                   Left:=ws.Cells(3, 2).Left, _
                                   Top:=ws.Cells(3, 2).Top, _
                                   Width:=120, Height:=22)
    oleObj.Name = "txtFechaReporte"
    With oleObj.Object
        .Value = Format(Date, "dd/mm/yyyy")
        .TextAlign = 2 ' Centrado
    End With
    
    ' 2. Crear combo de operarias
    Set oleObj = ws.OLEObjects.Add(ClassType:="Forms.ComboBox.1", _
                                   Left:=ws.Cells(4, 2).Left, _
                                   Top:=ws.Cells(4, 2).Top, _
                                   Width:=200, Height:=22)
    oleObj.Name = "cboOperariaReporte"
    Set cboOperaria = oleObj.Object
    cboOperaria.Clear
    ' Llenar con todas las operarias activas del sistema (Independiente de la fecha)
    Set colOps = ObtenerAllOperarias(True)
    If Not colOps Is Nothing Then
        For Each opId In colOps
            cboOperaria.AddItem opId & " - " & ObtenerNombreOperaria(CStr(opId))
        Next opId
    End If

    ' 3. Crear botones usando el patrón del Dashboard (Modulo 1)
    Call CrearBotonReporte(ws, "GENERAR REPORTE", "EjecutarGenerarReporte", 3, 4, RGB(0, 122, 204))
    Call CrearBotonReporte(ws, "IMPRIMIR", "EjecutarImprimirReporte", 4, 4, RGB(44, 62, 80))
End Sub

'==============================================================================
' SUB: CrearBotonReporte (Basado en Modulo 1)
' Propósito: Crea botones uniformes como Formas para asegurar ejecución de macro.
'==============================================================================
Private Sub CrearBotonReporte(ws As Worksheet, texto As String, macro As String, fila As Integer, col As Integer, color As Long)
    Dim btn As Shape
    ' Altura de fila para que el botón quepa bien
    ws.Rows(fila).RowHeight = 30
    
    Set btn = ws.Shapes.AddShape(msoShapeRoundedRectangle, ws.Cells(fila, col).Left + 5, ws.Cells(fila, col).Top + 2, 130, 26)
    With btn
        .Name = "btn_" & macro ' Nombre único para limpieza
        .TextFrame.Characters.Text = texto
        .TextFrame.HorizontalAlignment = xlHAlignCenter
        .TextFrame.VerticalAlignment = xlVAlignCenter
        .Fill.ForeColor.RGB = color
        .Line.Visible = msoFalse
        .TextFrame.Characters.Font.Color = RGB(255, 255, 255)
        .TextFrame.Characters.Font.Bold = True
        .OnAction = macro
        .Shadow.Type = msoShadow25
    End With
End Sub

'==============================================================================
' SUB: LimpiarBotonesReporte
' Propósito: Elimina solo las formas que funcionan como botones.
'==============================================================================
Private Sub LimpiarBotonesReporte(ws As Worksheet)
    Dim shp As Shape
    For Each shp In ws.Shapes
        If shp.Type <> msoOLEControlObject Then
            If Left(shp.Name, 4) = "btn_" Or shp.Name = "btnGenerarReporte" Or shp.Name = "btnImprimirReporte" Then
                shp.Delete
            End If
        End If
    Next shp
End Sub

'==============================================================================
' SUB: ActualizarControlesConSeleccion (Modificado)
'==============================================================================
Private Sub ActualizarControlesConSeleccion(ws As Worksheet, fecha As Date, idOperaria As String, nombreOperaria As String)
    Dim txtFecha As Object
    Dim cboOperaria As Object
    Dim i As Integer
    
    On Error Resume Next
    Set txtFecha = ws.OLEObjects("txtFechaReporte").Object
    Set cboOperaria = ws.OLEObjects("cboOperariaReporte").Object
    
    If Not txtFecha Is Nothing Then
        txtFecha.Value = Format(fecha, "dd/mm/yyyy")
    End If
    
    If Not cboOperaria Is Nothing Then
        For i = 0 To cboOperaria.ListCount - 1
            If cboOperaria.List(i) = idOperaria & " - " & nombreOperaria Then
                cboOperaria.ListIndex = i
                Exit For
            End If
        Next i
    End If
    On Error GoTo 0
End Sub

'==============================================================================
' FUNCIÓN: ExtraerIDOperaria
' Propósito: Extrae solo el ID de una cadena "ID - Nombre"
'==============================================================================
Private Function ExtraerIDOperaria(valor As String) As String
    If InStr(valor, " - ") > 0 Then
        ExtraerIDOperaria = Trim(Split(valor, " - ")(0))
    Else
        ExtraerIDOperaria = Trim(valor)
    End If
End Function

'==============================================================================
' SUB: ActualizarControlesConSeleccion
' Propósito: Sincroniza los controles con los valores seleccionados
'==============================================================================
' SUB: EjecutarGenerarReporte
' Propósito: Macro asociada al botón GENERAR REPORTE
'==============================================================================
Public Sub EjecutarGenerarReporte()
    Dim ws As Worksheet
    Dim txtFecha As Object
    Dim cboOperaria As Object
    Dim fechaSeleccionada As String
    Dim operariaSeleccionada As String
    
    On Error Resume Next
    Set ws = ThisWorkbook.Sheets("REPORTE_ASIGNACIONES_OPERARIA")
    If ws Is Nothing Then
        MsgBox "La hoja de reporte no existe. Ejecute GenerarReporteAsignacionesPorOperaria primero.", vbCritical
        Exit Sub
    End If
    
    Set txtFecha = ws.OLEObjects("txtFechaReporte").Object
    Set cboOperaria = ws.OLEObjects("cboOperariaReporte").Object
    
    If txtFecha Is Nothing Or cboOperaria Is Nothing Then
        MsgBox "Controles no encontrados. Reintente.", vbCritical
        Exit Sub
    End If
    
    fechaSeleccionada = txtFecha.Value
    operariaSeleccionada = cboOperaria.Value
    
    If Not IsDate(fechaSeleccionada) Then
        MsgBox "Ingrese una fecha válida (dd/mm/yyyy).", vbExclamation
        Exit Sub
    End If
    
    If operariaSeleccionada = "" Then
        MsgBox "Seleccione una operaria de la lista.", vbExclamation
        Exit Sub
    End If
    
    On Error GoTo 0
    
    Call GenerarReporteAsignacionesPorOperaria(CDate(fechaSeleccionada), operariaSeleccionada)
End Sub

'==============================================================================
' SUB: EjecutarImprimirReporte
' Propósito: Macro asociada al botón IMPRIMIR
'==============================================================================
Public Sub EjecutarImprimirReporte()
    Dim ws As Worksheet
    
    On Error Resume Next
    Set ws = ThisWorkbook.Sheets("REPORTE_ASIGNACIONES_OPERARIA")
    If ws Is Nothing Then
        MsgBox "La hoja de reporte no existe. Genere un reporte primero.", vbCritical
        Exit Sub
    End If
    
    If ws.Cells(7, 1).Value = "" Then
        MsgBox "No hay datos para imprimir. Genere un reporte primero.", vbExclamation
        Exit Sub
    End If
    
    On Error GoTo 0
    
    ws.PrintOut Preview:=True
End Sub
